36 std::pair<double, double>
37 calculate_pressure_drop(const DoFHandler<dim> &dof_handler,
38  std::shared_ptr<Mapping<dim>> mapping,
39  const VectorType &evaluation_point,
40  const Quadrature<dim> &cell_quadrature_formula,

42  const unsigned int inlet_boundary_id,
43  const unsigned int outlet_boundary_id)

87  {
88  if (cell->face(face_id)->at_boundary())

112  velocity_values[q];
113  }

115  {
116  inlet_surface += temp_surface;
117  dynamic_pressure_inlet_boundary +=
118  temp_dynamic_pressure_at_boundary;
119  static_pressure_inlet_boundary +=
120  temp_static_pressure_at_boundary;
121  }
122  if (boundary_id == outlet_boundary_id)
123  {

126  temp_dynamic_pressure_at_boundary;

128  temp_static_pressure_at_boundary;
129  }
130  }
131  }

150  static_pressure_outlet_boundary / outlet_surface;
151  static_pressure_inlet_boundary =
152  static_pressure_inlet_boundary / inlet_surface;

154  static_pressure_inlet_boundary - static_pressure_outlet_boundary;
155

157  dynamic_pressure_outlet_boundary / outlet_surface;
158  dynamic_pressure_inlet_boundary =
159  dynamic_pressure_inlet_boundary / inlet_surface;

226 template <int dim, typename VectorType>
227 double
228 calculate_CFL(const DoFHandler<dim> &dof_handler,
229  const VectorType &evaluation_point,
230  const double time_step,
231  const Quadrature<dim> &quadrature_formula,
232  const Mapping<dim> &mapping)
233 {
234  FESystem<dim, dim> fe = dof_handler.get_fe();
235  FEValues<dim> fe_values(mapping,
236  fe,
237  quadrature_formula,
238  update_values | update_quadrature_points |
239  update_JxW_values);
240 
241  const FEValuesExtractors::Vector velocities(0);
242  const unsigned int n_q_points = quadrature_formula.size();
243 
244 
245  std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
246 
247  // Element size
248  double h;
249 
250  // Element degree
251  double degree = double(fe.degree);
252 
253  // CFL
254  double CFL = 0;
255 
256  for (const auto &cell : dof_handler.active_cell_iterators())
257  {
258  if (cell->is_locally_owned())
259  {
260  if (dim == 2)
261  h = std::sqrt(4. * cell->measure() / M_PI) / degree;
262  else if (dim == 3)
263  h = pow(6 * cell->measure() / M_PI, 1. / 3.) / degree;
264  fe_values.reinit(cell);
265  fe_values[velocities].get_function_values(evaluation_point,
266  present_velocity_values);
267  for (unsigned int q = 0; q < n_q_points; ++q)
268  {
269  const double localCFL =
270  present_velocity_values[q].norm() / h * time_step;
271  CFL = std::max(CFL, localCFL);
272  }
273  }
274  }
275  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
276  CFL = Utilities::MPI::max(CFL, mpi_communicator);
277  return (CFL);
278 }

329 template <int dim, typename VectorType>
330 double
331 calculate_enstrophy(const DoFHandler<dim> &dof_handler,
332  const VectorType &evaluation_point,
333  const Quadrature<dim> &quadrature_formula,
334  const Mapping<dim> &mapping)
335 {
336  const FESystem<dim, dim> fe = dof_handler.get_fe();
337  FEValues<dim> fe_values(mapping,
338  fe,
339  quadrature_formula,
340  update_values | update_gradients |
341  update_quadrature_points | update_JxW_values);
342 
343  const FEValuesExtractors::Vector velocities(0);
344 
345  const unsigned int n_q_points = quadrature_formula.size();
346 
347  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
348  double en = 0.0;
349  double domain_volume =
350  GridTools::volume(dof_handler.get_triangulation(), mapping);
351 
352  for (const auto &cell : dof_handler.active_cell_iterators())
353  {
354  if (cell->is_locally_owned())
355  {
356  fe_values.reinit(cell);
357 
358  fe_values[velocities].get_function_gradients(
359  evaluation_point, present_velocity_gradients);
360 
361  for (unsigned int q = 0; q < n_q_points; q++)
362  {
363  // Find the values of gradient of ux and uy (the finite element
364  // solution) at the quadrature points
365  double ux_y = present_velocity_gradients[q][0][1];
366  double uy_x = present_velocity_gradients[q][1][0];
367 
368  if (dim == 2)
369  {
370  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
371  domain_volume;
372  }
373  else
374  {
375  double uz_y = present_velocity_gradients[q][2][1];
376  double uy_z = present_velocity_gradients[q][1][2];
377  double ux_z = present_velocity_gradients[q][0][2];
378  double uz_x = present_velocity_gradients[q][2][0];
379  en += 0.5 * (uz_y - uy_z) * (uz_y - uy_z) * fe_values.JxW(q) /
380  domain_volume;
381  en += 0.5 * (ux_z - uz_x) * (ux_z - uz_x) * fe_values.JxW(q) /
382  domain_volume;
383  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
384  domain_volume;
385  }
386  }
387  }
388  }
389  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
390  en = Utilities::MPI::sum(en, mpi_communicator);
391  return (en);
392 }

532 template <int dim, typename VectorType>
533 double
534 calculate_apparent_viscosity(const DoFHandler<dim> &dof_handler,
535  const VectorType &evaluation_point,
536  const Quadrature<dim> &quadrature_formula,
537  const Mapping<dim> &mapping,
538  PhysicalPropertiesManager &properties_manager)
539 {
540  double integral_viscosity_x_shear_rate = 0;
541  double integral_shear_rate = 0;
542  double shear_rate_magnitude;
543  Tensor<2, dim> shear_rate;
544  double kinematic_viscosity;
545  const auto rheological_model = properties_manager.get_rheology();
546

548  FEValues<dim> fe_values(mapping,
549  fe,
550  quadrature_formula,

552  update_quadrature_points | update_JxW_values);
553

555 
556  const unsigned int n_q_points = quadrature_formula.size();
557 
558  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
559 
560  for (const auto &cell : dof_handler.active_cell_iterators())
561  {
562  if (cell->is_locally_owned())
563  {
564  fe_values.reinit(cell);
565 
566  fe_values[velocities].get_function_gradients(
567  evaluation_point, present_velocity_gradients);
568 
569  for (unsigned int q = 0; q < n_q_points; q++)
570  {
571  shear_rate = present_velocity_gradients[q] +
572  transpose(present_velocity_gradients[q]);
573 
574  double shear_rate_x_velocity_gradient = 0;
575  for (int i = 0; i < dim; ++i)
576  {
577  for (int j = 0; j < dim; ++j)
578  {
579  shear_rate_x_velocity_gradient +=
580  shear_rate[i][j] * present_velocity_gradients[q][j][i];
581  }
582  }
583 
584  shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
585 
586  std::map<field, double> field_values;
587  field_values[field::shear_rate] = shear_rate_magnitude;
588 
589  kinematic_viscosity = rheological_model->value(field_values);
590 
591  integral_viscosity_x_shear_rate +=
592  kinematic_viscosity * shear_rate_x_velocity_gradient *
593  fe_values.JxW(q);

603  Utilities::MPI::sum(integral_shear_rate, mpi_communicator);

655 template <int dim, typename VectorType>
656 std::vector<std::vector<Tensor<1, dim>>>
657 calculate_forces(
658  const DoFHandler<dim> &dof_handler,
659  const VectorType &evaluation_point,
660  PhysicalPropertiesManager &properties_manager,
661  const BoundaryConditions::NSBoundaryConditions<dim> &boundary_conditions,
662  const Quadrature<dim - 1> &face_quadrature_formula,
663  const Mapping<dim> &mapping)
664 {
665  const FESystem<dim, dim> fe = dof_handler.get_fe();
666 
667  // Rheological model for viscosity properties
668  double kinematic_viscosity;
669  const auto rheological_model = properties_manager.get_rheology();
670 
671 
672  const unsigned int n_q_points = face_quadrature_formula.size();
673  const FEValuesExtractors::Vector velocities(0);
674  const FEValuesExtractors::Scalar pressure(dim);
675  std::vector<double> pressure_values(n_q_points);
676  std::vector<Tensor<2, dim>> velocity_gradients(n_q_points);
677  Tensor<1, dim> normal_vector;
678  Tensor<2, dim> shear_rate;
679  Tensor<2, dim> fluid_stress;
680  Tensor<2, dim> fluid_viscous_stress;
681  Tensor<2, dim> fluid_pressure;
682 
683  std::vector<Tensor<1, dim>> viscous_force_vector(boundary_conditions.size);
684  std::vector<Tensor<1, dim>> pressure_force_vector(boundary_conditions.size);
685  std::vector<Tensor<1, dim>> force_vector(boundary_conditions.size);
686 
687 
688  FEFaceValues<dim> fe_face_values(mapping,
689  fe,
690  face_quadrature_formula,
691  update_values | update_quadrature_points |
692  update_gradients | update_JxW_values |
693  update_normal_vectors);
694  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
695 
696  for (const auto &cell : dof_handler.active_cell_iterators())
697  {
698  if (cell->is_locally_owned())
699  {
700  if (cell->at_boundary())
701  {
702  for (const auto face : cell->face_indices())
703  {
704  if (cell->face(face)->at_boundary())
705  {
706  const auto boundary_id =
707  std::find(begin(boundary_conditions.id),
708  end(boundary_conditions.id),
709  cell->face(face)->boundary_id());
710 
711 
712  if (boundary_id != end(boundary_conditions.id))
713  {
714  unsigned int vector_index =
715  boundary_id - begin(boundary_conditions.id);
716  fe_face_values.reinit(cell, face);
717 
718  std::vector<Point<dim>> q_points =
719  fe_face_values.get_quadrature_points();
720  fe_face_values[velocities].get_function_gradients(
721  evaluation_point, velocity_gradients);
722  fe_face_values[pressure].get_function_values(
723  evaluation_point, pressure_values);
724  for (unsigned int q = 0; q < n_q_points; q++)
725  {
726  normal_vector = -fe_face_values.normal_vector(q);
727  for (int d = 0; d < dim; ++d)
728  {
729  fluid_pressure[d][d] = pressure_values[q];
730  }
731  shear_rate = velocity_gradients[q] +
732  transpose(velocity_gradients[q]);
733 
734  const double shear_rate_magnitude =
735  calculate_shear_rate_magnitude(shear_rate);
736 
737  std::map<field, double> field_values;
738  field_values[field::shear_rate] =
739  shear_rate_magnitude;
740 
741  kinematic_viscosity =
742  rheological_model->value(field_values);
743  fluid_viscous_stress =
744  -kinematic_viscosity * shear_rate;
745  fluid_stress =
746  -fluid_viscous_stress - fluid_pressure;
747 
748  viscous_force_vector[vector_index] -=
749  fluid_viscous_stress * normal_vector *
750  fe_face_values.JxW(q);
751  pressure_force_vector[vector_index] -=
752  fluid_pressure * normal_vector *
753  fe_face_values.JxW(q);
754  force_vector[vector_index] +=
755  fluid_stress * normal_vector *
756  fe_face_values.JxW(q);
757  }
758  }
759  }
760  }
761  }
762  }
763  }
764 
765  for (unsigned int i_bc = 0; i_bc < boundary_conditions.size; ++i_bc)
766  {
767  viscous_force_vector[i_bc] =
768  Utilities::MPI::sum(viscous_force_vector[i_bc], mpi_communicator);
769  pressure_force_vector[i_bc] =
770  Utilities::MPI::sum(pressure_force_vector[i_bc], mpi_communicator);
771  force_vector[i_bc] =
772  Utilities::MPI::sum(force_vector[i_bc], mpi_communicator);
773  }
774  std::vector<std::vector<Tensor<1, dim>>> forces{force_vector,
775  viscous_force_vector,
776  pressure_force_vector};
777  return forces;
778 }

879  if (cell->is_locally_owned())

881  for (const auto face : cell->face_indices())
882  {
883  if (cell->face(face)->at_boundary())
884  {
885  fe_face_values.reinit(cell, face);
886  if (cell->face(face)->boundary_id() == boundary_id)

933  distance[1] * force[0];
934  }
935  }
936  }
937  }
938  }
939  }
940  }

1000 // Find the l2 norm of the error between the finite element sol'n and the exact
1001 // sol'n for both the velocity and the pressure
1002 // Mean pressure is removed from both the analytical and the simulation solution
1003 template <int dim, typename VectorType>
1004 std::pair<double, double>
1005 calculate_L2_error(const DoFHandler<dim> &dof_handler,
1006  const VectorType &evaluation_point,
1007  const Function<dim> *exact_solution,
1008  const Quadrature<dim> &quadrature_formula,
1009  const Mapping<dim> &mapping)
1010 {
1011  const FESystem<dim, dim> fe = dof_handler.get_fe();
1012  FEValues<dim> fe_values(mapping,
1013  fe,
1014  quadrature_formula,
1015  update_values | update_gradients |
1016  update_quadrature_points | update_JxW_values);
1017 
1018  const FEValuesExtractors::Vector velocities(0);
1019  const FEValuesExtractors::Scalar pressure(dim);
1020  const unsigned int n_q_points = quadrature_formula.size();
1021 
1022  std::vector<Vector<double>> q_exactSol(n_q_points, Vector<double>(dim + 1));
1023 
1024  std::vector<Tensor<1, dim>> local_velocity_values(n_q_points);
1025  std::vector<double> local_pressure_values(n_q_points);
1026 
1027  double pressure_integral = 0;
1028  double exact_pressure_integral = 0;
1029 
1030  // loop over elements to calculate average pressure
1031  for (const auto &cell : dof_handler.active_cell_iterators())
1032  {
1033  if (cell->is_locally_owned())
1034  {
1035  fe_values.reinit(cell);
1036 
1037  fe_values[pressure].get_function_values(evaluation_point,
1038  local_pressure_values);
1039  // Get the exact solution at all gauss points
1040  exact_solution->vector_value_list(fe_values.get_quadrature_points(),
1041  q_exactSol);
1042 
1043  for (unsigned int q = 0; q < n_q_points; q++)
1044  {
1045  pressure_integral += local_pressure_values[q] * fe_values.JxW(q);
1046  exact_pressure_integral += q_exactSol[q][dim] * fe_values.JxW(q);
1047  }
1048  }
1049  }
1050  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
1051  pressure_integral = Utilities::MPI::sum(pressure_integral, mpi_communicator);
1052  exact_pressure_integral =
1053  Utilities::MPI::sum(exact_pressure_integral, mpi_communicator);
1054 
1055  // double global_volume =
1056  // GridTools::volume(dof_handler.get_triangulation(),*mapping);
1057  double global_volume =
1058  GridTools::volume(dof_handler.get_triangulation(), mapping);
1059  double average_pressure = pressure_integral / global_volume;
1060  double average_exact_pressure = exact_pressure_integral / global_volume;
1061 
1062 
1063  double l2errorU = 0.;
1064  double l2errorP = 0.;
1065 
1066  for (const auto &cell : dof_handler.active_cell_iterators())
1067  {
1068  if (cell->is_locally_owned())
1069  {
1070  fe_values.reinit(cell);
1071  fe_values[velocities].get_function_values(evaluation_point,
1072  local_velocity_values);
1073  fe_values[pressure].get_function_values(evaluation_point,
1074  local_pressure_values);
1075 
1076  // Get the exact solution at all gauss points
1077  exact_solution->vector_value_list(fe_values.get_quadrature_points(),
1078  q_exactSol);
1079 
1080  for (unsigned int q = 0; q < n_q_points; q++)
1081  {
1082  // Find the values of x and u_h (the finite element solution) at
1083  // the quadrature points
1084  double ux_sim = local_velocity_values[q][0];
1085  double ux_exact = q_exactSol[q][0];
1086 
1087  double uy_sim = local_velocity_values[q][1];
1088  double uy_exact = q_exactSol[q][1];
1089 
1090  l2errorU +=
1091  (ux_sim - ux_exact) * (ux_sim - ux_exact) * fe_values.JxW(q);
1092  l2errorU +=
1093  (uy_sim - uy_exact) * (uy_sim - uy_exact) * fe_values.JxW(q);
1094 
1095  if (dim == 3)
1096  {
1097  double uz_sim = local_velocity_values[q][2];
1098  double uz_exact = q_exactSol[q][2];
1099  l2errorU += (uz_sim - uz_exact) * (uz_sim - uz_exact) *
1100  fe_values.JxW(q);
1101  }
1102 
1103  double p_sim = local_pressure_values[q] - average_pressure;
1104  double p_exact = q_exactSol[q][dim] - average_exact_pressure;
1105  l2errorP +=
1106  (p_sim - p_exact) * (p_sim - p_exact) * fe_values.JxW(q);
1107  }
1108  }
1109  }
1110  l2errorU = Utilities::MPI::sum(l2errorU, mpi_communicator);
1111  l2errorP = Utilities::MPI::sum(l2errorP, mpi_communicator);
1112 
1113  return std::make_pair(std::sqrt(l2errorU), std::sqrt(l2errorP));
1114 }

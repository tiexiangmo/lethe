523  * @brief Return the sign of the parameter. To be removed once PR#794 is merged.
524  * @param prm parameter
525  */
526  inline double
527  sign(const double prm) const
528  {
529  if (prm > 0)
530  return 1;
531  else if (prm < 0)

537  /**
538  * @brief Computes the value of the superquadric from its equation
539  * @param centered_point point at which we make the evaluation, in the shape referential
540  */
541  inline double
542  superquadric(const Point<dim> &centered_point) const
543  {
544  return pow(abs(centered_point[0] / half_lengths[0]), exponents[0]) +
545  pow(abs(centered_point[1] / half_lengths[1]), exponents[1]) +
546  pow(abs(centered_point[2] / half_lengths[2]), exponents[2]) - 1.0;
547  }
548 
549  /**
550  * @brief Computes the gradient of the superquadric from its equation
551  * @param centered_point point at which we make the evaluation, in the shape referential
552  */
553  inline Point<dim>
554  superquadric_gradient(const Point<dim> &centered_point) const
555  {
556  Point<dim> gradient{};
557  for (unsigned int d = 0; d < dim; d++)
558  {
559  // For cases where the coordinate is of value 0, we avoid computing the
560  // gradient. That is an issue when the exponent is lower than or equal
561  // to 1
562  if (abs(centered_point[d]) > epsilon)
563  gradient[d] = exponents[d] *
564  pow(abs(half_lengths[d]), -exponents[d]) *
565  pow(abs(centered_point[d]), exponents[d]) /
566  (centered_point[d] + DBL_MIN);
567  else
568  gradient[d] = exponents[d] *
569  pow(abs(half_lengths[d]), -exponents[d]) *
570  pow(abs(epsilon), exponents[d]) / (epsilon + DBL_MIN);
571  }
572  return gradient;
573  }
574 
575 private:
576  Tensor<1, dim> half_lengths;
577  Tensor<1, dim> exponents;
578  double epsilon;
579 };

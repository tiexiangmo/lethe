28 /**
29  * Copies a two-dimensional tensor in a three-dimensional tensor. The third
30  * element of the three-dimensional tensor (in z direction) is set to zero.
31  * If the tensor given is already 3d, do nothing and return the tensor.
32  *
33  * @param tensor the tensor that should be transformed in 3d
34  * @return tensor_3d Three-dimensional output tensor
35  *
36  */
37 
38 template <int dim>
39 inline Tensor<1, 3>
40 tensor_nd_to_3d(const Tensor<1, dim> &tensor)
41 {
42  Tensor<1, 3> tensor_3d;
43  tensor_3d[0] = tensor[0];
44  tensor_3d[1] = tensor[1];
45  if constexpr (dim > 2)
46  tensor_3d[2] = tensor[2];
47 
48  if constexpr (dim == 2)
49  tensor_3d[2] = 0;
50 
51  return tensor_3d;
52 }

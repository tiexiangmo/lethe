445  const typename parallel::distributed::Triangulation<dim>::cell_iterator &cell,
446  const CellStatus status) const
447 #endif
448 
449 
450 {
451  // Assign no weight to cells we do not own.
452  if (!cell->is_locally_owned())
453  return 0;
454 
455  // This determines how important particle work is compared to cell
456  // work (by default every cell has a weight of 1000).
457  // We set the weight per particle higher to indicate that
458  // the particle load is more important than the fluid load. The optimal
459  // value of this number depends on the application and can range from 0
460  // (cheap particle operations, expensive cell operations) to much larger
461  // than 1000 (expensive particle operations, cheap cell operations, like in
462  // this case). This parameter will need to be tuned for different cases of
463  // CFD-DEM coupling.
464  const unsigned int particle_weight =
465  dem_parameters.model_parameters.load_balance_particle_weight;
466 
467  // This does not use adaptive refinement, therefore every cell
468  // should have the status CELL_PERSIST. However this function can also
469  // be used to distribute load during refinement, therefore we consider
470  // refined or coarsened cells as well.
471 
472 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
473  if (status == parallel::distributed::Triangulation<dim>::CELL_PERSIST ||
474  status == parallel::distributed::Triangulation<dim>::CELL_REFINE)
475 #else
476  if (status == CellStatus::cell_will_persist ||
477  status == CellStatus::cell_will_be_refined)
478 #endif
479 
480  {
481  const unsigned int n_particles_in_cell =
482  this->particle_handler.n_particles_in_cell(cell);
483  return n_particles_in_cell * particle_weight;
484  }
485 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
486  else if (status == parallel::distributed::Triangulation<dim>::CELL_COARSEN)

886 template <int dim>
887 void
888 CFDDEMSolver<dim>::dem_contact_build(unsigned int counter)
889 {
890  // Check to see if it is contact search step
891  contact_detection_step =
892  check_contact_detection_method(counter,
893  this->cfd_dem_simulation_parameters,
894  displacement,
895  this->particle_handler,
896  this->mpi_communicator,
897  this->simulation_control,
898  contact_detection_step,
899  checkpoint_step,
900  load_balance_step,
901  smallest_contact_search_criterion);
902 
903  // Sort particles in cells
904  // The last statement in the if condition allows the calling of
905  // dem_contact_build for the first DEM iteration in the first CFD time step
906  // directly after reading the dem initial checkpoint files
907 
908  if (contact_detection_step || checkpoint_step || load_balance_step ||
909  (this->simulation_control->is_at_start() && (counter == 0)))
910  {
911  this->pcout << "DEM contact search at dem step " << counter << std::endl;
912  contact_build_number++;
913

915  if (has_periodic_boundaries)
916  periodic_boundaries_object.execute_particles_displacement(
917  this->particle_handler, periodic_boundaries_cells_information);
918 
919  this->particle_handler.sort_particles_into_subdomains_and_cells();
920 
921  displacement.resize(
922  this->particle_handler.get_max_local_particle_index());
923  force.resize(displacement.size());
924  torque.resize(displacement.size());
925 
926  this->particle_handler.exchange_ghost_particles(true);
927 
928  if (has_disabled_contacts && !this->simulation_control->is_at_start())
929  {
930  // Update the active and ghost cells set (this should be done after a
931  // load balance or a checkpoint, but since the fem-dem code do not

934  disable_contacts_object.update_local_and_ghost_cell_set(

941  }
942

946  else
947  {
948  this->particle_handler.update_ghost_particles();
949  }
950 
951  // Modify particles contact containers by search sequence
952  if (load_balance_step || checkpoint_step || contact_detection_step ||
953  (this->simulation_control->is_at_start() && (counter == 0)))
954  {
955  // Execute broad search by filling containers of particle-particle
956  // contact pair candidates and containers of particle-wall
957  // contact pair candidates
958  if (!contacts_are_disabled(counter))
959  {
960  contact_manager.execute_particle_particle_broad_search(
961  this->particle_handler, has_periodic_boundaries);
962 
963  contact_manager.execute_particle_wall_broad_search(
964  this->particle_handler,
965  boundary_cell_object,
966  floating_mesh_info,
967  dem_parameters.floating_walls,
968  this->simulation_control->get_current_time());
969  }
970  else // disabling particle contacts are enabled & counter > 1
971  {
972  contact_manager.execute_particle_particle_broad_search(
973  this->particle_handler,
974  disable_contacts_object,
975  has_periodic_boundaries);
976 
977  contact_manager.execute_particle_wall_broad_search(
978  this->particle_handler,
979  boundary_cell_object,
980  floating_mesh_info,
981  dem_parameters.floating_walls,
982  this->simulation_control->get_current_time(),
983  disable_contacts_object);
984  }
985 
986  // Update contacts, remove replicates and add new contact pairs
987  // to the contact containers when particles are exchanged between
988  // processors
989  contact_manager.update_contacts(has_periodic_boundaries);
990 
991  // Updates the iterators to particles in local-local contact
992  // containers
993  contact_manager.update_local_particles_in_cells(this->particle_handler,
994  load_balance_step,
995  has_periodic_boundaries);
996 
997  // Execute fine search by updating particle-particle contact
998  // containers regards the neighborhood threshold
999  contact_manager.execute_particle_particle_fine_search(
1000  neighborhood_threshold_squared,
1001  has_periodic_boundaries,
1002  periodic_offset);
1003 
1004  // Execute fine search by updating particle-wall contact containers
1005  // regards the neighborhood threshold
1006  contact_manager.execute_particle_wall_fine_search(
1007  dem_parameters.floating_walls,
1008  this->simulation_control->get_current_time(),
1009  neighborhood_threshold_squared);
1010

1012  // we restart the simulation or everytime load balancing is performed. At
1013  // the end of the restart step or the load balance step, and after all
1014  // necessary contact build, vector resizing, and solution transfers have
1015  // been performed, this functions are set to false. The checkpoint_step
1016  // remains false for the duration of the simulation while the load
1017  // balancing is reset everytime load balancing is called.
1018  checkpoint_step = false;
1019  load_balance_step = false;

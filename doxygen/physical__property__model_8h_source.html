54 /**
55  * @brief Abstract class that defines the interface for a physical property model
56  * Physical property model provides an abstract interface to calculate the
57  * value of a physical property or a vector of physical property value for
58  * given field value. By default, the interface does not require that all (or
59  * any) fields be specified. This is why a map is used to pass the dependent
60  * variable. To allow for the calculation of the appropriate jacobian matrix
61  * (when that is necessary) the interface also provides a jacobian function,
62  * which must provide the derivative with respect to the field specified as an
63  * argument.
64  */

132  /**
133  * @brief numerical_jacobian Calculates the jacobian through a forward finite difference (Euler) approximation.
134  * This approach, although not preferable, is meant as a fall-back when
135  * calculating the jacobian manually is too difficult.
136  * @param fields_values The values of the various fields
137  * @param id Field id of the field with respect to which the jacobian should be calculated
138  * @return
139  */
140  inline double
141  numerical_jacobian(const std::map<field, double> &field_values,
142  const field id)
143  {
144  double f_x = this->value(field_values);
145  auto x_dx = field_values;
146  double dx = std::max(1e-6 * x_dx[id], 1e-8);
147  x_dx[id] = x_dx[id] + dx;
148  double f_xdx = this->value(x_dx);
149  return (f_xdx - f_x) / dx;
150  }
151 
152  /**
153  * @brief vector_numerical_jacobian Calculates the vector of jacobian through forward finite difference (Euler) approximation.
154  * This approach, although not preferable, is meant as a fall-back when
155  * calculating the jacobian manually is too difficult.
156  * @param field_vectors
157  * @param field_id
158  * @param jacobian_vector
159  * @return
160  */
161  inline void
162  vector_numerical_jacobian(
163  const std::map<field, std::vector<double>> &field_vectors,
164  const field id,
165  std::vector<double> &jacobian_vector)
166  {
167  const unsigned int n_pts = jacobian_vector.size();
168 
169  // Evaluate the properties using the values of the field vector
170  std::vector<double> f_x(n_pts);
171  vector_value(field_vectors, f_x);
172 
173  // Make a copy of the field vector for the field we wil perturbate
174  std::map<field, std::vector<double>> perturbed_field_vectors =
175  field_vectors;
176  std::vector<double> &x = perturbed_field_vectors.at(id);
177  std::vector<double> dx(n_pts);
178 
179  // Perturb the field by dx
180  for (unsigned int i = 0; i < n_pts; ++i)
181  {
182  dx[i] = std::max(1e-6 * x[i], 1e-8);
183  x[i] += dx[i];
184  }
185 
186  // Evaluate the properties using the perturbed value of the field vector
187  std::vector<double> f_xdx(n_pts);
188  vector_value(perturbed_field_vectors, f_xdx);
189 
190  // Fill jacobian
191  for (unsigned int i = 0; i < n_pts; ++i)
192  jacobian_vector[i] = (f_xdx[i] - f_x[i]) / dx[i];
193  }
194 
195 protected:
196  // Map to indicate on which variables the model depends on
197  std::map<field, bool> model_depends_on;
198 };

19 template <int dim, int spacedim>
20 void
21 attach_grid_to_triangulation(
22  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
23  const Parameters::Mesh &mesh_parameters)
24 
25 {
26  // GMSH input
27  if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
28  {
29  if (mesh_parameters.simplex)
30  {
31  auto comm = triangulation.get_communicator();
32  std::string file_name = mesh_parameters.file_name;
33 
34  auto construction_data = TriangulationDescription::Utilities::
35  create_description_from_triangulation_in_groups<dim, spacedim>(
36  [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
37  GridIn<dim, spacedim> grid_in;
38  grid_in.attach_triangulation(basetria);
39  std::ifstream input_file(file_name);
40 
41  grid_in.read_msh(input_file);
42  },
43  [](dealii::Triangulation<dim, spacedim> &basetria,
44  const MPI_Comm comm,
45  const unsigned int /*group_size*/) {
46  GridTools::partition_triangulation(
47  Utilities::MPI::n_mpi_processes(comm), basetria);
48  },
49  comm,
50  Utilities::MPI::n_mpi_processes(comm) /* group size */,
51  dealii::Triangulation<dim, spacedim>::none);
52 
53  triangulation.create_triangulation(construction_data);
54  }
55  else
56  {
57  GridIn<dim, spacedim> grid_in;
58  grid_in.attach_triangulation(triangulation);
59  std::ifstream input_file(mesh_parameters.file_name);
60  grid_in.read_msh(input_file);
61  }
62  }
63  // Dealii grids
64  else if (mesh_parameters.type == Parameters::Mesh::Type::dealii)
65  {
66  if (mesh_parameters.simplex)
67  {
68  Triangulation<dim, spacedim> temporary_quad_triangulation;
69  GridGenerator::generate_from_name_and_arguments(
70  temporary_quad_triangulation,
71  mesh_parameters.grid_type,
72  mesh_parameters.grid_arguments);
73 
74  // initial refinement
75  const unsigned int initial_refinement =
76  mesh_parameters.initial_refinement;
77  temporary_quad_triangulation.refine_global(initial_refinement);
78 
79  // flatten the triangulation
80  Triangulation<dim, spacedim> flat_temp_quad_triangulation;
81  GridGenerator::flatten_triangulation(temporary_quad_triangulation,
82  flat_temp_quad_triangulation);
83 
84  Triangulation<dim, spacedim> temporary_tri_triangulation(
85  Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
86  GridGenerator::convert_hypercube_to_simplex_mesh(
87  flat_temp_quad_triangulation, temporary_tri_triangulation);
88 
89  GridTools::partition_triangulation_zorder(
90  Utilities::MPI::n_mpi_processes(triangulation.get_communicator()),
91  temporary_tri_triangulation);
92  GridTools::partition_multigrid_levels(temporary_tri_triangulation);
93 
94  // extract relevant information from distributed triangulation
95  auto construction_data = TriangulationDescription::Utilities::
96  create_description_from_triangulation(
97  temporary_tri_triangulation,
98  triangulation.get_communicator(),
99  TriangulationDescription::Settings::
100  construct_multigrid_hierarchy);
101  triangulation.create_triangulation(construction_data);
102  }
103  else
104  {
105  GridGenerator::generate_from_name_and_arguments(
106  triangulation,
107  mesh_parameters.grid_type,
108  mesh_parameters.grid_arguments);
109  }
110  }
111  // Customizable cylinder mesh
112  else if (mesh_parameters.type == Parameters::Mesh::Type::cylinder)
113  {
114  if (mesh_parameters.simplex)
115  {
116  throw std::runtime_error(
117  "Unsupported mesh type - custom cylinder mesh with simplex is not supported. Use a dealii cylinder to use simplex mesh.");
118  }
119  else if (dim != 3)
120  {
121  throw std::runtime_error(
122  "Unsupported mesh type - custom cylinder mesh is only supported in 3d.");
123  }
124 
125  if constexpr (dim == 3)
126  {
127  // Separate arguments of the string
128  std::vector<std::string> arguments;
129  std::stringstream s_stream(mesh_parameters.grid_arguments);
130  while (s_stream.good())
131  {
132  std::string substr;
133  getline(s_stream, substr, ':');
134  arguments.push_back(substr);
135  }
136 
137  // Arguments declaration
138  unsigned int subdivisions;
139  double radius, half_height;
140  if (arguments.size() != 3)
141  {
142  throw std::runtime_error(
143  "Mandatory cylinder parameters are (x subdivisions: radius : half height)");
144  }
145  else
146  {
147  std::vector<double> arguments_double =
148  dealii::Utilities::string_to_double(arguments);
149  subdivisions = static_cast<int>(arguments_double[0]);
150  radius = arguments_double[1];
151  half_height = arguments_double[2];
152  }
153 
154  if (mesh_parameters.grid_type == "classic")
155  {
156  // Create a subdivided cylinder from deal.ii
157  GridGenerator::subdivided_cylinder(triangulation,
158  subdivisions,
159  radius,
160  half_height);
161  }
162  else
163  {
164  // Create a temporary 2d mesh
165  Triangulation<2, spacedim - 1> temporary_triangulation;
166 
167  // Create a spherical manifold for 2d mesh
168  Point<2> center(0.0, 0.0);
169  const SphericalManifold<2, spacedim - 1> m0(center);
170 
171  if (mesh_parameters.grid_type == "regularized" ||
172  mesh_parameters.grid_type == "squared")
173  {
174  // Create a square mesh
175  double real_radius = radius * std::sin(M_PI_4);
176  GridGenerator::hyper_cube(temporary_triangulation,
177  -real_radius,
178  real_radius,
179  true);
180 
181  // Assign boundary 0 to perimeter as for cylinder
182  for (const auto &cell :
183  temporary_triangulation.active_cell_iterators())
184  {
185  if (cell->is_locally_owned())
186  {
187  // Looping through all the faces of the cell
188  for (const auto &face : cell->face_iterators())
189  {
190  // Check to see if the face is located at boundary
191  if (face->at_boundary())
192  {
193  face->set_boundary_id(0);
194  }
195  }
196  }
197  }
198  }
199  else if (mesh_parameters.grid_type == "balanced")
200  {
201  GridGenerator::hyper_ball_balanced(temporary_triangulation,
202  center,
203  radius);
204  }
205  else
206  {
207  throw std::runtime_error(
208  "Unknown grid type. Choices are <classic|balanced|squared|regularized>.");
209  }
210 
211  temporary_triangulation.reset_all_manifolds();
212  temporary_triangulation.set_all_manifold_ids_on_boundary(0);
213  temporary_triangulation.set_manifold(0, m0);
214 
215  if (mesh_parameters.grid_type == "regularized")
216  {
217  // Pre-refinement to reduce mesh size at corners before
218  // regularization
219  temporary_triangulation.refine_global(2);
220  GridTools::regularize_corner_cells(temporary_triangulation);
221 
222  // Flatten the triangulation
223  Triangulation<2, spacedim - 1> flat_temporary_triangulation;
224  flat_temporary_triangulation.copy_triangulation(
225  temporary_triangulation);
226  temporary_triangulation.clear();
227  GridGenerator::flatten_triangulation(
228  flat_temporary_triangulation, temporary_triangulation);
229  }
230 
231  // Extrude the 2d temporary mesh to 3d cylinder
232  GridGenerator::extrude_triangulation(temporary_triangulation,
233  subdivisions + 1,
234  2.0 * half_height,
235  triangulation,
236  true);
237 
238  // Rotate mesh in x-axis and set the (0,0,0) at the barycenter
239  // to be comparable to dealii cylinder meshes
240  Tensor<1, spacedim> axis_vector({0.0, 1.0, 0.0});
241  GridTools::rotate(axis_vector, M_PI_2, triangulation);
242  Tensor<1, spacedim> shift_vector({-half_height, 0.0, 0.0});
243  GridTools::shift(shift_vector, triangulation);
244 
245  // Force the manifold id to be zero in the case of the balanced
246  // cylinder
247  if (mesh_parameters.grid_type == "balanced")
248  triangulation.reset_manifold(1);
249 
250  // Add a cylindrical manifold on the final unrefined mesh
251  const CylindricalManifold<3, spacedim> m1(0);
252  triangulation.set_manifold(0, m1);
253  }
254  }
255  }
256  // Periodic Hills grid
257  else if (mesh_parameters.type == Parameters::Mesh::Type::periodic_hills &&
258  !mesh_parameters.simplex)
259  {
260  if (mesh_parameters.simplex)
261  {
262  throw std::runtime_error(
263  "Unsupported mesh type - periodic hills mesh with simplex is not supported");
264  }
265  else
266  {
267  PeriodicHillsGrid<dim, spacedim> grid(mesh_parameters.grid_arguments);
268  grid.make_grid(triangulation);
269  }
270  }
271  else
272  throw std::runtime_error(
273  "Unsupported mesh type - mesh will not be created");
274 }

277 template <int dim, int spacedim>
278 void
279 setup_periodic_boundary_conditions(
280  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
281  const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
282 
283 {
284  // Setup periodic boundary conditions
285  for (unsigned int i_bc = 0; i_bc < boundary_conditions.size; ++i_bc)
286  {
287  if (boundary_conditions.type[i_bc] ==
288  BoundaryConditions::BoundaryType::periodic)
289  {
290  auto triangulation_ptr = &triangulation;
291 
292  std::vector<GridTools::PeriodicFacePair<
293  typename Triangulation<dim, spacedim>::cell_iterator>>
294  periodicity_vector;
295  GridTools::collect_periodic_faces(
296  *dynamic_cast<Triangulation<dim, spacedim> *>(triangulation_ptr),
297  boundary_conditions.id[i_bc],
298  boundary_conditions.periodic_id[i_bc],
299  boundary_conditions.periodic_direction[i_bc],
300  periodicity_vector);
301  triangulation.add_periodicity(periodicity_vector);
302  }
303  }

306 template <int dim, int spacedim>
307 void
308 read_mesh_and_manifolds(
309  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
310  const Parameters::Mesh &mesh_parameters,
311  const Parameters::Manifolds &manifolds_parameters,
312  const bool &restart,
313  const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
314 {
315  attach_grid_to_triangulation(triangulation, mesh_parameters);
316 
317  setup_periodic_boundary_conditions(triangulation, boundary_conditions);
318 
319  // If the mesh is a GMSH mesh, we need to manually set the manifold id
320  // of the face to be that of the boundary
321  // ID, otherwise, we will be unable to use external manifold.
322  // This is done manually by looping through all faces and giving them a
323  // manifold id if there is a manifold associated with this number.
324  if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
325  {
326  // Gather all the manifold ids within a set
327  std::set<int> manifold_ids;
328  for (unsigned int i = 0; i < manifolds_parameters.size; ++i)
329  manifold_ids.insert(manifolds_parameters.id[i]);
330 
331  // Reset all the manifolds manually and force them to zero
332  triangulation.reset_all_manifolds();
333 
334  // If the parameter file forces the occurrence of manifold,
335  // loop over the faces of the triangulation. If the face of the
336  // triangulation has a boundary id which corresponds to a manifold id
337  // identified within the parameter file, then fix the manifold id of this
338  // face manually to be that of the boundary id. In the past, this was done
339  // by default for every face, but since 2023-12 this throws (rightfully)
340  // an error in deal.II
341  if (manifolds_parameters.size > 0)
342  {
343  for (const auto &face : triangulation.active_face_iterators())
344  {
345  if (face->at_boundary() &&
346  manifold_ids.find(face->boundary_id()) != manifold_ids.end())
347  face->set_all_manifold_ids(face->boundary_id());
348  }
349  }
350  }
351 
352  // Finally attach the manifolds to the triangulation
353  // Right now this should only occur for GMSH mesh, but the function is generic
354  // enough.
355  attach_manifolds_to_triangulation(triangulation, manifolds_parameters);
356 
357  if (mesh_parameters.simplex)
358  {
359  // Refinement isn't possible yet
360  }
361  else
362  {
363  if (mesh_parameters.refine_until_target_size)
364  {
365  double minimal_cell_size =
366  GridTools::minimal_cell_diameter(triangulation);
367  double target_size = mesh_parameters.target_size;
368  unsigned int number_refinement =
369  floor(std::log(minimal_cell_size / target_size) / std::log(2));
370  triangulation.refine_global(number_refinement);
371  }
372  else if (!restart)
373  {
374  const int initial_refinement = mesh_parameters.initial_refinement;
375  triangulation.refine_global(initial_refinement);
376  }
377  }
378 }

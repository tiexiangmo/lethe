51 /**
52  * @brief Class that stores the information required by the assembly procedure
53  * for a heat transfer advection-diffusion equation. Consequently, this class
54  * calculates the heat transfer (values, gradients, laplacians) and the shape
55  * function (values, gradients, laplacians) at all the gauss points for all
56  * degrees of freedom and stores it into arrays. Additionnaly, the user can
57  * request that this class gathers additional fields for physics which are
58  * coupled to the heat transfer equation, such as the velocity which is
59  * required. This class serves as a seperation between the evaluation at the
60  * gauss point of the variables of interest and their use in the assembly, which
61  * is carried out by the assembler methods. For more information on this
62  * design, the reader can consult deal.II step-9
63  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
64  * example, the scratch is a struct instead of a templated class because of the
65  * simplicity of step-9.
66  *
67  * @tparam dim An integer that denotes the dimension of the space in which
68  * the flow is solved
69  *
70  * @ingroup solvers
71  **/
72

79  * to fill the member variables of the scratch. It also allocated the
80  * necessary memory for all member variables. However, it does not do any
81  * evalution, since this needs to be done at the cell level.
82  *
83  * @param properties_manager The physical properties Manager (see physical_properties_manager.h)
84  *
85  * @param fe_ht The FESystem used to solve the Heat Transfer equations
86  *
87  * @param quadrature The quadrature to use for the assembly
88  *
89  * @param mapping The mapping of the domain in which the Navier-Stokes equations are solved
90  *

164  /** @brief Reinitialize the content of the scratch
165  *
166  * Using the FeValues and the content of the solutions, previous solutions ,
167  * fills all of the class member of the scratch
168  *
169  * @tparam VectorType The Vector type used for the solvers
170  *
171  * @param cell The cell over which the assembly is being carried.
172  * This cell must be compatible with the fe which is used to fill the FeValues
173  *
174  * @param current_solution The present value of the solution for the Heat Transfer
175  *
176  * @param previous_solutions The solutions at the previous time steps
177  *
178  * @param source_function The function describing the Heat Transfer source term
179  */
180 
181  template <typename VectorType>
182  void
183  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
184  const VectorType &current_solution,
185  const std::vector<TrilinosWrappers::MPI::Vector> &previous_solutions,
186  Function<dim> *source_function)
187  {
188  material_id = cell->material_id();
189  this->fe_values_T.reinit(cell);
190 
191  quadrature_points = this->fe_values_T.get_quadrature_points();
192  auto &fe_T = this->fe_values_T.get_fe();
193 
194  source_function->value_list(quadrature_points, source);
195 
196  if (dim == 2)
197  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe_T.degree;
198  else if (dim == 3)
199  this->cell_size = pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_T.degree;
200 
201  // Gather temperature (values, gradient and laplacian)
202  this->fe_values_T.get_function_values(current_solution,
203  this->present_temperature_values);
204  this->fe_values_T.get_function_gradients(current_solution,
205  this->temperature_gradients);
206  this->fe_values_T.get_function_laplacians(
207  current_solution, this->present_temperature_laplacians);
208 
209  // Gather previous temperature values
210  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
211  {
212  this->fe_values_T.get_function_values(previous_solutions[p],
213  previous_temperature_values[p]);
214 
215  this->fe_values_T.get_function_gradients(
216  previous_solutions[p], previous_temperature_gradients[p]);
217  }
218 
219  for (unsigned int q = 0; q < n_q_points; ++q)
220  {
221  this->JxW[q] = this->fe_values_T.JxW(q);
222 
223  for (unsigned int k = 0; k < n_dofs; ++k)
224  {
225  // Shape function
226  this->phi_T[q][k] = this->fe_values_T.shape_value(k, q);
227  this->grad_phi_T[q][k] = this->fe_values_T.shape_grad(k, q);
228  this->hess_phi_T[q][k] = this->fe_values_T.shape_hessian(k, q);
229  this->laplacian_phi_T[q][k] = trace(this->hess_phi_T[q][k]);
230  }
231  }
232 
233 
234  // Arrays related to faces must be re-initialized for each cell, since they
235  // might depend on reference cell
236  // Only carry out this initialization if the cell is a boundary cell,
237  // otherwise these are wasted calculations
238  this->is_boundary_cell = cell->at_boundary();
239  if (cell->at_boundary())
240  {
241  n_faces = cell->n_faces();
242  is_boundary_face = std::vector<bool>(n_faces, false);
243  n_faces_q_points = fe_face_values_ht.get_quadrature().size();
244  boundary_face_id = std::vector<unsigned int>(n_faces);
245 
246  face_JxW = std::vector<std::vector<double>>(
247  n_faces, std::vector<double>(n_faces_q_points));
248 
249 
250  this->phi_face_T = std::vector<std::vector<std::vector<double>>>(
251  n_faces,
252  std::vector<std::vector<double>>(n_faces_q_points,
253  std::vector<double>(n_dofs)));
254 
255  this->temperature_face_value = std::vector<std::vector<double>>(
256  n_faces, std::vector<double>(n_faces_q_points));
257 
258  for (const auto face : cell->face_indices())
259  {
260  this->is_boundary_face[face] = cell->face(face)->at_boundary();
261  if (this->is_boundary_face[face])
262  {
263  fe_face_values_ht.reinit(cell, face);
264  boundary_face_id[face] = cell->face(face)->boundary_id();
265  this->fe_face_values_ht.get_function_values(
266  current_solution, this->temperature_face_value[face]);
267 
268  for (unsigned int q = 0; q < n_faces_q_points; ++q)
269  {
270  face_JxW[face][q] = fe_face_values_ht.JxW(q);
271  for (const unsigned int k : fe_face_values_ht.dof_indices())
272  {
273  this->phi_face_T[face][q][k] =
274  this->fe_face_values_ht.shape_value(k, q);
275  }
276  }
277  }
278  }
279  }

282  /** @brief Reinitialize the velocity, calculated by the fluid dynamics while also taking into account ALE
283  *
284  * @tparam VectorType The Vector type used for the solvers
285  *
286  * @param cell The cell for which the velocity is reinitialized
287  * This cell must be compatible with the Fluid Dynamics FE
288  *
289  * @param current_solution The present value of the solution for [u,p]
290  *
291  * @param ale The ALE parameters which include the ALE function
292  *
293  */
294 
295  template <typename VectorType>
296  void
297  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
298  const VectorType &current_solution,
299  const Parameters::ALE<dim> &ale)
300  {
301  this->fe_values_fd.reinit(cell);
302 
303  this->fe_values_fd[velocities].get_function_values(current_solution,
304  velocity_values);
305 
306  if (!ale.enabled())
307  return;
308 
309  // ALE enabled, so extract the ALE velocity and subtract it from the
310  // velocity obtained from the fluid dynamics
311  Tensor<1, dim> velocity_ale;
312  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
313  ale.velocity;
314  Vector<double> velocity_ale_vector(dim);
315 
316  for (unsigned int q = 0; q < n_q_points; ++q)
317  {
318  velocity_ale_function->vector_value(quadrature_points[q],
319  velocity_ale_vector);
320  for (unsigned int d = 0; d < dim; ++d)
321  velocity_ale[d] = velocity_ale_vector[d];
322 
323  velocity_values[q] -= velocity_ale;
324  }
325  }

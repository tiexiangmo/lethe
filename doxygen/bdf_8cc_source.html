22 Vector<double>
23 bdf_coefficients(const unsigned int p, const std::vector<double> dt)
24 {
25  // There should be at least p time steps
26  assert(dt.size() >= p);
27 
28  // Create a time table for the bdf formula
29  Vector<double> times(p + 1);
30  for (unsigned int i = 0; i < p + 1; ++i)
31  {
32  times[i] = 0.;
33  for (unsigned int j = 0; j < i; ++j)
34  times[i] -= dt[j];
35  }
36 
37  // The alphas are the bdf coefficients
38  Vector<double> alpha(p + 1);
39  alpha = 0;
40 
41  for (unsigned int j = 1; j < p + 1; ++j)
42  {
43  double factor = 1.;
44  for (unsigned int i = 1; i < j; ++i)
45  factor *= times[0] - times[i];
46 
47  Vector<double> term(p + 1);
48  term.equ(factor, delta(p, 0, j, times));
49  alpha += term;
50  }
51  return alpha;
52 }

75 Vector<double>
76 delta(const unsigned int p,
77  const unsigned int n,
78  const unsigned int j,
79  const Vector<double> times)
80 {
81  if (j == 0)
82  {
83  Vector<double> arr(p + 1);
84  arr = 0.;
85  arr[n] = 1;
86  return arr;
87  }
88  else
89  {
90  Vector<double> delta_1 = delta(p, n, j - 1, times);
91  Vector<double> delta_2 = delta(p, n + 1, j - 1, times);
92  Vector<double> delta_sol(p + 1);
93  for (unsigned int i_del = 0; i_del < p + 1; ++i_del)
94  delta_sol[i_del] =
95  (delta_1[i_del] - delta_2[i_del]) / (times[n] - times[n + j]);
96  return delta_sol;
97  }
98 }

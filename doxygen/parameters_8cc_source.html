868  void
869  Material::parse_parameters(ParameterHandler &prm,
870  std::string material_prefix,
871  const unsigned int id,
872  const Parameters::Dimensionality dimensions)
873  {
874  prm.enter_subsection(material_prefix + " " +
875  Utilities::int_to_string(id, 1));
876  {
877  // String that will be used to parse the models
878  std::string op;
879 
880  //---------------------------------------------------
881  // Density
882  //---------------------------------------------------
883  op = prm.get("density model");
884  if (op == "constant")
885  {
886  density_model = DensityModel::constant;
887  }
888  else if (op == "isothermal_ideal_gas")
889  {
890  density_model = DensityModel::isothermal_ideal_gas;
891  }
892  density = prm.get_double("density");
893  // Density is in M L^-3, rescale
894  density *= dimensions.density_scaling;
895  isothermal_ideal_gas_density_parameters.parse_parameters(prm, dimensions);
896 
897  //---------------------------------------------------
898  // Kinematic viscosity and Rheology
899  //---------------------------------------------------
900  op = prm.get("rheological model");
901  if (op == "power-law")
902  {
903  rheological_model = RheologicalModel::powerlaw;
904  }
905  else if (op == "carreau")
906  {
907  rheological_model = RheologicalModel::carreau;
908  }
909  else if (op == "newtonian")
910  {
911  rheological_model = RheologicalModel::newtonian;
912  }
913  else if (op == "phase_change")
914  {
915  rheological_model = RheologicalModel::phase_change;
916  }
917 
918  kinematic_viscosity = prm.get_double("kinematic viscosity");
919  // Kinematic viscosity is in L^2 T^-1, rescale
920  kinematic_viscosity *= dimensions.viscosity_scaling;
921  non_newtonian_parameters.parse_parameters(prm, dimensions);
922 
923  //--------------
924  // Specific heat
925  //--------------
926  op = prm.get("specific heat model");
927  if (op == "constant")
928  specific_heat_model = SpecificHeatModel::constant;
929  else if (op == "phase_change")
930  specific_heat_model = SpecificHeatModel::phase_change;
931  specific_heat = prm.get_double("specific heat");
932 
933  // specific heat is in L^2 T^-2 theta^-1
934  specific_heat *= dimensions.specific_heat_scaling;
935 
936 
937  //----------------------
938  // Thermal conductivity
939  //----------------------
940  op = prm.get("thermal conductivity model");
941  if (op == "constant")
942  thermal_conductivity_model = ThermalConductivityModel::constant;
943  else if (op == "linear")
944  thermal_conductivity_model = ThermalConductivityModel::linear;
945  else if (op == "phase_change")
946  thermal_conductivity_model = ThermalConductivityModel::phase_change;
947 
948  thermal_conductivity = prm.get_double("thermal conductivity");
949  // thermal conductivity is in M L T^-3 theta^-1
950  thermal_conductivity *= dimensions.thermal_conductivity_scaling;
951 
952 
953  // Linear conductivity model parameters
954  k_A0 = prm.get_double("k_A0");
955  // k_A0 is in M L T^-3 theta^-1
956  k_A0 *= dimensions.thermal_conductivity_scaling;
957 
958  k_A1 = prm.get_double("k_A1");
959  // k_A1 is in M L T^-3 theta ^-2
960  k_A1 *= dimensions.thermal_conductivity_scaling * dimensions.temperature;
961 
962 
963  //------------------
964  // Thermal expansion
965  //------------------
966  op = prm.get("thermal expansion model");
967  if (op == "constant")
968  thermal_expansion_model = ThermalExpansionModel::constant;
969  else if (op == "phase_change")
970  thermal_expansion_model = ThermalExpansionModel::phase_change;
971 
972  thermal_expansion = prm.get_double("thermal expansion");
973  // thermal expansion is in theta^-1
974  thermal_expansion *= dimensions.temperature;
975 
976  //-------------------
977  // Tracer diffusivity
978  //-------------------
979  tracer_diffusivity = prm.get_double("tracer diffusivity");
980  // Diffusivity is in L^2 T^-1
981  tracer_diffusivity *= dimensions.diffusivity_scaling;
982 
983  //--------------------------------
984  // Phase change properties
985  //--------------------------------
986  phase_change_parameters.parse_parameters(prm, dimensions);

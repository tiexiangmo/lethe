30 /**
31  * Insertion of particles using cells cut by a plane
32  *
33  * @note
34  *
35  * Particle insertion using cells cut by a plane. Locally owned cells that are
36  * cut by the plane are flagged. From those flagged cells, we insert a particle
37  * at their centroids if they are individually empty (contain no particle). This
38  * way, no significant overlap is occurring on the insertion of new particle
39  * which can occur with other insertion method.
40  */
41 
42 template <int dim>
43 class PlaneInsertion : public Insertion<dim>
44 {
45 public:
46  /**
47  * The plane insertion class insert particles using a plane
48  * define by a point an a normal vector. This method of insertion can be
49  * useful when dealing with a domain close to be fully filled with particle.
50  * In this situation, other insertion method have a high risk to create a big
51  * overlap between particles on insertion. The plane insertion method mitigate
52  * this risk by insertion at the center of empty cells.
53  *
54  * @param dem_parameters DEM parameters declared in the .prm file
55  * @param triangulation Triangulation object used in the simulation.
56  */
57  PlaneInsertion(const DEMSolverParameters<dim> &dem_parameters,
58  const parallel::distributed::Triangulation<dim> &triangulation,
59  const std::vector<std::shared_ptr<Distribution>>
60  &distribution_object_container);

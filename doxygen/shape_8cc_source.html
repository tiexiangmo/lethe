49 template <int dim>
50 Point<dim>
51 Shape<dim>::align_and_center(const Point<dim> &evaluation_point) const
52 {
53  // Translation and rotation to standard position and orientation for
54  // distance calculations
55  Point<dim> center_of_rotation = position;
56  Point<dim> rotated_point;
57  Point<dim> translated_point;
58 
59  // Rotation from the solid orientation
60  // Angular position around x, y and z axis
61  Tensor<1, 3> theta = orientation;
62 
63  // The centralized point is the one to be rotated, and it is updated after
64  // each rotation around one axis The centralized rotated point is the result
65  // of each rotation, and it is initialized in case no rotation is performed
66  Point<dim> centralized_point;
67  centralized_point = evaluation_point - center_of_rotation;
68  Point<dim> centralized_rotated = centralized_point;
69 
70  // Selection of the first axis around which to rotate:
71  // x -> 0, y -> 1, z -> 2
72  // In 2D, only rotation around the z axis is possible
73  if constexpr (dim == 2)
74  {
75  if (std::abs(theta[2]) > 1e-10 * this->effective_radius)
76  {
77  Tensor<2, 2> rotation_matrix =
78  Physics::Transformations::Rotations::rotation_matrix_2d(-theta[2]);
79 
80  // Multiplication
81  centralized_rotated.clear();
82  for (unsigned int j = 0; j < dim; ++j)
83  {
84  for (unsigned int k = 0; k < dim; ++k)
85  {
86  centralized_rotated[j] +=
87  rotation_matrix[j][k] * centralized_point[k];
88  }
89  }
90  centralized_point = centralized_rotated;
91  }
92  }
93  else // (dim == 3)
94  {
95  for (unsigned int i = 0; i < 3; ++i)
96  {
97  if (std::abs(theta[i]) > 1e-10 * this->effective_radius)
98  {
99  Tensor<1, 3> axis;
100  axis[i] = 1.0;
101  Tensor<2, 3> rotation_matrix =
102  Physics::Transformations::Rotations::rotation_matrix_3d(
103  axis, -theta[i]);
104 
105  // Multiplication
106  centralized_rotated.clear();
107  for (unsigned int j = 0; j < dim; ++j)
108  {
109  for (unsigned int k = 0; k < dim; ++k)
110  {
111  centralized_rotated[j] +=
112  rotation_matrix[j][k] * centralized_point[k];
113  }
114  }
115  centralized_point = centralized_rotated;
116  }
117  }
118  }
119  rotated_point = centralized_rotated + center_of_rotation;
120 
121  // Translation from the solid position
122  translated_point = rotated_point - position;
123 
124  return translated_point;
125 }

232 Shape<dim>::closest_surface_point(
233  const Point<dim> &p,
234  Point<dim> &closest_point,
235  const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
236 {
237  Tensor<1, dim> actual_gradient;

239  actual_gradient = this->gradient_with_cell_guess(p, cell_guess);

244  closest_point = p - (actual_gradient / (actual_gradient.norm() +

313 template <int dim>
314 Tensor<1, dim>
315 Sphere<dim>::gradient(const Point<dim> &evaluation_point,
316  const unsigned int /*component*/) const
317 {
318  // We make sure that the evaluation point and the sphere center are different,
319  // because if they are the same the analytical gradient is not defined: the
320  // function returns a NaN. We use the numerical gradient if the points are the
321  // same.
322  if ((evaluation_point - this->position).norm() <
323  1e-12 * this->effective_radius)
324  return AutoDerivativeFunction<dim>::gradient(evaluation_point);
325 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
326  const Tensor<1, dim> center_to_point = evaluation_point - this->position;
327  const Tensor<1, dim> grad = center_to_point / center_to_point.norm();
328  return grad;
329 #else
330  return sphere_function->gradient(evaluation_point);
331 #endif
332 }

396 template <int dim>
397 void
398 Superquadric<dim>::closest_surface_point(const Point<dim> &p,
399  Point<dim> &closest_point) const
400 {
401  auto point_in_string = this->point_to_string(p);
402  auto iterator = this->closest_point_cache.find(point_in_string);
403  if (iterator == this->closest_point_cache.end())
404  {
405  // The initial guess is chosen as the centered point (i.e. evaluation
406  // point, in the superquadric referential). It should already be somewhat
407  // close to the closest point, and it is already in the right octant.
408  Point<dim> current_point = this->align_and_center(p);
409 
410  unsigned int iteration = 0;
411  const unsigned int iteration_max = 1e2;
412 
413  Point<dim> dx{}, distance_gradient{};
414  double current_distance = superquadric(current_point);
415 
416  const double relaxation = 0.5;
417  while (iteration < iteration_max && abs(current_distance) > epsilon)
418  {
419  distance_gradient = superquadric_gradient(current_point);
420  if (distance_gradient.norm() < epsilon)
421  // Gradient can be null if the evaluation point is exactly on the
422  // centroid of the shape. In this case it is also the closest point.
423  break;
424  dx = -relaxation * (current_distance * distance_gradient) /
425  distance_gradient.norm_square();
426 
427  current_point = current_point + dx;
428  current_distance = superquadric(current_point);
429 
430  iteration++;
431  }
432 
433  closest_point = this->reverse_align_and_center(current_point);
434  }
435  else
436  closest_point = iterator->second;
437 }

439 template <int dim>
440 double
441 Superquadric<dim>::value(const Point<dim> &evaluation_point,
442  const unsigned int /*component*/) const
443 {
444  using numbers::PI;
445 
446  auto point_in_string = this->point_to_string(evaluation_point);
447  auto iterator = this->value_cache.find(point_in_string);
448  if (iterator == this->value_cache.end())
449  {
450  // The closest point has to be found first, because it is used in value
451  // calculation.
452  Point<dim> closest_point{};
453  this->closest_surface_point(evaluation_point, closest_point);
454 
455  Point<dim> centered_point = this->align_and_center(evaluation_point);
456  if (superquadric(centered_point) > 0)
457  return (closest_point - evaluation_point).norm() -
458  this->layer_thickening;
459  else
460  return -(closest_point - evaluation_point).norm() -
461  this->layer_thickening;
462  }
463  else
464  return iterator->second;
465 }

566 template <int dim>
567 double
568 OpenCascadeShape<dim>::value(const Point<dim> &evaluation_point,
569  const unsigned int /*component*/) const
570 {
571 #ifdef DEAL_II_WITH_OPENCASCADE
572  auto point_in_string = this->point_to_string(evaluation_point);
573  auto iterator = this->value_cache.find(point_in_string);
574  if (iterator != this->value_cache.end())
575  return iterator->second;
576 
577  Point<dim> centered_point = this->align_and_center(evaluation_point);
578  Point<dim> projected_point;
579  auto pt = OpenCASCADE::point(centered_point);
580  TopoDS_Vertex vertex = BRepBuilderAPI_MakeVertex(pt);
581  BRepExtrema_DistShapeShape distancetool(shape, vertex);
582  distancetool.Perform();
583  gp_Pnt pt_on_surface = distancetool.PointOnShape1(1);
584  projected_point[0] = pt_on_surface.X();
585  projected_point[1] = pt_on_surface.Y();
586  if constexpr (dim == 3)
587  {
588  projected_point[2] = pt_on_surface.Z();
589  }
590 
591  BRepClass3d_SolidClassifier point_classifier(shape);
592  point_classifier.Perform(pt, shape_tol);
593  TopAbs_State point_state = point_classifier.State();
594  // Check if the evaluation point is inside the shape (This check can return
595  // true only if the shape is a solid). By default, step file format will
596  // define a solid. This is necessary to sign the distance function evaluation.
597  if (point_state == TopAbs_State::TopAbs_IN)
598  {
599  // If the evaluation point is inside the shape and the shape is a solid,
600  // the distance to the shape will be 0. This is why we need to evaluate
601  // the distance of the point with the shell of the shape.
602  // Rotate the solution found to the global reference frame.
603  auto rotate_in_globalpoint =
604  this->reverse_align_and_center(projected_point);
605  return -(rotate_in_globalpoint - evaluation_point).norm() -
606  this->layer_thickening;
607  }
608  else
609  {
610  auto rotate_in_globalpoint =
611  this->reverse_align_and_center(projected_point);
612  return (rotate_in_globalpoint - evaluation_point).norm() -
613  this->layer_thickening;
614  }
615 
616 #else
617  (void)evaluation_point;
618  return 0;
619 #endif
620 }

635  Point<dim> centered_point = this->align_and_center(evaluation_point);

640  distancetool.LoadS2(vertex);

644  gp_Pnt pt_on_surface = distancetool.PointOnShape1(1);
645  projected_point[0] = pt_on_surface.X();

647  if constexpr (dim == 3)

653  TopAbs_State point_state = point_classifier.State();
654  // Check if the evaluation point is inside the shape (This check can
655  // return true only if the shape is a solid). By default, step file format
656  // will define a solid. This is necessary to sign the distance function
657  // evaluation.
658  if (point_state == TopAbs_State::TopAbs_IN)

661  // solid, the distance to the shape will be 0. This is why we need to

663  // Rotate the solution found to the global reference frame and cache
664  // the solution.
665  if (!this->part_of_a_composite)
666  {

1021  intermediate_q[1] * b_intermediate1[1]});
1022  Point<2> b;
1023  b = w - b_intermediate2;
1024  double k = (intermediate_q[1] > 0) ? 1 : ((intermediate_q[1] < 0) ? -1 : 0);
1025  double d = std::min(scalar_product<1, 2, double>(a, a),
1026  scalar_product<1, 2, double>(b, b));
1027  double s = std::max(k * (w[0] * intermediate_q[1] - w[1] * intermediate_q[0]),
1028  k * (w[1] - intermediate_q[1]));
1029 
1030  return sqrt(d) * ((s > 0) ? 1 : ((s < 0) ? -1 : 0)) - this->layer_thickening;

1281  auto point_in_string = this->point_to_string(evaluation_point);
1282  auto iterator = this->gradient_cache.find(point_in_string);
1283  if (iterator == this->gradient_cache.end())
1284  {
1285  // We align and center the evaluation point according to the shape
1286  // referential
1287  Point<dim> centered_point = this->align_and_center(evaluation_point);
1288  // The levelset value and gradient of all component shapes is computed
1289  std::map<unsigned int, double> constituent_shapes_values;
1290  std::map<unsigned int, Tensor<1, dim>> constituent_shapes_gradients;
1291  for (auto const &[component_id, component] : constituents)
1292  {
1293  constituent_shapes_values[component_id] =
1294  component->value_with_cell_guess(centered_point, cell);
1295  constituent_shapes_gradients[component_id] =
1296  component->gradient_with_cell_guess(centered_point, cell);
1297  }
1298 
1299  double levelset;
1300  Tensor<1, dim> gradient;
1301  std::tie(levelset, gradient) =
1302  apply_boolean_operations(constituent_shapes_values,
1303  constituent_shapes_gradients);
1304 
1305  // The gradient obtained is in the shape referential. It needs to be
1306  // returned to the global referential. We find the closest point (in the
1307  // shape referential), we return that point to its global referential
1308  // equivalent, then compute the gradient in the global referential.
1309  Point<dim> closest_point_shape_referential =
1310  centered_point -
1311  (gradient / (gradient.norm() + 1e-16 * this->effective_radius)) *
1312  levelset;
1313  Point<dim> closest_point_global_referential =
1314  this->reverse_align_and_center(closest_point_shape_referential);
1315  gradient = -(closest_point_global_referential - evaluation_point) /
1316  ((closest_point_global_referential - evaluation_point).norm() +
1317  1.0e-16);
1318  if (!this->part_of_a_composite)
1319  {
1320  this->gradient_cache[point_in_string] = gradient;
1321  }
1322  return gradient;
1323  }
1324  else
1325  return this->gradient_cache[point_in_string];
1326 }

1386 template <int dim>
1387 std::pair<double, Tensor<1, dim>>
1388 CompositeShape<dim>::apply_boolean_operations(
1389  std::map<unsigned int, double> constituent_shapes_values,
1390  std::map<unsigned int, Tensor<1, dim>> constituent_shapes_gradients) const
1391 {
1392  // The boolean operations between the shapes are applied in order
1393  // The last computed values are considered to be the correct values to return
1394  double levelset = constituent_shapes_values[0];
1395  Tensor<1, dim> gradient = constituent_shapes_gradients[0];
1396  for (auto const &[operation_id, op_triplet] : operations)
1397  {
1398  BooleanOperation operation;
1399  unsigned int first_id;
1400  unsigned int second_id;
1401  std::tie(operation, first_id, second_id) = op_triplet;
1402 
1403  double value_first_component, value_second_component;
1404  Tensor<1, dim> gradient_first_component{};
1405  Tensor<1, dim> gradient_second_component{};
1406  value_first_component = constituent_shapes_values.at(first_id);
1407  value_second_component = constituent_shapes_values.at(second_id);
1408  gradient_first_component = constituent_shapes_gradients.at(first_id);
1409  gradient_second_component = constituent_shapes_gradients.at(second_id);
1410  switch (operation)
1411  {
1412  case BooleanOperation::Union:
1413  if (value_first_component < value_second_component)
1414  {
1415  constituent_shapes_values[operation_id] = value_first_component;
1416  constituent_shapes_gradients[operation_id] =
1417  gradient_first_component;
1418  }
1419  else
1420  {
1421  constituent_shapes_values[operation_id] =
1422  value_second_component;
1423  constituent_shapes_gradients[operation_id] =
1424  gradient_second_component;
1425  }
1426  break;
1427  case BooleanOperation::Difference:
1428  if (-value_first_component > value_second_component)
1429  {
1430  constituent_shapes_values[operation_id] =
1431  -value_first_component;
1432  constituent_shapes_gradients[operation_id] =
1433  -gradient_first_component;
1434  }
1435  else
1436  {
1437  constituent_shapes_values[operation_id] =
1438  value_second_component;
1439  constituent_shapes_gradients[operation_id] =
1440  gradient_second_component;
1441  }
1442  break;
1443  default: // BooleanOperation::Intersection
1444  if (value_first_component > value_second_component)
1445  {
1446  constituent_shapes_values[operation_id] = value_first_component;
1447  constituent_shapes_gradients[operation_id] =
1448  gradient_first_component;
1449  }
1450  else
1451  {
1452  constituent_shapes_values[operation_id] =
1453  value_second_component;
1454  constituent_shapes_gradients[operation_id] =
1455  gradient_second_component;
1456  }
1457  }
1458  levelset = constituent_shapes_values[operation_id];
1459  gradient = constituent_shapes_gradients[operation_id];
1460  }
1461  return {levelset, gradient};
1462 }

1565 template <int dim>
1566 double
1567 RBFShape<dim>::value(const Point<dim> &evaluation_point,
1568  const unsigned int /*component*/) const
1569 {
1570  double bounding_box_distance = bounding_box->value(evaluation_point);
1571  if (bounding_box_distance >= 0)
1572  return bounding_box_distance + this->effective_radius -
1573  this->layer_thickening;
1574 
1575  auto point_in_string = this->point_to_string(evaluation_point);
1576  auto iterator = this->value_cache.find(point_in_string);
1577  if (iterator != this->value_cache.end())
1578  return iterator->second;
1579 
1580  double value = std::max(bounding_box_distance, 0.0);
1581  double normalized_distance, basis;
1582  // Algorithm inspired by Optimad Bitpit. https://github.com/optimad/bitpit
1583  // Here we loop on every portion (of RBF nodes located in active cells close
1584  // to the evaluation point) and on every RBF node located in these active
1585  // cells.
1586  for (size_t portion_id = 0; portion_id < iterable_nodes.size(); portion_id++)
1587  {
1588  for (const size_t &node_id : *(std::get<2>(iterable_nodes[portion_id])))
1589  {
1590  normalized_distance =
1591  (evaluation_point - rotated_nodes_positions[node_id]).norm() /
1592  support_radii[node_id];
1593  // We cast the basis function value to the proper RBFBasisFunction
1594  basis = evaluate_basis_function(
1595  static_cast<enum RBFShape<dim>::RBFBasisFunction>(
1596  round(basis_functions[node_id])),
1597  normalized_distance);
1598  value += basis * weights[node_id];
1599  }
1600  }
1601  return value - this->layer_thickening;
1602 }

1604 template <int dim>
1605 Tensor<1, dim>
1606 RBFShape<dim>::gradient(const Point<dim> &evaluation_point,
1607  const unsigned int /*component*/) const
1608 {
1609  Point<dim> centered_point = evaluation_point;
1610  double bounding_box_distance = bounding_box->value(centered_point);
1611  Tensor<1, dim> bounding_box_gradient = bounding_box->gradient(centered_point);
1612  if (bounding_box_distance > 0.)
1613  return bounding_box_gradient;
1614 
1615  double distance, normalized_distance;
1616  Point<dim> relative_position{};
1617  // We use chain derivation to express the gradient of the basis in regards to
1618  // the position d(basis)/dx = d(basis) /d(normalized_distance)
1619  // * d(normalized_distance)/d(distance)
1620  // * d(distance) /dx
1621  double dbasis_drnorm_derivative;
1622  double drnorm_dr_derivative;
1623  Tensor<1, dim> dr_dx_derivative{};
1624  Tensor<1, dim> gradient{};
1625 
1626  if (iterable_nodes.size() == 0)
1627  {
1628  throw std::logic_error(
1629  "Every location inside the bounding box should be covered by the radius "
1630  "of at least one RBF node. It this isn't the case, an error has been "
1631  "introduced in the code. ");
1632  }
1633  // Here we loop on every portion (of RBF nodes located in active cells close
1634  // to the evaluation point) and on every RBF node located in these active
1635  // cells.
1636  for (size_t portion_id = 0; portion_id < iterable_nodes.size(); portion_id++)
1637  {
1638  for (const size_t &node_id : *(std::get<2>(iterable_nodes[portion_id])))
1639  {
1640  // Calculation of dr/dx
1641  relative_position =
1642  (evaluation_point - rotated_nodes_positions[node_id]);
1643  distance = (relative_position).norm();
1644  normalized_distance = distance / support_radii[node_id];
1645  if (distance > 1e-16 * this->effective_radius)
1646  dr_dx_derivative = relative_position / distance;
1647  else
1648  {
1649  // If the evaluation point overlaps with the node position, we
1650  // assume that the contribution of this node is 0. This assumption
1651  // can be made because radial basis functions are symmetrical. If
1652  // can be made because radial basis functions are symmetrical. If
1653  // the basis function is not differentiable at its node (e.g.
1654  // linear function), this approximation will still hold since the
1655  // approximated distance field is already imperfect and
1656  // neighboring nodes will add their contribution at the evaluation
1657  // point.
1658  for (int d = 0; d < dim; d++)
1659  dr_dx_derivative[d] = 0;
1660  }
1661  // Calculation of dr_norm/dr
1662  drnorm_dr_derivative = 1.0 / support_radii[node_id];
1663  // Calculation of d(basis)/dr
1664  // We cast the basis function value to the proper RBFBasisFunction
1665  dbasis_drnorm_derivative = evaluate_basis_function_derivative(
1666  static_cast<enum RBFShape<dim>::RBFBasisFunction>(
1667  round(basis_functions[node_id])),
1668  normalized_distance);
1669  // Sum
1670  gradient += dbasis_drnorm_derivative * drnorm_dr_derivative *
1671  dr_dx_derivative * weights[node_id];
1672  }
1673  }
1674  return gradient;
1675 }

1725 template <int dim>
1726 void
1727 RBFShape<dim>::update_precalculations(DoFHandler<dim> &dof_handler,
1728  const bool mesh_based_precalculations)
1729 {
1730  if (!mesh_based_precalculations)
1731  return;
1732  // We first reset the mapping, since the grid partitioning may change between
1733  // calls of this function. The precalculation cost is low enough that this
1734  // reset does not have a significant impact on global computational cost.
1735  likely_nodes_map.clear();
1736 
1737  this->dof_handler = &dof_handler;
1738  const unsigned int maximal_level = dof_handler.get_triangulation().n_levels();
1739 
1740  // We start by dividing the list of RBF nodes into manageable portions
1741  number_of_nodes = std::get<2>(iterable_nodes[0])->size();
1742  std::shared_ptr<std::vector<size_t>> temporary_node_numbers =
1743  std::make_shared<std::vector<size_t>>(number_of_nodes);
1744  std::iota(std::begin(*temporary_node_numbers),
1745  std::end(*temporary_node_numbers),
1746  0);
1747  std::map<const typename DoFHandler<dim>::cell_iterator,
1748  std::tuple<Point<dim>, double, std::shared_ptr<std::vector<size_t>>>>
1749  temporary_nodes_portions_map;
1750  temporary_nodes_portions_map.clear();
1751  for (unsigned int level = 0; level < maximal_level; level++)
1752  {
1753  max_number_of_inside_nodes = 1;
1754  const auto &cell_iterator = dof_handler.cell_iterators_on_level(level);
1755  for (const auto &cell : cell_iterator)
1756  {
1757  if (cell->is_artificial_on_level())
1758  continue;
1759  if (cell->level() > 0)
1760  try
1761  {
1762  const auto cell_parent = cell->parent();
1763  const auto parent_iterator =
1764  temporary_nodes_portions_map.find(cell_parent);
1765  if (parent_iterator != temporary_nodes_portions_map.end())
1766  {
1767  temporary_node_numbers =
1768  std::get<2>(parent_iterator->second);
1769  }
1770  else
1771  // If the parent doesn't contain any node, the child will not
1772  // either
1773  continue;
1774  }
1775  catch (TriaAccessorExceptions::ExcCellHasNoParent())
1776  {}
1777 
1778  std::shared_ptr<std::vector<size_t>> current_cell_nodes =
1779  std::make_shared<std::vector<size_t>>();
1780  current_cell_nodes->reserve(max_number_of_inside_nodes);
1781  for (unsigned int i = 0; i < temporary_node_numbers->size(); i++)
1782  {
1783  if (cell->point_inside(
1784  rotated_nodes_positions[(*temporary_node_numbers)[i]]))
1785  current_cell_nodes->push_back(temporary_node_numbers->at(i));
1786  }
1787  max_number_of_inside_nodes =
1788  std::max(max_number_of_inside_nodes, current_cell_nodes->size());
1789  current_cell_nodes->shrink_to_fit();
1790  if (!current_cell_nodes->empty())
1791  temporary_nodes_portions_map[cell] =
1792  std::make_tuple(cell->barycenter(),
1793  cell->diameter(),
1794  current_cell_nodes);
1795 
1796  // We currently ignore RBF nodes that are in no cells
1797  // A fix for this will be implemented shortly
1798  }
1799  // We need to remove all cells that are not needed anymore
1800  // from the map
1801  for (auto it = temporary_nodes_portions_map.cbegin();
1802  it != temporary_nodes_portions_map.cend();)
1803  {
1804  auto cell = it->first;
1805  const unsigned int cell_level = cell->level();
1806  bool cell_still_needed = cell->is_active() || cell_level == level;
1807 
1808  auto previous_it = it;
1809  ++it;
1810  if (!cell_still_needed)
1811  temporary_nodes_portions_map.erase(previous_it);
1812  }
1813  }
1814 
1815  // We give all the subsets to the 0 level, as an initial partitioning
1816  const auto &cell_iterator = dof_handler.cell_iterators_on_level(0);
1817  typename DoFHandler<dim>::cell_iterator temp_cell;
1818  std::tuple<Point<dim>, double, std::shared_ptr<std::vector<size_t>>>
1819  temp_cell_tuple{};
1820  for (const auto &cell : cell_iterator)
1821  {
1822  if (cell->is_artificial_on_level())
1823  continue;
1824 
1825  likely_nodes_map[cell] = std::make_shared<std::vector<
1826  std::
1827  tuple<Point<dim>, double, std::shared_ptr<std::vector<size_t>>>>>();
1828  for (auto it = temporary_nodes_portions_map.cbegin();
1829  it != temporary_nodes_portions_map.cend();
1830  it++)
1831  {
1832  temp_cell = it->first;
1833  if (temp_cell->is_active())
1834  {
1835  temp_cell_tuple = std::make_tuple(temp_cell->barycenter(),
1836  temp_cell->diameter(),
1837  std::get<2>(it->second));
1838  likely_nodes_map[cell]->push_back(temp_cell_tuple);
1839  }
1840  }
1841  }
1842 
1843  // We then treat all other levels
1844  for (unsigned int level = 1; level < maximal_level; level++)
1845  {
1846  const auto &cell_iterator_on_level =
1847  dof_handler.cell_iterators_on_level(level);
1848  for (const auto &cell : cell_iterator_on_level)
1849  {
1850  if (cell->is_artificial_on_level())
1851  continue;
1852 
1853  // We also check if the cell diameter is lower than the minimal
1854  // support radius. In that case, the likely nodes should stay more
1855  // or less the same.

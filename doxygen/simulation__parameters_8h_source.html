143  void
144  parse(ParameterHandler &prm)
145  {
146  dimensionality.parse_parameters(prm);
147  test.parse_parameters(prm);
148 
149  for (auto physics_name : physics_names)
150  {
151  PhysicsID physics_id = get_physics_id(physics_name);
152  linear_solver[physics_id].parse_parameters(prm, physics_name);
153  non_linear_solver[physics_id].parse_parameters(prm, physics_name);
154  }

157  mesh.parse_parameters(prm);

161  multiphysics.parse_parameters(prm);
162  timer.parse_parameters(prm);
163  fem_parameters.parse_parameters(prm);
164  laser_parameters->parse_parameters(prm);
165  forces_parameters.parse_parameters(prm);

167  flow_control.parse_parameters(prm);
168  restart_parameters.parse_parameters(prm);

170  boundary_conditions_ht.parse_parameters(prm);
171  boundary_conditions_tracer.parse_parameters(prm);
172  boundary_conditions_vof.parse_parameters(prm);
173  boundary_conditions_cahn_hilliard.parse_parameters(prm);
174  manifolds_parameters.parse_parameters(prm);
175  initial_condition->parse_parameters(prm);
176  analytical_solution->parse_parameters(prm);
177  source_term.parse_parameters(prm);
178  simulation_control.parse_parameters(prm);
179  velocity_sources.parse_parameters(prm);
180  particlesParameters->parse_parameters(prm);
181  multiphysics.parse_parameters(prm);
182  stabilization.parse_parameters(prm);
183  ale.parse_parameters(prm);
184  evaporation.parse_parameters(prm);
185 
186  physical_properties_manager.initialize(physical_properties);
187 
188 
189  // Check consistency of parameters parsed in different subsections
190  if (multiphysics.VOF && physical_properties.number_of_fluids != 2)
191  {
192  throw std::logic_error(
193  "Inconsistency in .prm!\n with VOF = true\n use: number of fluids = 2");
194  }
195 
196  if (not(multiphysics.VOF) && post_processing.postprocessed_fluid ==
197  Parameters::FluidIndicator::fluid1)
198  {
199  throw std::logic_error(
200  "Inconsistency in .prm!\n when VOF = false"
201  "\n use (default value): set postprocessed fluid = both"
202  "\n or: set postprocessed fluid = fluid 0");
203  }
204 
205  if (multiphysics.vof_parameters.sharpening.type ==
206  Parameters::SharpeningType::adaptive &&
207  not(multiphysics.vof_parameters.conservation.monitoring))
208  {
209  throw std::logic_error(
210  "Inconsistency in .prm!\n in subsection VOF, with sharpening type = adaptive\n "
211  "use: monitoring = true");
212  }
213 
214  // Interface physical property models consistency check
215  if (multiphysics.vof_parameters.surface_tension_force.enable)
216  {
217  std::string constant_surface_tension_model(
218  " subsection fluid-fluid interaction\n"
219  " set first fluid id = 0\n"
220  " set second fluid id = 1\n"
221  " set surface tension model = constant\n"
222  " set surface tension coefficient = $value_of_coefficient\n"
223  " end\n");
224 
225  std::string linear_surface_tension_model(
226  " subsection fluid-fluid interaction\n"
227  " set first fluid id = 0\n"
228  " set second fluid id = 1\n"
229  " set surface tension model = linear\n"
230  " set surface tension coefficient = $value_of_coefficient\n"
231  " set temperature-driven surface tension gradient = $value_of_gradient\n"
232  " end\n");
233 
234  std::string phase_change_surface_tension_model(
235  " subsection fluid-fluid interaction\n"
236  " set first fluid id = 0\n"
237  " set second fluid id = 1\n"
238  " set surface tension model = phase change\n"
239  " set surface tension coefficient = $value_of_coefficient\n"
240  " set temperature-driven surface tension gradient = $value_of_gradient\n"
241  " set solidus temperature = $value_of_solidus_temperature\n"
242  " set liquidus temperature = $value_of_liquidus_temperature\n"
243  " end\n");
244  if (!multiphysics.vof_parameters.surface_tension_force
245  .enable_marangoni_effect) // constant surface tension model
246  {
247  if (physical_properties.number_of_material_interactions == 0)
248  {
249  throw std::logic_error(
250  "Inconsistency in .prm!\n "
251  "In subsection VOF, with surface tension force enabled,\n "
252  "but no material interactions specified in\n "
253  "subsection physical properties.\n "
254  "Use:\n\n"
255  " set number of material interactions = 1\n"
256  " subsection material interaction 0\n"
257  " set type = fluid-fluid\n" +
258  constant_surface_tension_model + " end\n");
259  }
260  else if (multiphysics.VOF &&
261  multiphysics
262  .heat_transfer) // disabled Marangoni effect error
263  {
264  if (!is_constant_surface_tension_model(
265  physical_properties.material_interactions))
266  {
267  throw std::logic_error(
268  "Inconsistency in .prm!\n "
269  "In subsection multiphysics, VOF and heat transfer enabled,\n "
270  "and in subsection physical properties, a non-constant surface\n "
271  "tension model, but Marangoni effect disabled in subsection\n "
272  "surface tension force of subsection VOF. This is necessary to account\n "
273  "for Marangoni effect. In subsection VOF, use:\n\n "
274  " subsection surface tension force\n"
275  " set enable = true\n"
276  " set enable marangoni effect = true\n"
277  " end\n");
278  }
279  }
280  else
281  {
282  if (no_fluid_fluid_interaction_error(
283  physical_properties.material_interactions))
284  {
285  throw std::logic_error(
286  "Inconsistency in .prm!\n "
287  "in subsection VOF, surface tension force enabled,\n "
288  "but no fluid-fluid material interactions specified in \n "
289  "subsection physical properties\n "
290  "Use:\n\n"
291  " subsection material interaction $material_interaction_id\n"
292  " set type = fluid-fluid\n" +
293  constant_surface_tension_model + " end\n");
294  }
295  }
296  }
297  else // non-constant surface tension model
298  {
299  if (physical_properties.number_of_material_interactions == 0)
300  {
301  throw std::logic_error(
302  "Inconsistency in .prm!\n "
303  "In subsection VOF, marangoni effect enabled,\n "
304  "but no material interactions specified in subsection physical\n "
305  "properties. This is necessary to account for Marangoni \n "
306  "effect. In subsection physical properties, use:\n\n"
307  " set number of material interactions = 1\n"
308  " subsection material interaction 0\n"
309  " set type = fluid-fluid\n" +
310  linear_surface_tension_model +
311  "\n"
312  "or: \n\n"
313  " set number of material interactions = 1\n"
314  " subsection material interaction 0\n"
315  " set type = fluid-fluid\n" +
316  phase_change_surface_tension_model + " end\n");
317  }
318  else
319  {
320  if (no_fluid_fluid_interaction_error(
321  physical_properties.material_interactions))
322  {
323  throw std::logic_error(
324  "Inconsistency in .prm!\n "
325  "In subsection VOF, Marangoni effect enabled,\n "
326  "but no fluid-fluid material interactions specified in subsection\n "
327  "physical properties. This is necessary to account for Marangoni\n "
328  "effect. In subsection physical properties, use:\n\n"
329  " subsection material interaction $material_interaction_id\n"
330  " set type = fluid-fluid\n" +
331  linear_surface_tension_model +
332  "\n"
333  "or:\n\n"
334  " set number of material interactions = 1\n"
335  " subsection material interaction 0\n"
336  " set type = fluid-fluid\n" +
337  phase_change_surface_tension_model + " end\n");
338  }
339  if (is_constant_surface_tension_model(
340  physical_properties.material_interactions))
341  {
342  throw std::logic_error(
343  "Inconsistency in .prm!\n "
344  "In subsection VOF, Marangoni effect enabled,\n "
345  "but a constant surface tension model is specified in subsection\n "
346  "physical properties. This is necessary to account for Marangoni \n "
347  "effect. In subsection physical properties, use:\n\n"
348  " subsection material interaction $material_interaction_id\n"
349  " set type = fluid-fluid\n" +
350  linear_surface_tension_model +
351  "\n"
352  "or:\n\n"
353  " set number of material interactions = 1\n"
354  " subsection material interaction 0\n"
355  " set type = fluid-fluid\n" +
356  phase_change_surface_tension_model + " end\n");
357  }
358  }
359  }
360  }
361 
362  if (multiphysics.cahn_hilliard && multiphysics.VOF)
363  {
364  throw std::runtime_error(
365  "Cannot solve a multiphase problem using VOF and Cahn-Hilliard at the same time");
366  }
367 
368  if (multiphysics.cahn_hilliard &&
369  physical_properties.number_of_material_interactions == 0)
370  {
371  throw std::logic_error(
372  "Inconsistency in .prm!\n "
373  "cahn hilliard = true, \n "
374  "but no fluid-fluid material interactions specified in\n "
375  "subsection physical properties\n "
376  "Use:\n\n"
377  " subsection material interaction $material_interaction_id\n"
378  " set type = fluid-fluid\n"
379  " subsection fluid-fluid interaction\n"
380  " set first fluid id = 0\n"
381  " set second fluid id = 1\n"
382  " set cahn hilliard mobility model = constant\n"
383  " set cahn hilliard mobility coefficient = $value_of_coefficient\n"
384  " end\n"
385  " end\n");
386  }
387  if (laser_parameters->activate_laser &&
388  laser_parameters->laser_type ==
389  Parameters::Laser<dim>::LaserType::heat_flux_vof_interface &&
390  !multiphysics.VOF)
391  {
392  throw std::logic_error(
393  "At the moment, the laser surface heat flux is not implemented for 1 fluid simulations."
394  "Please enable the VOF auxiliary physic in the 'multiphysics' subsection, \n"
395  "specify a 2nd fluid in the 'physical properties' subsection,\n"
396  "and define appropriate initial conditions in the 'initial conditions' subsection.");
397  }

400  inline bool
401  no_fluid_fluid_interaction_error(
402  std::vector<Parameters::MaterialInteractions> &material_interactions)
403  {
404  for (const Parameters::MaterialInteractions &material_interaction :
405  material_interactions)
406  {
407  if (material_interaction.material_interaction_type ==
408  Parameters::MaterialInteractions::MaterialInteractionsType::
409  fluid_fluid)
410  {
411  return false;
412  }
413  }
414  return true;
415  }
416 
417  inline bool
418  is_constant_surface_tension_model(
419  std::vector<Parameters::MaterialInteractions> &material_interactions)
420  {
421  for (const Parameters::MaterialInteractions &material_interaction :
422  material_interactions)
423  {
424  if (material_interaction.surface_tension_model !=
425  Parameters::MaterialInteractions::SurfaceTensionModel::constant)
426  {
427  return false;
428  }
429  }
430  return true;
431  }

525  // Surface tension gradient with respect to the temperature

528 
529  const double &curvature_value = scratch_data.curvature_values[q];

539  filtered_phase_gradient_value_q /

541

553  -surface_tension_gradient *

557  filtered_phase_gradient_norm;

637 template <int dim>
638 void
639 GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix(
640  NavierStokesScratchData<dim> &scratch_data,
641  StabilizedMethodsTensorCopyData<dim> &copy_data)
642 {
643  // Loop and quadrature information
644  const auto &JxW_vec = scratch_data.JxW;
645  const unsigned int n_q_points = scratch_data.n_q_points;
646  const unsigned int n_dofs = scratch_data.n_dofs;
647  const double h = scratch_data.cell_size;
648 
649  // Copy data elements
650  auto &strong_residual_vec = copy_data.strong_residual;
651  auto &strong_jacobian_vec = copy_data.strong_jacobian;
652  auto &local_matrix = copy_data.local_matrix;
653 
654  // Time steps and inverse time steps which is used for stabilization constant
655  std::vector<double> time_steps_vector =
656  this->simulation_control->get_time_steps_vector();
657  const double dt = time_steps_vector[0];
658  const double sdt = 1. / dt;
659 
660  Assert(
661  scratch_data.properties_manager.density_is_constant(),
662  RequiresConstantDensity(
663  "GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix"));
664 
665  // Loop over the quadrature points
666  for (unsigned int q = 0; q < n_q_points; ++q)
667  {
668  // Gather into local variables the relevant fields
669  const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
670  const Tensor<2, dim> &velocity_gradient =
671  scratch_data.velocity_gradients[q];
672  const Tensor<1, dim> &velocity_laplacian =
673  scratch_data.velocity_laplacians[q];
674  const Tensor<3, dim> &velocity_hessian =
675  scratch_data.velocity_hessians[q];
676  const Tensor<1, dim> &pressure_gradient =
677  scratch_data.pressure_gradients[q];
678 
679  // Calculate shear rate (at each q)
680  const Tensor<2, dim> shear_rate =
681  velocity_gradient + transpose(velocity_gradient);
682 
683  // Calculate the shear rate magnitude
684  double shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
685  // Set the shear rate magnitude to 1e-12 if it is too close to zero,
686  // since the viscosity gradient is undefined for shear_rate_magnitude = 0
687  shear_rate_magnitude =
688  shear_rate_magnitude > 1e-12 ? shear_rate_magnitude : 1e-12;
689 
690  // Calculate kinematic viscosity gradient
691  const Tensor<1, dim> kinematic_viscosity_gradient =
692  this->get_kinematic_viscosity_gradient(
693  velocity_gradient,
694  velocity_hessian,
695  shear_rate_magnitude,
696  scratch_data.grad_kinematic_viscosity_shear_rate[q]);
697 
698  // Forcing term
699  Tensor<1, dim> force = scratch_data.force[q];
700 
701  // Calculation of the magnitude of the velocity for the
702  // stabilization parameter
703  const double u_mag = std::max(velocity.norm(), 1e-12);
704 
705  // Store JxW in local variable for faster access;
706  const double JxW = JxW_vec[q];
707 
708  // Calculation of the equivalent properties at the quadrature point
709  double density_eq = scratch_data.density[q];
710  const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
711  const Tensor<1, dim> dynamic_viscosity_gradient =
712  density_eq * kinematic_viscosity_gradient;
713 
714  // Calculation of the GLS stabilization parameter. The
715  // stabilization parameter used is different if the simulation
716  // is steady or unsteady. In the unsteady case it includes the
717  // value of the time-step
718  const double tau =
719  this->simulation_control->get_assembly_method() ==
720  Parameters::SimulationControl::TimeSteppingMethod::steady ?
721  calculate_navier_stokes_gls_tau_steady(
722  u_mag, dynamic_viscosity_eq / density_eq, h) :
723  calculate_navier_stokes_gls_tau_transient(
724  u_mag, dynamic_viscosity_eq / density_eq, h, sdt);
725 
726  // Calculate the strong residual for GLS stabilization
727  auto strong_residual = density_eq * velocity_gradient * velocity +
728  pressure_gradient -
729  shear_rate * dynamic_viscosity_gradient -
730  dynamic_viscosity_eq * velocity_laplacian -
731  density_eq * force + strong_residual_vec[q];
732 
733  std::vector<Tensor<1, dim>> grad_phi_u_j_x_velocity(n_dofs);
734  std::vector<Tensor<1, dim>> velocity_gradient_x_phi_u_j(n_dofs);
735 
736  // Pressure scaling factor
737  const double pressure_scaling_factor =
738  scratch_data.pressure_scaling_factor;
739 
740  // We loop over the column first to prevent recalculation
741  // of the strong jacobian in the inner loop
742  for (unsigned int j = 0; j < n_dofs; ++j)
743  {
744  const auto &phi_u_j = scratch_data.phi_u[q][j];
745  const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
746  const auto &laplacian_phi_u_j = scratch_data.laplacian_phi_u[q][j];
747 
748  const auto &grad_phi_p_j =
749  scratch_data.grad_phi_p[q][j] * pressure_scaling_factor;
750 
751  const auto &grad_phi_u_j_non_newtonian =
752  grad_phi_u_j + transpose(grad_phi_u_j);
753 
754  strong_jacobian_vec[q][j] +=
755  density_eq * velocity_gradient * phi_u_j +
756  density_eq * grad_phi_u_j * velocity + grad_phi_p_j -
757  dynamic_viscosity_eq * laplacian_phi_u_j -
758  grad_phi_u_j_non_newtonian * dynamic_viscosity_gradient;
759 
760  // Store these temporary products in auxiliary variables for speed
761  grad_phi_u_j_x_velocity[j] = grad_phi_u_j * velocity;
762  velocity_gradient_x_phi_u_j[j] = velocity_gradient * phi_u_j;
763  }
764 
765  shear_rate_magnitude =
766  shear_rate_magnitude > 1e-3 ? shear_rate_magnitude : 1e-3;
767 
768  for (unsigned int i = 0; i < n_dofs; ++i)
769  {
770  const auto &phi_u_i = scratch_data.phi_u[q][i];
771  const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
772  const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
773  const auto &phi_p_i = scratch_data.phi_p[q][i];
774  const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
775 
776  // Store these temporary products in auxiliary variables for speed
777  const auto grad_phi_u_i_x_velocity = grad_phi_u_i * velocity;
778  const auto strong_residual_x_grad_phi_u_i =
779  strong_residual * grad_phi_u_i;
780 
781  for (unsigned int j = 0; j < n_dofs; ++j)
782  {
783  const auto &phi_u_j = scratch_data.phi_u[q][j];
784  const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
785  const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
786 
787  const auto &grad_phi_u_j_non_newtonian =
788  grad_phi_u_j + transpose(grad_phi_u_j);
789 
790  const auto &phi_p_j =
791  scratch_data.phi_p[q][j] * pressure_scaling_factor;
792 
793  const auto &strong_jac = strong_jacobian_vec[q][j];
794 
795  double local_matrix_ij =
796  dynamic_viscosity_eq *
797  scalar_product(grad_phi_u_j_non_newtonian, grad_phi_u_i) +
798  0.5 * scratch_data.grad_kinematic_viscosity_shear_rate[q] /
799  shear_rate_magnitude *
800  scalar_product(grad_phi_u_j_non_newtonian, shear_rate) *
801  scalar_product(shear_rate, grad_phi_u_i) +
802  density_eq * velocity_gradient_x_phi_u_j[j] * 0.5 * phi_u_i +
803  density_eq * grad_phi_u_j_x_velocity[j] * phi_u_i -
804  div_phi_u_i * phi_p_j;
805 
806  // Continuity
807  local_matrix_ij += phi_p_i * div_phi_u_j;
808 
809  // PSPG GLS term
810  local_matrix_ij += tau * (strong_jac * grad_phi_p_i);
811 
812  // The jacobian matrix for the SUPG formulation
813  // currently does not include the jacobian of the stabilization
814  // parameter tau. Our experience has shown that does not alter the
815  // number of newton iteration for convergence, but greatly
816  // simplifies assembly.
817  if (SUPG)
818  {
819  local_matrix_ij +=
820  tau * (strong_jac * grad_phi_u_i_x_velocity +
821  strong_residual_x_grad_phi_u_i * phi_u_j);
822  }
823  local_matrix_ij *= JxW;
824  local_matrix(i, j) += local_matrix_ij;
825  }
826  }
827  }
828 }

901

903  double density_eq = scratch_data.density[q];
904  const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
905  const Tensor<1, dim> dynamic_viscosity_gradient =
906  density_eq * kinematic_viscosity_gradient;
907

918  u_mag, dynamic_viscosity_eq / density_eq, h, sdt);
919

921  // Calculate the strong residual for GLS stabilization
922  auto strong_residual = density_eq * velocity_gradient * velocity +
923  pressure_gradient -
924  shear_rate * dynamic_viscosity_gradient -
925  dynamic_viscosity_eq * velocity_laplacian -
926  density_eq * force + strong_residual_vec[q];

929  for (unsigned int i = 0; i < n_dofs; ++i)

931  const auto &phi_u_i = scratch_data.phi_u[q][i];

956  -tau * (strong_residual * (grad_phi_u_i * velocity)) * JxW;
957  }
958  }

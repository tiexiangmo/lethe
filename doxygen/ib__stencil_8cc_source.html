23 void
24 IBStencil<dim>::p_base(const unsigned int order)
25 {
26  using numbers::PI;
27  // define the sampling point position of the stencil on the reference 1D
28  // stencil 0 to 1. 1 being the position of the dof
29  reference_points.resize(order + 1);
30  for (unsigned int i = 0; i < order + 1; ++i)

36 template <int dim>
37 std::vector<double>
38 IBStencil<dim>::coefficients(const unsigned int order,
39  const double length_ratio)
40 {
41  p_base(order);
42  // Initialize the coefficient vector
43  std::vector<double> coef(order + 1);
44 
45  FullMatrix<double> vandermonde(order + 1, order + 1);
46  FullMatrix<double> stencil(order + 1, order + 1);
47  FullMatrix<double> inv_vandermonde(order + 1, order + 1);
48 
49  Vector<double> rhs(order + 1);
50  // Define the Vandermonde matrix
51  for (unsigned int i = 0; i < order + 1; ++i)
52  {
53  for (unsigned int j = 0; j < order + 1; ++j)
54  {
55  vandermonde[i][j] = std::pow(reference_points[i], j);
56  }
57  rhs[i] = std::pow(1 + length_ratio, i);
58  }
59  // Inverte the vandermond matrix
60  inv_vandermonde.invert(vandermonde);
61 
62  // Multiply each line of the inverted matrix by (1+length_ratio)^i
63  for (unsigned int i = 0; i < order + 1; ++i)
64  {
65  for (unsigned int j = 0; j < order + 1; ++j)
66  {
67  stencil[i][j] = inv_vandermonde[i][j] * rhs[i];
68  }
69  }
70  // Sum the colones to get the coefficient
71  for (unsigned int i = 0; i < order + 1; ++i)
72  {
73  for (unsigned int j = 0; j < order + 1; ++j)
74  {
75  coef[order - i] += stencil[j][i];
76  }
77  }
78 
79  // Fill the coefficient vector based on the order.
80 
81  if (order > 4)
82  {
83  // In this case the cell is directly used to find the solution at the IB
84  // position. In this case only one point is needed (the position of the
85  // point on the IB) and its coefficient is 1.
86  coef.resize(1);
87  coef[0] = 1;
88  }
89 
90  return coef;
91 }

93 template <int dim>
94 std::tuple<Point<dim>, std::vector<Point<dim>>>
95 IBStencil<dim>::support_points_for_interpolation(
96  const unsigned int order,
97  const double length_ratio,
98  IBParticle<dim> &p,
99  const Point<dim> &dof_point,
100  const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
101 {
102  // Create the vector of points used for the stencil based on the order of the
103  // stencil. Also return the DOF position or the position of the point on the
104  // IB depending if the cell is used directly
105  p_base(order);
106  Point<dim> point;
107  Point<dim> surface_point;
108  p.closest_surface_point(dof_point, surface_point, cell_guess);
109  std::vector<Point<dim>> interpolation_points;
110 
111  if (order > 4)
112  {
113  // In this case the cell is directly used to find the solution at the IB
114  // position. In this case only one point is needed (the position of the
115  // point on the IB).
116  Tensor<1, dim, double> vect_ib = dof_point - surface_point;
117  point = surface_point;
118 
119  Point<dim, double> interpolation_point_1(dof_point +
120  vect_ib * 1. / length_ratio);
121 
122  interpolation_points.resize(1);
123  interpolation_points[0] = interpolation_point_1;
124  }
125  else
126  {
127  interpolation_points.resize(order);
128  Tensor<1, dim, double> vect_ib = dof_point - surface_point;
129  point = dof_point;
130  for (unsigned int i = 1; i < order + 1; ++i)
131  {
132  interpolation_points[i - 1] =
133  dof_point +
134  vect_ib * (1 - reference_points[order - i]) / (length_ratio);
135  }
136  }
137  return {point, interpolation_points};
138 }

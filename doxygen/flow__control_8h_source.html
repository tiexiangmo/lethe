41 template <int dim>
42 class FlowControl
43 {
44 public:
45  FlowControl(const Parameters::DynamicFlowControl &flow_control);
46  /**
47  * @brief calculate_beta. This function calculates a beta coefficient which
48  * applies a force to the flow in order to adjust the flow rate to a desired
49  * average velocity through the previous flow rate. Once the flow rate is
50  * reached, the algorithm calculates a new beta to keep a constant flow rate.
51  *
52  * @param average_velocity. The last step average_velocity
53  *
54  * @param dt. The current time step
55  *
56  * @param step_number. The current step
57  */
58  void
59  calculate_beta(const double &average_velocity,
60  const double &dt,
61  const unsigned int &step_number);
62 
63  /**
64  * @brief proportional_flow_controller. This function calculates a beta only
65  * with the last step average velocity. It is used to initialize the beta
66  * if there's no initial beta value at 1st time step and the 2nd time step.
67  */
68  inline double
69  proportional_flow_controller(const double &average_velocity_n,
70  const double &dt)
71  {
72  return 0.5 * alpha * (average_velocity_0 - average_velocity_n) / dt;
73  }
74 
75  /**
76  * @brief main_flow_controller. This function calculates a beta coefficient
77  * based on a formula described by Wang, 2009 in "A non-body conformal grid
78  * method for simulations
79  * of laminar and turbulent flows with a compressible
80  * large eddy simulation solver"
81  */
82  inline double
83  main_flow_controller(const double &average_velocity_n, const double &dt)
84  {
85  double beta_n1 = beta_n + alpha *
86  (average_velocity_0 - 2 * average_velocity_n +
87  average_velocity_1n) /
88  dt;
89 
90  // If desired average velocity is reached, new beta only maintains the force
91  // to keep the flow at the desired value. If calculated beta is
92  // negative it is set to 0 to avoided +/- force.
93  if (average_velocity_0 * beta_n1 < 0 && no_force == false)
94  return 0.0;
95  else
96  return beta_n1;
97  }

99  /**
100  * @brief get_beta. This function gives the beta force of the step time
101  */
102  Tensor<1, dim>
103  get_beta()
104  {
105  return beta;
106  }
107 
108  /**
109  * @brief get_beta_particle. This function gives the beta force of the
110  * CFD time step scaled for particles.
111  */
112  inline Tensor<1, 3>
113  get_beta_particles(const double &fluid_density,
114  const double &particle_density)
115  {
116  // The beta force for particles must have a tensor of dim = 3 since the body
117  // force g parameters is a tensor of dim 3 (those forces will add up)
118  Tensor<1, 3> beta_particle;
119  beta_particle[0] = beta[0];
120  beta_particle[1] = beta[1];
121 
122  if constexpr (dim == 3)
123  beta_particle[2] = beta[2];
124 
125  // From beta_p * rho_p = beta_f * rho_f [=] M/(T²L²) units of ΔP/L
126  return beta_particle * fluid_density / particle_density;
127  }

32 /**
33  * Finds the neighbors lists of all the active cells in the input triangulation.
34  * find_cell_neighbors() is written to avoid any repetition, for instance if
35  * cell B is recognized as the neighbor of cell A once, cell A will not appear
36  * in the neighbor list of cell B again. On the other hand,
37  * find_full_cell_neighbors() function finds the neighbors list with this
38  * repetition.
39  *
40  * @note
41  *
42  * @author Shahab Golshan, Polytechnique Montreal 2019-
43  */
44 
45 template <int dim>
46 class FindCellNeighbors
47 {
48 public:
49  FindCellNeighbors();
50 
51  /**
52  * @brief Finds the neighbor list (without repetition) of all the active
53  * cells in the triangulation. It gets the vertices of the cells to get lists
54  * of the neighbor for each cells. There is some check to prevent repetition
55  * of a cell in a list (up to 4 vertices can have the same cell in common in
56  * 3D). 2 types of container are used for cell neighbors : local-local cells
57  * and local-ghost cells.
58  *
59  * @param triangulation Triangulation to access the information of the cells
60  * @param cells_local_neighbor_list A vector (with size of the local cell
61  * number) of vectors (local adjacent cells of each local cell). First element
62  * of each set shows the main cell itself
63  * @param cells_ghost_neighbor_list A vector (with size of the local cell
64  * number) of vectors (ghost adjacent cells of each local cell). First element
65  * of each set shows the main cell itself
66  */
67 
68  void
69  find_cell_neighbors(
70  const parallel::distributed::Triangulation<dim> &triangulation,
71  typename DEM::dem_data_structures<dim>::cells_neighbor_list
72  &cells_local_neighbor_list,
73  typename DEM::dem_data_structures<dim>::cells_neighbor_list
74  &cells_ghost_neighbor_list);
75 
76  /**
77  * @brief Finds the periodic neighbor list (without repetition) of all the
78  * active cells in the triangulation. It gets the coinciding vertices of the
79  * cells at the periodic boundary 0 to get lists of the periodic neighbor
80  * cells on the periodic boundary 1 for each cells. There is some check to
81  * prevent repetition of a cell in a list (up to 4 vertices can have the
82  * same cell in common in 3D). Also, 3 types of container are used for
83  * periodic mapping of cell neighbors : local-local cells, local-ghost cells
84  * and ghost-local cells. The last container is necessary since the mapping
85  * are only from periodic boundary 0 to periodic boundary 1 and the
86  * ghost-local particle pairs need distinction for proper handling of search
87  * of particle pairs and contact forces.
88  *
89  * @param triangulation Triangulation to access the information of the cells
90  * @param periodic_boundaries_cells_information A container of information
91  * related to the pairs of cell at periodic boundaries, used to get the cells
92  * on periodic boundary 0
93  * @param cells_local_periodic_neighbor_list A vector (with size of the local
94  * cell number) of vectors (local adjacent cells of each local cell). First
95  * element of each set shows the main cell itself
96  * @param cells_ghost_periodic_neighbor_list A vector (with size of the local
97  * cell number) of vectors (ghost adjacent cells of each local cell). First
98  * element of each set shows the main cell itself
99  * @param cells_ghost_local_periodic_neighbor_list A vector (with size of the
100  * ghost cell number) of vectors (local adjacent cells of each ghost cell).
101  * First element of each set shows the main ghost cell itself
102  */
103

116  /**
117  * Finds the full neighbor list (with repetition) of all the active cells in
118  * the triangulation. This function is used in particle-floating mesh
119  * contacts, as in this situation, we need to define all the particles located
120  * in the neighbor cells of the background cell (cut by the floating mesh) as
121  * contact candidates
122  *
123  * @param triangulation Triangulation to access the information of the cells
124  * @param cells_total_neighbor_list An unordered_map (with size of the local
125  * cell number) of vectors (all adjacent cells of each local cell)
126  */
127 
128  void
129  find_full_cell_neighbors(
130  const parallel::distributed::Triangulation<dim> &triangulation,
131  typename DEM::dem_data_structures<dim>::cells_total_neighbor_list
132  &cells_total_neighbor_list);
133 
134 private:
135  /**
136  * @brief Generate a periodic neighbor cells list of the cell. With the
137  * coinciding_vertex_groups and the vertex_to_coinciding_vertex_group, we can
138  * get the coinciding vertices on a periodic boundary and with v_to_c we can
139  * get the list of the periodic neighbor cells to the main cell.
140  *
141  * Here is an example to understand how the search works :
142  * Cell 8 have vertices 0, 1, 2, 3 on the periodic boundary. First,
143  * it checks in the coinciding_vertex_groups if the vertex 0 is a key in the
144  * map, and if it is, it gets a label, let's say label is 10. In the
145  * vertex_to_coinciding_vertex_group, it can finds the the coinciding vertices
146  * with the label. Using the label 10, it gets the vertices 0 and 34, which
147  * means that vertices 0 and 34 are periodic. We do not want the cells
148  * attached to the vertex 0 since they are already found with the regular
149  * find_cell_neighbor function, so it skips vertex 0, but gets the cells 21,
150  * 22, 23, 24 attached to the vertex 34. The same checks are done for vertices
151  * 1, 2 and 3 and all periodic cells are return as a vector.
152  *
153  * @param cell The cell that needs the periodic neighbor list
154  * @param coinciding_vertex_groups A map of coinciding vertices labeled by an
155  * arbitrary element from them
156  * @param vertex_to_coinciding_vertex_group Map of a vertex to the label of a
157  * group of coinciding vertices
158  * @param v_to_c A vector of set with adjacent cells of all the vertices
159  * @param periodic_neighbor_list A vector which is the list of periodic cell
160  * neighbors
161  */
162 
163  void
164  get_periodic_neighbor_list(
165  const typename Triangulation<dim>::active_cell_iterator &cell,
166  const std::map<unsigned int, std::vector<unsigned int>>
167  &coinciding_vertex_groups,
168  const std::map<unsigned int, unsigned int>
169  &vertex_to_coinciding_vertex_group,
170  const std::vector<
171  std::set<typename Triangulation<dim>::active_cell_iterator>> &v_to_c,
172  typename DEM::dem_data_structures<dim>::cell_vector
173  &periodic_neighbor_list);
174 };

196 IBParticle<dim>::is_inside_crown(
197  const Point<dim> &evaluation_point,
198  const double outer_radius,
199  const double inside_radius,
200  const bool absolute_distance,
201  const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
202 {
203  const double radius = shape->effective_radius;
204 
205  double distance = shape->value_with_cell_guess(evaluation_point, cell_guess);
206  bool is_inside_outer_ring;
207  bool is_outside_inner_ring;
208  if (absolute_distance)
209  {
210  is_inside_outer_ring = distance <= outer_radius;
211  is_outside_inner_ring = distance >= inside_radius;
212  }
213  else
214  {
215  is_inside_outer_ring = distance <= radius * (outer_radius - 1);
216  is_outside_inner_ring = distance >= radius * (inside_radius - 1);
217  }
218 
219  return is_inside_outer_ring && is_outside_inner_ring;
220 }

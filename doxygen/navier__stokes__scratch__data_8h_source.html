48 /**
49  * @brief Class that stores the information required by the assembly procedure
50  * for a Navier-Stokes equation. Consequently, this class calculates
51  * the velocity (values, gradients, laplacians) and the shape function
52  * (values, gradients, laplacians) at all the gauss points for all degrees
53  * of freedom and stores it into arrays. Additionally, the use can request
54  * that this class gathers additional fields for physics which are coupled
55  * to the Navier-Stokes equation, such as the VOF. This class
56  * serves as a separation between the evaluation at the gauss point of the
57  * variables of interest and their use in the assembly, which is carried out
58  * by the assembler functions. For more information on this design, the reader
59  * can consult deal.II step-9
60  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
61  * example, the scratch is a struct instead of a templated class because of the
62  * simplicity of step-9.
63  *
64  * @tparam dim An integer that denotes the dimension of the space in which
65  * the flow is solved
66  * @ingroup solvers
67  **/

202  /** @brief Reinitialize the content of the scratch
203  *
204  * Using the FeValues and the content of the solutions and previous solutions,
205  * fills all of the class member of the scratch
206  *
207  * @param cell The cell over which the assembly is being carried.
208  * This cell must be compatible with the fe which is used to fill the FeValues
209  *
210  * @param current_solution The present value of the solution for [u,p]
211  *
212  * @param previous_solutions The solutions at the previous time steps
213  *
214  *
215  * @param forcing_function The function describing the momentum/mass source term
216  *
217  * @param beta_force The additional force for flow control. TODO : Deprecate this argument and pass it
218  * to the constructor of the assembler
219  *
220  */
221 
222  template <typename VectorType>
223  void
224  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
225  const VectorType &current_solution,
226  const std::vector<VectorType> &previous_solutions,
227  std::shared_ptr<Function<dim>> forcing_function,
228  Tensor<1, dim> beta_force,
229  const double pressure_scaling_factor)
230  {
231  this->fe_values.reinit(cell);
232 
233  quadrature_points = this->fe_values.get_quadrature_points();
234  auto &fe = this->fe_values.get_fe();
235 
236  forcing_function->vector_value_list(quadrature_points, this->rhs_force);
237 
238  // Establish the force vector
239  for (unsigned int q = 0; q < n_q_points; ++q)
240  {
241  for (int d = 0; d < dim; ++d)
242  {
243  const unsigned int component_i =
244  fe.system_to_component_index(d).first;
245  this->force[q][d] = this->rhs_force[q](component_i);
246  }
247 
248  const unsigned int component_mass =
249  fe.system_to_component_index(dim).first;
250  this->mass_source[q] = this->rhs_force[q](component_mass);
251 
252  // Correct force to include the dynamic forcing term for flow
253  // control
254  force[q] = force[q] + beta_force;
255  }
256 
257  for (const unsigned int k : fe_values.dof_indices())
258  {
259  components[k] = fe.system_to_component_index(k).first;
260  }
261 
262  if (dim == 2)
263  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe.degree;
264  else if (dim == 3)
265  this->cell_size = pow(6 * cell->measure() / M_PI, 1. / 3.) / fe.degree;
266 
267  // Gather velocity (values, gradient and laplacian)
268  this->fe_values[velocities].get_function_values(current_solution,
269  this->velocity_values);
270  this->fe_values[velocities].get_function_gradients(
271  current_solution, this->velocity_gradients);
272  this->fe_values[velocities].get_function_laplacians(
273  current_solution, this->velocity_laplacians);
274  if (gather_hessian)
275  this->fe_values[velocities].get_function_hessians(
276  current_solution, this->velocity_hessians);
277 
278  for (unsigned int q = 0; q < this->n_q_points; ++q)
279  {
280  this->velocity_divergences[q] = trace(this->velocity_gradients[q]);
281  }
282 
283  // Gather pressure (values, gradient)
284  fe_values[pressure].get_function_values(current_solution,
285  this->pressure_values);
286  fe_values[pressure].get_function_gradients(current_solution,
287  this->pressure_gradients);
288  this->pressure_scaling_factor = pressure_scaling_factor;
289 
290  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
291  {
292  this->fe_values[velocities].get_function_values(
293  previous_solutions[p], previous_velocity_values[p]);
294  }
295 
296  // Only gather the pressure when a pressure history is necessary
297  // (compressible Navier-Stokes)
298  if (!this->properties_manager.density_is_constant())
299  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
300  {
301  this->fe_values[pressure].get_function_values(
302  previous_solutions[p], previous_pressure_values[p]);
303  }
304 
305  for (unsigned int q = 0; q < n_q_points; ++q)
306  {
307  this->JxW[q] = this->fe_values.JxW(q);
308  for (unsigned int k = 0; k < n_dofs; ++k)
309  {
310  // Velocity
311  this->phi_u[q][k] = this->fe_values[velocities].value(k, q);
312  this->div_phi_u[q][k] =
313  this->fe_values[velocities].divergence(k, q);
314  this->grad_phi_u[q][k] = this->fe_values[velocities].gradient(k, q);
315  this->hess_phi_u[q][k] = this->fe_values[velocities].hessian(k, q);
316  for (int d = 0; d < dim; ++d)
317  this->laplacian_phi_u[q][k][d] = trace(this->hess_phi_u[q][k][d]);
318  // Pressure
319  this->phi_p[q][k] = this->fe_values[pressure].value(k, q);
320  this->grad_phi_p[q][k] = this->fe_values[pressure].gradient(k, q);
321  }
322  }
323 
324  is_boundary_cell = cell->at_boundary();
325  if (is_boundary_cell)
326  {
327  n_faces = cell->n_faces();
328  is_boundary_face = std::vector<bool>(n_faces, false);
329  n_faces_q_points = fe_face_values.get_quadrature().size();
330  boundary_face_id = std::vector<unsigned int>(n_faces);
331 
332  face_JxW = std::vector<std::vector<double>>(
333  n_faces, std::vector<double>(n_faces_q_points));
334 
335 
336  // Velocity and pressure values
337  // First vector is face number, second quadrature point
338  this->face_velocity_values = std::vector<std::vector<Tensor<1, dim>>>(
339  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
340 
341  this->face_velocity_divergences = std::vector<std::vector<double>>(
342  n_faces, std::vector<double>(n_faces_q_points));
343 
344  this->face_velocity_gradients =
345  std::vector<std::vector<Tensor<2, dim>>>(
346  n_faces, std::vector<Tensor<2, dim>>(n_faces_q_points));
347 
348  this->face_velocity_laplacians =
349  std::vector<std::vector<Tensor<1, dim>>>(
350  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
351 
352  this->face_pressure_values = std::vector<std::vector<double>>(
353  n_faces, std::vector<double>(n_faces_q_points));
354 
355  this->face_pressure_gradients =
356  std::vector<std::vector<Tensor<1, dim>>>(
357  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
358 
359  this->face_normal = std::vector<std::vector<Tensor<1, dim>>>(
360  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
361 
362  this->face_quadrature_points = std::vector<std::vector<Point<dim>>>(
363  n_faces, std::vector<Point<dim>>(n_faces_q_points));
364 
365  this->face_div_phi_u = std::vector<std::vector<std::vector<double>>>(
366  n_faces,
367  std::vector<std::vector<double>>(n_faces_q_points,
368  std::vector<double>(n_dofs)));
369 
370  this->face_phi_u =
371  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
372  n_faces,
373  std::vector<std::vector<Tensor<1, dim>>>(
374  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
375 
376  this->face_hess_phi_u =
377  std::vector<std::vector<std::vector<Tensor<3, dim>>>>(
378  n_faces,
379  std::vector<std::vector<Tensor<3, dim>>>(
380  n_faces_q_points, std::vector<Tensor<3, dim>>(n_dofs)));
381 
382  this->face_laplacian_phi_u =
383  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
384  n_faces,
385  std::vector<std::vector<Tensor<1, dim>>>(
386  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
387 
388  this->face_grad_phi_u =
389  std::vector<std::vector<std::vector<Tensor<2, dim>>>>(
390  n_faces,
391  std::vector<std::vector<Tensor<2, dim>>>(
392  n_faces_q_points, std::vector<Tensor<2, dim>>(n_dofs)));
393 
394  this->face_phi_p = std::vector<std::vector<std::vector<double>>>(
395  n_faces,
396  std::vector<std::vector<double>>(n_faces_q_points,
397  std::vector<double>(n_dofs)));
398 
399  this->face_grad_phi_p =
400  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
401  n_faces,
402  std::vector<std::vector<Tensor<1, dim>>>(
403  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
404  for (const auto face : cell->face_indices())
405  {
406  is_boundary_face[face] = cell->face(face)->at_boundary();
407  if (is_boundary_face[face])
408  {
409  fe_face_values.reinit(cell, face);
410  n_dofs = fe_face_values.get_fe().n_dofs_per_cell();
411  boundary_face_id[face] = cell->face(face)->boundary_id();
412  // Shape functions
413  // First vector is face number, second quadrature point, third
414  // DOF
415 
416  // Gather velocity (values, gradient and laplacian)
417  this->fe_face_values[velocities].get_function_values(
418  current_solution, this->face_velocity_values[face]);
419  this->fe_face_values[velocities].get_function_gradients(
420  current_solution, this->face_velocity_gradients[face]);
421  this->fe_face_values[velocities].get_function_laplacians(
422  current_solution, this->face_velocity_laplacians[face]);
423 
424  // Gather pressure (values, gradient)
425  this->fe_values[pressure].get_function_values(
426  current_solution, this->face_pressure_values[face]);
427  this->fe_values[pressure].get_function_gradients(
428  current_solution, this->face_pressure_gradients[face]);
429 
430  for (unsigned int q = 0; q < n_faces_q_points; ++q)
431  {
432  this->face_JxW[face][q] = this->fe_face_values.JxW(q);
433  this->face_normal[face][q] =
434  this->fe_face_values.normal_vector(q);
435  this->face_quadrature_points[face][q] =
436  this->fe_face_values.quadrature_point(q);
437  for (const unsigned int k : fe_face_values.dof_indices())
438  {
439  // Velocity
440  this->face_phi_u[face][q][k] =
441  this->fe_face_values[velocities].value(k, q);
442  this->face_div_phi_u[face][q][k] =
443  this->fe_face_values[velocities].divergence(k, q);
444  this->face_grad_phi_u[face][q][k] =
445  this->fe_face_values[velocities].gradient(k, q);
446  this->face_hess_phi_u[face][q][k] =
447  this->fe_face_values[velocities].hessian(k, q);
448  for (int d = 0; d < dim; ++d)
449  this->face_laplacian_phi_u[face][q][k][d] =
450  trace(this->face_hess_phi_u[face][q][k][d]);
451  // Pressure
452  this->face_phi_p[face][q][k] =
453  this->fe_face_values[pressure].value(k, q);
454  this->face_grad_phi_p[face][q][k] =
455  this->fe_face_values[pressure].gradient(k, q);
456  }
457  }
458  }
459  }
460  }

648  * This cell must be compatible with the void fraction FE and not the

652  *
653  * @param previous_solutions The solutions at the previous time steps for [epsilon]
654  *
655  */
656 
657  template <typename VectorType>
658  void
659  reinit_particle_fluid_interactions(
660  const typename DoFHandler<dim>::active_cell_iterator &cell,
661  const VectorType /*current_solution*/,
662  const VectorType previous_solution,
663  const VectorType void_fraction_solution,
664  const Particles::ParticleHandler<dim> &particle_handler,
665  DoFHandler<dim> &dof_handler,
666  DoFHandler<dim> &void_fraction_dof_handler)
667  {
668  const FiniteElement<dim> &fe = this->fe_values.get_fe();
669  const FiniteElement<dim> &fe_void_fraction =
670  this->fe_values_void_fraction->get_fe();
671 
672  bool interpolated_void_fraction = this->interpolated_void_fraction;
673  const unsigned int dofs_per_cell = fe.dofs_per_cell;
674  std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
675  std::vector<types::global_dof_index> void_fraction_dof_indices(
676  fe_void_fraction.dofs_per_cell);
677 
678  pic = particle_handler.particles_in_cell(this->fe_values.get_cell());
679 
680  average_particle_velocity = 0;
681  cell_volume = this->fe_values.get_cell()->measure();
682 
683  // Get the local dof indices for velocity and void fraction field
684  // interpolation at particle's position
685  const auto &dh_cell =
686  typename DoFHandler<dim>::cell_iterator(*this->fe_values.get_cell(),
687  &dof_handler);
688  dh_cell->get_dof_indices(fluid_dof_indices);
689 
690  const auto &void_fraction_dh_cell = typename DoFHandler<dim>::cell_iterator(
691  *this->fe_values_void_fraction->get_cell(), &void_fraction_dof_handler);
692  void_fraction_dh_cell->get_dof_indices(void_fraction_dof_indices);
693 
694  // Loop over particles in cell
695  double total_particle_volume = 0;
696  {
697  unsigned int particle_no = 0;
698  for (auto &particle : pic)
699  {
700  auto particle_properties = particle.get_properties();
701  // Set the particle_fluid_interactions properties and vectors to 0
702  for (int d = 0; d < dim; ++d)
703  {
704  particle_properties[DEM::PropertiesIndex::fem_force_x + d] = 0.;
705  particle_properties[DEM::PropertiesIndex::fem_torque_x + d] = 0.;
706  undisturbed_flow_force[d] = 0.;
707  }
708 
709  fluid_velocity_at_particle_location[particle_no] = 0;
710 
711  // Stock the values of particle velocity in a tensor
712  particle_velocity[particle_no][0] =
713  particle_properties[DEM::PropertiesIndex::v_x];
714  particle_velocity[particle_no][1] =
715  particle_properties[DEM::PropertiesIndex::v_y];
716  if (dim == 3)
717  particle_velocity[particle_no][2] =
718  particle_properties[DEM::PropertiesIndex::v_z];
719 
720  // Interpolate velocity and void fraction at particle position
721  // Reference location of the particle
722  auto reference_location = particle.get_reference_location();
723 
724  for (unsigned int j = 0; j < dofs_per_cell; ++j)
725  {
726  const auto comp_j = fe.system_to_component_index(j).first;
727  if (comp_j < dim)
728  {
729  auto &evaluation_point = previous_solution;
730  fluid_velocity_at_particle_location[particle_no][comp_j] +=
731  evaluation_point[fluid_dof_indices[j]] *
732  fe.shape_value(j, reference_location);
733  }
734  }
735 
736  cell_void_fraction[particle_no] = 0;
737  if (interpolated_void_fraction == true)
738  {
739  for (unsigned int j = 0; j < fe_void_fraction.dofs_per_cell; ++j)
740  {
741  cell_void_fraction[particle_no] +=
742  void_fraction_solution[void_fraction_dof_indices[j]] *
743  fe_void_fraction.shape_value(j, reference_location);
744  }
745  }
746  else
747  {
748  total_particle_volume +=
749  M_PI * pow(particle_properties[DEM::PropertiesIndex::dp], dim) /
750  (2 * dim);
751  }
752 
753  average_particle_velocity += particle_velocity[particle_no];
754  particle_no++;
755  }
756  number_of_particles = particle_no;
757  }
758 
759  if (interpolated_void_fraction == false)
760  {
761  double cell_void_fraction_bulk = 0;
762  cell_void_fraction_bulk =
763  (cell_volume - total_particle_volume) / cell_volume;
764 
765  for (unsigned int j = 0; j < number_of_particles; ++j)
766  cell_void_fraction[j] = cell_void_fraction_bulk;
767  }
768 
769  if (number_of_particles > 0)
770  average_particle_velocity =
771  average_particle_velocity / number_of_particles;
772 
773  // Relative velocity and particle Reynolds
774  unsigned int particle_no = 0;
775  average_fluid_particle_relative_velocity = 0;
776  double kinematic_viscosity =
777  properties_manager.get_kinematic_viscosity_scale();
778 
779  for (auto &particle : pic)
780  {
781  auto particle_properties = particle.get_properties();
782  fluid_particle_relative_velocity_at_particle_location[particle_no] =
783  fluid_velocity_at_particle_location[particle_no] -
784  particle_velocity[particle_no];
785  average_fluid_particle_relative_velocity +=
786  fluid_particle_relative_velocity_at_particle_location[particle_no];
787 
788  Re_particle[particle_no] =
789  1e-3 +
790  cell_void_fraction[particle_no] *
791  fluid_particle_relative_velocity_at_particle_location[particle_no]
792  .norm() *
793  particle_properties[DEM::PropertiesIndex::dp] /
794  (kinematic_viscosity + DBL_MIN);
795  particle_no++;
796  }
797  if (particle_no > 0)
798  {
799  average_fluid_particle_relative_velocity =
800  average_fluid_particle_relative_velocity / particle_no;
801  }
802 
803  // If there are no particles in the cell,there is no need to try and gather
804  // the rest of the information
805  if (number_of_particles == 0)
806  return;
807 
808  // If particles are in the cell, gather the rest of the
809  // information
810 
811  // Create local vector that will be use to spawn an in-situ quadrature to
812  // interpolate at the location of the particles
813  std::vector<Point<dim>> particle_reference_location(number_of_particles);
814  std::vector<double> particle_weights(number_of_particles, 1);
815 
816  // Resize arrays to be of the right size
817  fluid_velocity_laplacian_at_particle_location.resize(number_of_particles);
818  fluid_velocity_curls_at_particle_location_2d.resize(number_of_particles);
819  fluid_velocity_curls_at_particle_location_3d.resize(number_of_particles);
820  fluid_pressure_gradients_at_particle_location.resize(number_of_particles);
821 
822  // Loop over particles in cell and cache their reference location
823  {
824  unsigned int particle_no = 0;
825  for (auto &particle : pic)
826  {
827  // Store particle positions and weights
828  // Reference location of the particle
829  particle_reference_location[particle_no] =
830  particle.get_reference_location();
831  particle_no++;
832  }
833  }
834 
835  // Create a quadrature that is based on the particle reference location
836  Quadrature<dim> q_local(particle_reference_location, particle_weights);
837  FEValues<dim> fe_values_local_particles(
838  fe, q_local, update_gradients | update_values | update_hessians);
839 
840  // Evaluate the relevant information at the
841  // quadrature points to do the interpolation.
842  fe_values_local_particles.reinit(cell);
843 
844  fe_values_local_particles[velocities].get_function_laplacians(
845  previous_solution, fluid_velocity_laplacian_at_particle_location);
846 
847  if constexpr (dim == 2)
848  {
849  fe_values_local_particles[velocities].get_function_curls(
850  previous_solution, fluid_velocity_curls_at_particle_location_2d);
851  }
852  else if constexpr (dim == 3)
853  {
854  fe_values_local_particles[velocities].get_function_curls(
855  previous_solution, fluid_velocity_curls_at_particle_location_3d);
856  }
857 
858  fe_values_local_particles[pressure].get_function_gradients(
859  previous_solution, fluid_pressure_gradients_at_particle_location);
860  }

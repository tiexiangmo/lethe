125 // This function finds all the boundary cells and faces in the triangulation,
126 // for each cell the boundary faces are specified and the normal vector as well
127 // as a point on the boundary faces are obtained
128 template <int dim>
129 void
130 BoundaryCellsInformation<dim>::find_boundary_cells_information(
131  const parallel::distributed::Triangulation<dim> &triangulation,
132  const std::vector<unsigned int> &outlet_boundaries)
133 {
134  // Initializing output_vector and a search_vector (containing boundary_id and
135  // cell) to avoid replication of a boundary face. All the found boundary faces
136  // will be searched in this vector before being added to the output_vector. If
137  // they are not found in this search_vector they will be added to the
138  // output_vector as well as the search_vector
139  // std::map<int, boundary_cells_info_struct<dim>> output_vector;
140  std::vector<std::pair<int, typename Triangulation<dim>::active_cell_iterator>>
141  search_vector;
142 
143  // Initialize a simple quadrature for on the system. This will be used to
144  // obtain a single sample point on the boundary faces
145  const FE_Q<dim> fe(1);
146  QGauss<dim - 1> face_quadrature_formula(1);
147  FEFaceValues<dim> fe_face_values(fe,
148  face_quadrature_formula,
149  update_values | update_quadrature_points |
150  update_normal_vectors);
151 
152  // Iterating over the active cells in the triangulation
153  for (const auto &cell : triangulation.active_cell_iterators())
154  {
155  // Iterating over the faces of each cell
156  for (const auto &face : cell->face_iterators())
157  {
158  // We search to see if the boundary is defined as an outlet, periodic
159  // or not. If it is not defined as one of those, we proceed.
160  int face_id = cell->face_iterator_to_index(face);
161  bool is_outlet =
162  std::find(outlet_boundaries.begin(),
163  outlet_boundaries.end(),
164  face->boundary_id()) != outlet_boundaries.end();
165  bool is_periodic = cell->has_periodic_neighbor(face_id);
166 
167  if (!is_outlet && !is_periodic)
168  {
169  // Check if the face is located at boundary
170  if (face->at_boundary())
171  {
172  fe_face_values.reinit(cell, face_id);
173 
174  // Finding the normal vector of the boundary face
175  Tensor<1, dim> normal_vector =
176  -fe_face_values.normal_vector(0);
177 
178  // Finding a point on the boundary face
179  Point<dim> quad_point = fe_face_values.quadrature_point(0);
180 
181  // Get global index of the face
182  unsigned int global_face_id = cell->face_index(face_id);
183 
184  // Storing these information into the
185  // boundary_cells_info_struct
186  boundary_cells_info_struct<dim> boundary_information;
187  boundary_information.cell = cell;
188  boundary_information.boundary_id = face->boundary_id();
189  boundary_information.global_face_id = global_face_id;
190  boundary_information.normal_vector = normal_vector;
191  boundary_information.point_on_face = quad_point;
192 
193  // Searching to see if boundary with this information was
194  // already found or not. If this is a new
195  // boundary face, it will be added to the
196  // output_vector as well as the search_vector to avoid
197  // repetition
198  auto information_search_element =
199  std::make_pair(face_id, cell);
200  auto search_iterator = std::find(search_vector.begin(),
201  search_vector.end(),
202  information_search_element);
203  if (search_iterator == search_vector.end())
204  {
205  if (cell->is_locally_owned())
206  {
207  boundary_cells_information.insert(
208  {global_face_id, boundary_information});
209  boundary_cells_with_faces.push_back(cell);
210  search_vector.push_back(information_search_element);
211  }
212  global_boundary_cells_information.insert(
213  {global_face_id, boundary_information});
214  }
215  }
216  }
217  }
218  }
219 }

223 // stored point of the boundary faces. It is used when the grid is moving.
224 // Updated points and normal vectors are then used to update the particle-wall
225 // contact list.
226 template <int dim>
227 void
228 BoundaryCellsInformation<dim>::update_boundary_info_after_grid_motion(
229  typename DEM::dem_data_structures<dim>::boundary_points_and_normal_vectors
230  &updated_boundary_points_and_normal_vectors)
231 {
232  // Initialize a simple quadrature for on the system. This will be used to
233  // obtain a single sample point on the boundary faces
234  const FE_Q<dim> fe(1);
235  QGauss<dim - 1> face_quadrature_formula(1);
236  FEFaceValues<dim> fe_face_values(fe,
237  face_quadrature_formula,

239  update_normal_vectors);

241  {
242  if (cell->is_locally_owned())

244  // Iterating over the faces of each cell

246  {
247  // Check to see if the face is located at boundary
248  if (face->at_boundary())

272 
273  updated_boundary_points_and_normal_vectors[cell->face_index(

275  }
276  }

281 // This function returns all the cells at which particle-line and particle-point
282 // contact search should be performed
283 template <int dim>
284 void
285 BoundaryCellsInformation<dim>::find_particle_point_and_line_contact_cells(
286  const parallel::distributed::Triangulation<dim> &triangulation)
287 {
288  // Here, we first loop through all the non-boundary faces of the local cells
289  // and find (add them to all cells_with_boundary_lines container) all boundary
290  // lines of these faces. Then the boundary lines which are the borders of two
291  // boundary faces should be removed from this container. Finally the remained
292  // elements of this container are added to boundary_cells_with_lines
293 
294  // Boundary lines only exist in three-dimensional cases
295  if (dim == 3)
296  {
297  std::unordered_map<std::string,
298  std::unordered_map<types::global_dof_index,
299  std::pair<Point<dim>, Point<dim>>>>
300  all_cells_with_boundary_lines;
301 
302  // Iterating over the active cells in the triangulation
303  for (const auto &cell : triangulation.active_cell_iterators())
304  {
305  if (cell->is_locally_owned())
306  {
307  // Looping through all the faces of the cell
308  for (const auto &face : cell->face_iterators())
309  {
310  // Check to see if the face is not located at boundary
311  if (!face->at_boundary())
312  {
313  // Adding all the boundary lines of these faces into the
314  // all_cells_with_boundary_lines container
315  for (unsigned int l = 0; l < face->n_lines(); ++l)
316  {
317  if (face->line(l)->at_boundary())
318  {
319  all_cells_with_boundary_lines[cell->id()
320  .to_string()]
321  .insert(
322  {face->line_index(l),
323  std::make_pair(face->line(l)->vertex(0),
324  face->line(l)->vertex(1))});
325  }
326  }
327  }
328  }
329  }
330  }
331 
332  // Now iterating again over the cells, faces and lines and if the boundary
333  // line of the boundary face exists in the container, it is removed from
334  // the container
335  if (!all_cells_with_boundary_lines.empty())
336  {
337  for (const auto &cell : triangulation.active_cell_iterators())
338  {
339  if (cell->is_locally_owned())
340  {
341  for (const auto &face : cell->face_iterators())
342  {
343  // Check to see if the face is located at boundary
344  if (face->at_boundary())
345  {
346  for (unsigned int l = 0; l < face->n_lines(); ++l)
347  {
348  all_cells_with_boundary_lines[cell->id()
349  .to_string()]
350  .erase(face->line_index(l));
351  }
352  }
353  }
354  }
355  }
356  }
357 
358  // Finally, adding the remained elements of the
359  // all_cells_with_boundary_lines container to boundary_cells_with_lines
360  if (!all_cells_with_boundary_lines.empty())
361  {
362  for (const auto &cell : triangulation.active_cell_iterators())
363  {
364  if (cell->is_locally_owned())
365  {
366  std::string cell_id_string = cell->id().to_string();
367  std::unordered_map<types::global_dof_index,
368  std::pair<Point<dim>, Point<dim>>>
369  &cell_boundary_lines =
370  all_cells_with_boundary_lines[cell_id_string];
371 
372  if (!cell_boundary_lines.empty())
373  {
374  for (auto &map_iterator : cell_boundary_lines)
375  {
376  boundary_cells_with_lines.insert(
377  {cell_id_string,
378  std::make_tuple(cell,
379  map_iterator.second.first,
380  map_iterator.second.second)});
381 
382  // Add the cell to local_cells_with_boundary_lines to
383  // be used in
384  // add_cells_with_boundary_lines_to_boundary_cells
385  local_cells_with_boundary_lines.push_back(cell);
386  }
387  }
388  }
389  }
390  }
391  }
392 
393  // Finding boundary points. We need this container because at_boundary()
394  // function is not defined for vertex id (unsigned int) nor vertex position
395  // (Point<dim>). For line and face objects, at_boundary() function is
396  // available to help
397 
398  // First getting a set of all boundary vertices
399  std::set<unsigned int> boundary_vertices;
400  for (const auto &face : triangulation.active_face_iterators())
401  {
402  if (face->at_boundary())
403  {
404  for (unsigned int v = 0; v < face->n_vertices(); ++v)
405  {
406  boundary_vertices.insert(face->vertex_index(v));
407  }
408  }
409  }
410 
411  // If a cell does not have any boundary line, but has boundary vertex, it is
412  // added to boundary_cells_with_points container Iterating over the active
413  // cells in the triangulation
414  for (const auto &cell : triangulation.active_cell_iterators())
415  {
416  if (cell->is_locally_owned() && !cell->has_boundary_lines())
417  {
418  for (const auto &face : cell->face_iterators())
419  {
420  if (!face->at_boundary())
421  {
422  for (unsigned int v = 0; v < face->n_vertices(); ++v)
423  {
424  if (boundary_vertices.count(face->vertex_index(v)) > 0)
425  {
426  boundary_cells_with_points.insert(
427  {cell->id().to_string(),
428  std::make_pair(cell, face->vertex(v))});
429  }
430  }
431  }
432  }
433  }
434  }

465 template <int dim>
466 void
467 BoundaryCellsInformation<dim>::add_cells_with_boundary_lines_to_boundary_cells(
468  const parallel::distributed::Triangulation<dim> &triangulation,
469  const std::vector<unsigned int> &outlet_boundaries,
470  const bool &check_diamond_cells,
471  const ConditionalOStream &pcout)
472 {
473  std::vector<typename Triangulation<dim>::active_cell_iterator>
474  boundary_neighbor_cells;
475 
476  // First we have to store all the global (including local and non-local) cells
477  // with boundary faces in a container. Note that the boundary_cells_with_faces
478  // container only stores the local boundary cells
479  find_global_boundary_cells_with_faces(triangulation, outlet_boundaries);
480 
481  if (!local_cells_with_boundary_lines.empty())
482  {
483  for (auto &cell_with_boundary_line : local_cells_with_boundary_lines)
484  {
485  // Each cell with boundary line must have two neighbor cells with
486  // boundary faces. We have to find these two neighbors first.
487  // Finding neighbors of the cell
488  std::vector<typename Triangulation<dim>::active_cell_iterator>
489  active_neighbors;
490  GridTools::get_active_neighbors<Triangulation<dim>>(
491  cell_with_boundary_line, active_neighbors);
492 
493  // Looping through the neighbors and storing the ones with boundary
494  // faces
495  boundary_neighbor_cells.clear();
496  for (auto &neighbor : active_neighbors)
497  {
498  if (std::count(global_boundary_cells_with_faces.begin(),
499  global_boundary_cells_with_faces.end(),
500  neighbor))
501  {
502  boundary_neighbor_cells.push_back(neighbor);
503  }
504  }
505 
506  // Find the information of every combination of two cells in
507  // global_boundary_cells_normals with counters
508  for (unsigned int counter_one = 0;
509  counter_one < boundary_neighbor_cells.size();
510  ++counter_one)
511  {
512  for (unsigned int counter_two = 0;
513  counter_two < boundary_neighbor_cells.size();
514  ++counter_two)
515  {
516  if (counter_one > counter_two)
517  {
518  // Get the pair of cells
519  typename Triangulation<dim>::active_cell_iterator
520  cell_one = boundary_neighbor_cells[counter_one];
521  typename Triangulation<dim>::active_cell_iterator
522  cell_two = boundary_neighbor_cells[counter_two];
523 
524  // Loop on all faces of the 2 cells
525  for (const auto &face_one : cell_one->face_iterators())
526  {
527  for (const auto &face_two :
528  cell_two->face_iterators())
529  {
530  // Check if faces are located at boundaries
531  if (face_one->at_boundary() &&
532  face_two->at_boundary())
533  {
534  unsigned int face_id_one, face_id_two;
535  face_id_one =
536  cell_one->face_iterator_to_index(face_one);
537  face_id_two =
538  cell_two->face_iterator_to_index(face_two);
539 
540  // Check if faces are on a wall boundary
541  // (not outlet nor periodic)
542  bool face_one_is_wall, face_two_is_wall;
543  face_one_is_wall =
544  (std::find(outlet_boundaries.begin(),
545  outlet_boundaries.end(),
546  face_one->boundary_id()) ==
547  outlet_boundaries.end() &&
548  !cell_one->has_periodic_neighbor(
549  face_id_one));
550  face_two_is_wall =
551  (std::find(outlet_boundaries.begin(),
552  outlet_boundaries.end(),
553  face_two->boundary_id()) ==
554  outlet_boundaries.end() &&
555  !cell_two->has_periodic_neighbor(
556  face_id_two));
557 
558  if (face_one_is_wall && face_two_is_wall)
559  {
560  // Get the normal vector of the first cell
561  Tensor<1, dim> normal_vector_one,
562  normal_vector_two;
563  normal_vector_one =
564  global_boundary_cells_information
565  .at(cell_one->face_index(face_id_one))
566  .normal_vector;
567 
568  normal_vector_two =
569  global_boundary_cells_information
570  .at(cell_two->face_index(face_id_two))
571  .normal_vector;
572 
573  // Check to see if the dot product of the
574  // two normal vectors is larger than a
575  // specified threshold (cos(45) = 0.707)
576  if (normal_vector_one *
577  normal_vector_two >
578  0.707)
579  {
580  if (display_diamond_cells_warning)
581  {
582  pcout
583  << std::endl
584  << "Warning: There are diamond-shaped cells in the input triangulation. It is strongly recommended to use a different triangulation without such cells. It should be mentioned that these cells are not detected if you have grid motion"
585  << std::endl;
586  display_diamond_cells_warning =
587  false;
588  }
589 
590  if (check_diamond_cells &&
591  cell_with_boundary_line
592  ->is_locally_owned())
593  {
594  // If this condition is true, add
595  // this cell to
596  // boundary_cells_information.
597  // Since the key of
598  // boundary_cells_information is
599  // the boundary face id, and this
600  // cell does not have a boundary
601  // face, we add it to the
602  // boundary_cells_information with
603  // a negative key
604  int imaginary_face_id =
605  -1 * cell_two->face_index(
606  face_id_two);
607 
608  // Update cell to the cell with
609  // boundary line
610  boundary_cells_info_struct<dim>
611  imaginary_information =
612  global_boundary_cells_information
613  .at(cell_two->face_index(
614  face_id_two));
615  imaginary_information.cell =
616  cell_with_boundary_line;
617  imaginary_information
618  .global_face_id =
619  imaginary_face_id;
620 
621  boundary_cells_information.insert(
622  {imaginary_face_id,
623  imaginary_information});
624  }
625  }
626  }
627  }
628  }
629  }

637 // This function finds the triangulation cells adjacent to the floating walls
638 template <int dim>
639 void
640 BoundaryCellsInformation<dim>::find_boundary_cells_for_floating_walls(
641  const parallel::distributed::Triangulation<dim> &triangulation,
642  const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
643  const double maximum_cell_diameter)
644 {
645  // Reading floating wall properties
646  std::vector<Point<dim>> point_on_wall =
647  floating_wall_properties.points_on_walls;
648  std::vector<Tensor<1, dim>> wall_normal_vector =
649  floating_wall_properties.floating_walls_normal_vectors;
650  unsigned int floating_wall_number =
651  floating_wall_properties.floating_walls_number;
652 
653  // Looping through floating walls
654  for (unsigned int floating_wall_id = 0;
655  floating_wall_id < floating_wall_number;
656  ++floating_wall_id)
657  {
658  // Looping through cells
659  for (const auto &cell : triangulation.active_cell_iterators())
660  {
661  // If the cell is owned by the processor
662  if (cell->is_locally_owned())
663  {
664  // Looping over vertices of each cell
665  for (unsigned int vertex = 0; vertex < cell->n_vertices();
666  ++vertex)
667  {
668  // Finding vertex-floating wall distance
669  Tensor<1, dim> connecting_vector =
670  cell->vertex(vertex) - point_on_wall[floating_wall_id];
671  double vertex_wall_distance =
672  connecting_vector * wall_normal_vector[floating_wall_id];
673 
674  // If the distance is less than the largest cell size,
675  // it should be added to the
676  // boundary_cells_for_floating_walls
677  if (abs(vertex_wall_distance) < maximum_cell_diameter)
678  {
679  boundary_cells_for_floating_walls[floating_wall_id]
680  .insert(cell);
681  }
682  }

688 template <int dim>
689 void
690 BoundaryCellsInformation<dim>::add_boundary_neighbors_of_boundary_cells(
691  const parallel::distributed::Triangulation<dim> &triangulation,
692  const std::vector<unsigned int> &outlet_boundaries,
693  std::map<int, boundary_cells_info_struct<dim>> &boundary_cells_information,
694  const std::map<int, boundary_cells_info_struct<dim>>
695  &global_boundary_cells_information)
696 {
697  // Create a vector of a set of adjacent cells of all the vertices
698  std::vector<std::set<typename Triangulation<dim>::active_cell_iterator>>
699  v_to_c = GridTools::vertex_to_cell_map(triangulation);
700 
701  for (auto &&boundary_cells_info :
702  boundary_cells_information | boost::adaptors::map_values)
703  {
704  // The boundary cell is local by definition
705  // This loop will check for all the cell neighbors located at vertices of
706  // a boundary cell and will add the proper information in the boundary
707  // cell list if faces are not at outlet nor periodic boundary.
708 
709  // Iterate over the vertices of each boundary cell
710  for (auto vertex_id : GeometryInfo<dim>::vertex_indices())
711  {
712  // Iterate over the neighbors of each boundary cell having vertex_id
713  for (const auto &neighbor :
714  v_to_c[boundary_cells_info.cell->vertex_index(vertex_id)])
715  {
716  // Iterate over the faces of the neighbor cell
717  for (const auto &face : neighbor->face_iterators())
718  {
719  // Check if the face of the neighbor is located at boundary
720  if (face->at_boundary())
721  {
722  // Check if face is on a wall boundary (not outlet nor
723  // periodic)
724  unsigned int face_id =
725  neighbor->face_iterator_to_index(face);
726  bool face_is_wall =
727  (std::find(outlet_boundaries.begin(),
728  outlet_boundaries.end(),
729  face->boundary_id()) ==
730  outlet_boundaries.end() &&
731  !neighbor->has_periodic_neighbor(face_id));
732 
733  if (face_is_wall)
734  {
735  // Get the boundary neighbor cell info
736  boundary_cells_info_struct<dim>
737  boundary_neighbor_information =
738  global_boundary_cells_information.at(
739  neighbor->face_index(face_id));
740 
741  // Add the main boundary cell with the information
742  // (point and normal vector) of the neighbor boundary
743  // cell to the boundary_cells_information container.
744  // Note that since we may already have an element with
745  // the key of face_id (key of the
746  // boundary_cells_information map) in the
747  // boundary_cells_information, we add the new element
748  // with a unique key to create a unique id in the map.
749  // This unique key is generated using Cantor pairing
750  // function: unique_key = 0.5 * (a + b) * (a + b + 1)
751  // + b where a and b are global boundary face ids of
752  // the main boundary cell and the neighbor boundary
753  // cell.
754  int imaginary_face_id =
755  -0.5 *
756  (boundary_cells_info.global_face_id +
757  boundary_neighbor_information.global_face_id) *
758  (boundary_cells_info.global_face_id +
759  boundary_neighbor_information.global_face_id +
760  1) +
761  boundary_neighbor_information.global_face_id;
762 
763  // Create a cell info object which is a copy of all
764  // the boundary neighbor information applied to the
765  // boundary cell & store in map with imaginary key.
766  boundary_cells_info_struct<dim> boundary_information =
767  boundary_neighbor_information;
768  boundary_information.cell = boundary_cells_info.cell;
769 
770  boundary_cells_information.insert(
771  {imaginary_face_id, boundary_information});
772  }
773  }
774  }
775  }
776  }
777  }

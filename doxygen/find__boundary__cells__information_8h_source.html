42 /**
43  * Finds all the boundary cells and faces in the triangulation, for each cell
44  * the boundary face is specified and the normal vector as well as a point on
45  * the boundary face are obtained
46  *
47  * @note
48  *
49  * @author Shahab Golshan, Polytechnique Montreal 2019-
50  */

58  /**
59  * @brief The build function builds all the boundary cell information
60  * structure that finds which cell has a boundary face, boundary line or a
61  * boundary point. Additionally, it identifies which cells are located in the
62  * vicinity (controlled by the max cell diameter of floating walls
63  * @param triangulation Triangulation to access the information of the cells
64  * @param floating_wall_properties Properties of floating walls specified in
65  * the parameter handler file
66  * @param outlet_boundaries A vector which contains the outlet boundary IDs
67  * @param check_diamond_cells If true, the diamond shaped cells are found
68  * and added to the particle-wall contact search cells
69  * @param expand_particle_wall_contact_search If true, expands the
70  * particle-wall contact search by adding the contact search between
71  * particles in the boundary cells with the boundary neighbors of these
72  * boundary cells
73  *
74  * @param pcout
75  */
76  void

80  const std::vector<unsigned int> &outlet_boundaries,
81  const bool &check_diamond_cells,
82  const bool &expand_particle_wall_contact_search,
83  const ConditionalOStream &pcout);

181  /**
182  * Adds the cells with boundary lines to the boundary cells information
183  * (boundary_cells_information), First , it loops through the
184  * boundary_cells_with_lines vector, then it loops through the neighbors of
185  * boundary cells with lines, if a neighbor is a member of
186  * boundary_cells_with_faces, then it adds the neighbor cell to the
187  * boundary_cells_information with the normal vector and point of the neighbor
188  * cell.
189  *
190  * @param triangulation Triangulation to access the information of the cells
191  * @param outlet_boundaries A vector which contains the outlet boundary IDs
192  * @param check_diamond_cells If true, the diamond shaped cells are found and added to the particle-wall contact search cells
193  * @param pcout
194  */
195  void
196  add_cells_with_boundary_lines_to_boundary_cells(

199  const bool &check_diamond_cells,
200  const ConditionalOStream &pcout);

219  /**
220  * Carries out adding new elements with the boundary cells (cells with
221  * boundary faces) and information (point and normal vector) of their neighbor
222  * boundary cells to the boundary_cells_information. By adding these
223  * elements, the broad and fine searches will search for potential contacts
224  * of the particles in each boundary cell with the boundary neighbors of these
225  * cells. This function is used to solve the problem of late contact detection
226  * in triangulations with curved boundaries.
227  *
228  * @param boundary_cells_information A container that contains the information
229  * of all the boundary cells with boundary faces
230  * @param outlet_boundaries A vector which contains the outlet boundary IDs
231  * @param global_boundary_cells_information A vector that contains the geometrical
232  * information of all (global) boundary cells
233  */
234  void
235  add_boundary_neighbors_of_boundary_cells(
236  const parallel::distributed::Triangulation<dim> &triangulation,
237  const std::vector<unsigned int> &outlet_boundaries,
238  std::map<int, boundary_cells_info_struct<dim>> &boundary_cells_information,
239  const std::map<int, boundary_cells_info_struct<dim>>
240  &global_boundary_cells_information);

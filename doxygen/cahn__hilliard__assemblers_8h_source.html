27 /**
28  * @brief A pure virtual class that serves as an interface for all

30  *
31  * @tparam dim An integer that denotes the number of spatial dimensions

36 class CahnHilliardAssemblerBase
37 {
38 public:
39  CahnHilliardAssemblerBase(
40  std::shared_ptr<SimulationControl> p_simulation_control)

42  {}

44  /**
45  * @brief assemble_matrix Interface for the call to matrix assembly
46  * @param scratch_data Scratch data containing the information for Cahn-Hilliard
47  * equations.
48  * It is important to note that the scratch data has to have been re-inited
49  * before calling for matrix assembly.
50  * @param copy_data Destination where the local_rhs and local_matrix are copied to.
51  */
52 
53  virtual void
54  assemble_matrix(CahnHilliardScratchData<dim> &scratch_data,
55  StabilizedMethodsCopyData &copy_data) = 0;
56 
57 
58  /**
59  * @brief assemble_rhs Interface for the call to rhs assembly

61  * equations.
62  * It is important to note that the scratch data has to have been re-inited
63  * before calling for rhs assembly.
64  * @param copy_data Destination where the local_rhs and local_matrix are copied to.
65  */
66 
67  virtual void
68  assemble_rhs(CahnHilliardScratchData<dim> &scratch_data,
69  StabilizedMethodsCopyData &copy_data) = 0;
70 
71 protected:
72  std::shared_ptr<SimulationControl> simulation_control;
73 };

81  *

83  *
84  * @ingroup assemblers

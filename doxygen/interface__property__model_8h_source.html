25 /**
26  * @brief Abstract class that defines the interface for a interface property model.
27  * Interface property model provides an abstract interface to calculate the
28  * value of an interface property or a vector of interface property value for
29  * given field value. By default, the interface does not require that all (or
30  * any) fields be specified. This is why a map is used to pass the dependent
31  * variable. To allow for the calculation of the appropriate jacobian matrix
32  * (when that is necessary) the interface also provides a jacobian function,
33  * which must provide the derivative with respect to the field specified as an
34  * argument.
35  */

103  /**
104  * @brief numerical_jacobian Calculates the jacobian through a forward finite difference (Euler) approximation.
105  * This approach, although not preferable, is meant as a fall-back when
106  * calculating the jacobian manually is too difficult.
107  * @param fields_values The values of the various fields
108  * @param id Field id of the field with respect to which the jacobian should be calculated
109  * @return
110  */
111  inline double
112  numerical_jacobian(const std::map<field, double> &field_values,
113  const field id)
114  {
115  double f_x = this->value(field_values);
116  auto x_dx = field_values;
117  double dx = std::max(1e-6 * x_dx[id], 1e-8);
118  x_dx[id] = x_dx[id] + dx;
119  double f_xdx = this->value(x_dx);
120  return (f_xdx - f_x) / dx;
121  }
122 
123  /**
124  * @brief vector_numerical_jacobian Calculates the vector of jacobian through forward finite difference (Euler) approximation.
125  * This approach, although not preferable, is meant as a fall-back when
126  * calculating the jacobian manually is too difficult.
127  * @param field_vectors
128  * @param field_id
129  * @param jacobian_vector
130  * @return
131  */
132  inline void
133  vector_numerical_jacobian(
134  const std::map<field, std::vector<double>> &field_vectors,
135  const field id,
136  std::vector<double> &jacobian_vector)
137  {
138  const unsigned int n_pts = jacobian_vector.size();
139 
140  // Evaluate the properties using the values of the field vector
141  std::vector<double> f_x(n_pts);
142  vector_value(field_vectors, f_x);
143 
144  // Make a copy of the field vector for the field we will perturbate
145  std::map<field, std::vector<double>> perturbed_field_vectors =
146  field_vectors;
147  std::vector<double> &x = perturbed_field_vectors.at(id);
148  std::vector<double> dx(n_pts);
149 
150  // Perturb the field by dx
151  for (unsigned int i = 0; i < n_pts; ++i)
152  {
153  dx[i] = std::max(1e-6 * x[i], 1e-8);
154  x[i] += dx[i];
155  }
156 
157  // Evaluate the properties using the perturbed value of the field vector
158  std::vector<double> f_xdx(n_pts);
159  vector_value(perturbed_field_vectors, f_xdx);
160 
161  // Fill jacobian
162  for (unsigned int i = 0; i < n_pts; ++i)
163  jacobian_vector[i] = (f_xdx[i] - f_x[i]) / dx[i];
164  }
165 
166 protected:
167  // Map to indicate on which variables the model depends on
168  std::map<field, bool> model_depends_on;
169 };

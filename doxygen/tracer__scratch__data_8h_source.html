42 /**
43  * @brief Class that stores the information required by the assembly procedure
44  * for a Tracer advection-diffusion equation. Consequently, this class
45  *calculates the tracer (values, gradients, laplacians) and the shape function
46  * (values, gradients, laplacians) at all the gauss points for all degrees
47  * of freedom and stores it into arrays. Additionnaly, the use can request
48  * that this class gathers additional fields for physics which are coupled
49  * to the Tracer equation, such as the velocity which is required. This class
50  * serves as a seperation between the evaluation at the gauss point of the
51  * variables of interest and their use in the assembly, which is carried out
52  * by the assembler functions. For more information on this design, the reader
53  * can consult deal.II step-9
54  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
55  * example, the scratch is a struct instead of a templated class because of the
56  * simplicity of step-9.
57  *
58  * @tparam dim An integer that denotes the dimension of the space in which
59  * the flow is solved
60  * @ingroup solvers
61  **/
62

73  * @param fe The FESystem used to solve the Navier-Stokes equations
74  *

133  /** @brief Reinitialize the content of the scratch
134  *
135  * Using the FeValues and the content ofthe solutions and previous solutions,
136  * fills all of the class member of the scratch
137  *
138  * @param cell The cell over which the assembly is being carried.
139  * This cell must be compatible with the fe which is used to fill the FeValues
140  *
141  * @param current_solution The present value of the solution for [u,p]
142  *
143  * @param previous_solutions The solutions at the previous time steps
144  *
145  * @param source_function The function describing the tracer source term
146  *
147  */
148 
149  template <typename VectorType>
150  void
151  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
152  const VectorType &current_solution,
153  const std::vector<VectorType> &previous_solutions,
154  Function<dim> *source_function)
155  {
156  this->fe_values_tracer.reinit(cell);
157 
158  quadrature_points = this->fe_values_tracer.get_quadrature_points();
159  auto &fe_tracer = this->fe_values_tracer.get_fe();
160 
161  source_function->value_list(quadrature_points, source);
162 
163  if (dim == 2)
164  this->cell_size =
165  std::sqrt(4. * cell->measure() / M_PI) / fe_tracer.degree;
166  else if (dim == 3)
167  this->cell_size =
168  pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_tracer.degree;
169 
170  // Gather tracer (values, gradient and laplacian)
171  this->fe_values_tracer.get_function_values(current_solution,
172  this->tracer_values);
173  this->fe_values_tracer.get_function_gradients(current_solution,
174  this->tracer_gradients);
175  this->fe_values_tracer.get_function_laplacians(current_solution,
176  this->tracer_laplacians);
177 
178  // Gather previous tracer values
179  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
180  {
181  this->fe_values_tracer.get_function_values(previous_solutions[p],
182  previous_tracer_values[p]);
183  }
184 
185  for (unsigned int q = 0; q < n_q_points; ++q)
186  {
187  this->JxW[q] = this->fe_values_tracer.JxW(q);
188 
189  for (unsigned int k = 0; k < n_dofs; ++k)
190  {
191  // Shape function
192  this->phi[q][k] = this->fe_values_tracer.shape_value(k, q);
193  this->grad_phi[q][k] = this->fe_values_tracer.shape_grad(k, q);
194  this->hess_phi[q][k] = this->fe_values_tracer.shape_hessian(k, q);
195  this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
196  }
197  }
198  }

200  /** @brief Reinitialize the velocity, calculated by the fluid dynamics while also taking into account ALE
201  *
202  * @tparam VectorType The Vector type used for the solvers
203  *
204  * @param cell The cell for which the velocity is reinitialized
205  * This cell must be compatible with the Fluid Dynamics FE
206  *
207  * @param current_solution The present value of the solution for [u,p]
208  *
209  * @param ale The ALE parameters which include the ALE function
210  *
211  */
212 
213  template <typename VectorType>
214  void
215  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
216  const VectorType &current_solution,
217  const Parameters::ALE<dim> &ale)
218  {
219  this->fe_values_fd.reinit(cell);
220 
221  this->fe_values_fd[velocities].get_function_values(current_solution,
222  velocity_values);
223 
224  if (!ale.enabled())
225  return;
226 
227  // ALE enabled, so extract the ALE velocity and subtract it from the
228  // velocity obtained from the fluid dynamics
229  Tensor<1, dim> velocity_ale;
230  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
231  ale.velocity;
232  Vector<double> velocity_ale_vector(dim);
233 
234  for (unsigned int q = 0; q < n_q_points; ++q)
235  {
236  velocity_ale_function->vector_value(quadrature_points[q],
237  velocity_ale_vector);
238  for (unsigned int d = 0; d < dim; ++d)
239  velocity_ale[d] = velocity_ale_vector[d];
240 
241  velocity_values[q] -= velocity_ale;
242  }
243  }

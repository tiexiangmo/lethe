25 /**
26  * @brief The SimulationControl class is responsible for the control of steady-state and transient
27  * simulations carried out with Lethe. This base class is a pure virtual class
28  * and cannot be instantiated. However, it stores the core variables which are
29  * necessary for its serialization (write and read)
30  **/
31 
32 class SimulationControl
33 {
34 protected:
35  // Time stepping method used for the simulation
36  Parameters::SimulationControl::TimeSteppingMethod method;
37 
38  // Current method used for the present assembly
39  // This is used to differentiate the substeps of SDIRK methods or
40  // to start-up BDF simulations
41  Parameters::SimulationControl::TimeSteppingMethod assembly_method;
42 
43  // Time of the current iteration being solved for
44  double current_time;
45 
46  // Time of the previous iteration
47  double previous_time;
48 
49  // Time step linking the previous iteration and the current time
50  double time_step;
51 
52  // Initial time step given in the parameter
53  double initial_time_step;
54 
55  // Simulation end time
56  double end_time;
57 
58  // Time step vector. This vector accumulates the time steps of the previous
59  // iterations. This is required for multiple steps methods such as the bdfs.
60  std::vector<double> time_step_vector;
61 
62  // Iteration. Iterations start at 0, but the first actual iteration is
63  // iteration 1.
64  unsigned int iteration_number;
65 
66  // Number of mesh adaptation iteration
67  unsigned int number_mesh_adapt;
68 
69  // Courant-Friedrich-Levy (CFL) condition
70  // Since the simulation control is unaware of the information propagation
71  // mechanism (for instance the velocity), the current CFL must be set by the
72  // solver itself
73  double CFL;
74 
75  // Maximal CFL condition. This is used to control adaptative time stepping. In
76  // the case of constant time stepping or steady-state simulations, this
77  // parameter remains unused.
78  double max_CFL;
79 
80  // Current value of the norm of the rhs
81  double residual;
82 
83  // Residual at which the simulation stops when adjoint time-stepping is used
84  double stop_tolerance;
85 
86  // Number of time steps stored
87  // BDF methods require a number of previous time steps. This number is known a
88  // priori and depends on the method used. We do not keep all the time steps to
89  // prevent the accumulation within a large vector.
90  static const unsigned int n_previous_time_steps = 4;
91 
92  // Output iteration frequency
93  // Controls the output of the simulation results when the output is controlled
94  // by the iteration number.
95  unsigned int output_frequency;
96 
97  // Output time frequency
98  // Controls the output of the simulation results when the output is controlled
99  // by the time
100  double output_time_frequency;
101 
102  // Log iteration frequency
103  // Controls the frequency at which status of the simulation is written to
104  // the terminal
105  unsigned int log_frequency;
106 
107  // Log precision
108  // Controls the number of significant digits displayed on the standard outputs
109  unsigned int log_precision;
110 
111  // Number of mesh subdivision to be used when outputting the results
112  unsigned int subdivision;
113 
114  // Number of parallel file to generate
115  unsigned int group_files;
116 
117  // Output name
118  std::string output_name;
119 
120  // Output path
121  std::string output_path;
122 
123  // Output boundaries
124  // Control if the boundaries of the domain are outputted when writing results
125  //
126  bool output_boundaries;
127 
128  // Indicator to tell if this is the first assembly of a step
129  bool first_assembly;
130 
131  // The method use to start high order bdf scheme
132  Parameters::SimulationControl::BDFStartupMethods bdf_start_method;
133 
134  // The time scaling used to do small time-steps at the startup of the

136  double startup_timestep_scaling;

152  /**

417  std::vector<double>
418  get_simulation_times() const
419  {
420  // Create a vector of the previous times
421  std::vector<double> times(n_previous_time_steps + 1);
422  for (unsigned int i = 0; i < n_previous_time_steps + 1; ++i)
423  {
424  times[i] = current_time;
425  for (unsigned int j = 0; j < i; ++j)
426  times[i] -= time_step_vector[j];
427  }
428  return times;
429  }

440 class SimulationControlTransient : public SimulationControl
441 {
442 protected:
443  // Enable adaptative time stepping
444  bool adapt;
445 
446  // Time step scaling for adaptative time stepping
447  double adaptative_time_step_scaling;
448 
449  /**
450  * @brief Calculates the next value of the time step. If adaptation
451  * is enabled, the time step is calculated in order to ensure
452  * that the CFL condition is bound by the maximal CFL value.
453  * The new time step is equal to adaptative_time_step_scaling * the previous
454  * time step. If this surpasses the simulation time or if it surpasses the
455  * maximal CFL value, the time step is scaled down to ensure that this is
456  * respected.
457  */
458  virtual double
459  calculate_time_step() override;
460 
461 
462 public:
463  SimulationControlTransient(Parameters::SimulationControl param);
464 
465  virtual void
466  print_progression(const ConditionalOStream &pcout) override;
467 
468  /**
469  * @brief Proceeds with the simulation until the end condition is reached
470  */
471  virtual bool
472  integrate() override;
473 
474 
475  /**
476  * @brief Ends the simulation when the end time is reached
477  */
478  virtual bool
479  is_at_end() override;
480 };

89 template <int dim>
90 void
91 GLSSharpNavierStokesSolver<dim>::generate_cut_cells_map()
92 {
93  // check all the cells if they are cut or not. Put the information in a map
94  // with the key being the cell.
95  TimerOutput::Scope t(this->computing_timer, "cut_cells_mapping");
96  std::map<types::global_dof_index, Point<dim>> support_points;
97 
98  // A vector of the unordered map. Each map stores if a point is inside or
99  // outside of a given particle.
100  std::vector<std::unordered_map<types::global_dof_index, bool>>
101  inside_outside_support_point_vector(particles.size());
102  DoFTools::map_dofs_to_support_points(*this->mapping,
103  this->dof_handler,
104  support_points);
105 
106  // When the finite element order > 1, overconstrained cells are impossible
107  // since there is always at least a DOF inside the element that is not
108  // overconstrained. We therefore only have to check when the velocity order
109  // == 1.
110  const bool mapping_overconstrained_cells =
111  this->simulation_parameters.fem_parameters.velocity_order == 1;
112 
113  cut_cells_map.clear();
114  cells_inside_map.clear();
115  overconstrained_fluid_cell_map.clear();
116  if (mapping_overconstrained_cells)
117  {
118  local_dof_overconstrained.reinit(this->locally_owned_dofs,
119  this->mpi_communicator);
120  dof_overconstrained.reinit(this->locally_owned_dofs,
121  this->locally_relevant_dofs,
122  this->mpi_communicator);
123  }
124 
125  const auto &cell_iterator = this->dof_handler.active_cell_iterators();
126  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
127  const unsigned int dofs_per_face = this->fe->dofs_per_face;
128 
129  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
130  std::vector<types::global_dof_index> local_face_dof_indices(dofs_per_face);
131 
132  auto &v_x_fe = this->fe->get_sub_fe(0, 1);
133  const unsigned int dofs_per_cell_local_v_x = v_x_fe.dofs_per_cell;
134  // // Loop on all the cells and check if they are cut.
135  for (const auto &cell : cell_iterator)
136  {
137  if (cell->is_locally_owned() || cell->is_ghost())
138  {
139  bool cell_is_cut = false;
140  bool cell_is_inside = false;
141  unsigned int particle_id_which_cuts_this_cell = 0;
142  unsigned int particle_id_in_which_this_cell_is_embedded = 0;
143  unsigned int number_of_particles_cutting_this_cell = 0;
144  // is the particle index that cuts the cell if it's cut. If the cell
145  // is not cut the default value is stored (0). If the cell is not cut
146  // this value will never be used.
147  // number_of_particles_cutting_this_cell count the number of particles
148  // that cut a cell if multiple particles cut this cell. This is used
149  // to treat cells that are cut by multiple particles differently.
150  cell->get_dof_indices(local_dof_indices);
151 
152  for (unsigned int p = 0; p < particles.size(); ++p)
153  {
154  // Particles defined from a stl or an iges don't have a signed
155  // distance function. To identify cells that are cut a special
156  // algorithm is required. For this type of particle, no cell are
157  // identified as being inside the particle. This means that the
158  // fluid is always solved inside the particle.
159  if (particles[p].shape->additional_info_on_shape == "iges")
160  {
161  cell_is_cut =
162  cell_cut_by_p_absolute_distance(cell, support_points, p);
163  particle_id_which_cuts_this_cell = p;
164  if (cell_is_cut)
165  {
166  number_of_particles_cutting_this_cell += 1;
167  }
168  }
169  else
170  {
171  unsigned int nb_dof_inside = 0;
172  for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
173  {
174  // Count the number of DOFs that are inside
175  // of the particles. If all the DOfs are on one side
176  // the cell is not cut by the boundary.
177  if (0 == this->fe->system_to_component_index(j).first)
178  {
179  // Check if we already have checked this point in a
180  // previously evaluated cell. If we didn't find it in
181  // the previous evaluation, we assess whether the DOF
182  // is inside or outside the shape.
183  if (particles[p].get_levelset(
184  support_points[local_dof_indices[j]], cell) <=
185  0)
186  {
187  ++nb_dof_inside;
188  inside_outside_support_point_vector
189  [p][local_dof_indices[j]] = true;
190  }
191  }
192  }
193 
194  // If some DOFs are inside the boundary, the cell is inside
195  // the particle or cut by the particle.
196  if (nb_dof_inside != 0)
197  {
198  // If all the DOFs are inside the boundary this cell is
199  // inside the particle. Otherwise the particle is cut.
200  if (nb_dof_inside == dofs_per_cell_local_v_x)
201  {
202  // We register only the particle with the lowest id
203  // as the particle in which this cell is embedded if
204  // the cell is in multiple particles.
205  if (number_of_particles_cutting_this_cell == 0)
206  {
207  cell_is_cut = false;
208  particle_id_which_cuts_this_cell = 0;
209  cell_is_inside = true;
210  particle_id_in_which_this_cell_is_embedded = p;
211  }
212  break;
213  }
214  else
215  {
216  // We only register the particle with the lowest id as
217  // the particle by this cell is cut if the cell is cut
218  // in multiple particles.
219  if (number_of_particles_cutting_this_cell == 0)
220  {
221  cell_is_cut = true;
222  particle_id_which_cuts_this_cell = p;
223  cell_is_inside = false;
224  particle_id_in_which_this_cell_is_embedded = 0;
225  }
226  number_of_particles_cutting_this_cell += 1;
227  }
228  }
229  else
230  {
231  if (number_of_particles_cutting_this_cell == 0)
232  {
233  cell_is_cut = false;
234  particle_id_which_cuts_this_cell = 0;
235  cell_is_inside = false;
236  particle_id_in_which_this_cell_is_embedded = 0;
237  }
238  }
239  }
240  }
241 
242  if (mapping_overconstrained_cells)
243  {
244  // If a cell is cut, we register its DOFs as "cut" also. This
245  // will allow us to detect fluid cells that have all their DOFs
246  // cut, and consider them as cut cells.
247  if (cell_is_cut)
248  {
249  size_t id;
250  for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
251  {
252  id = local_dof_indices[j];
253  local_dof_overconstrained(id) = 1;
254  }
255  }
256 
257  // We loop on every face of the cell, and if the face is at a
258  // boundary we count it as constrained.
259  size_t nb_faces = cell->n_faces();
260  for (unsigned int f = 0; f < nb_faces; ++f)
261  {
262  const auto face = cell->face(f);
263  if (face->at_boundary())
264  {
265  face->get_dof_indices(local_face_dof_indices);
266  size_t id;
267  for (unsigned int v = 0;
268  v < local_face_dof_indices.size();
269  ++v)
270  {
271  id = local_face_dof_indices[v];
272  local_dof_overconstrained(id) = 1;
273  }
274  }
275  }
276  }
277 
278  cut_cells_map[cell] = {cell_is_cut,
279  particle_id_which_cuts_this_cell,
280  number_of_particles_cutting_this_cell};
281  cells_inside_map[cell] = {cell_is_inside,
282  particle_id_in_which_this_cell_is_embedded};
283  }
284  }
285 
286  if (mapping_overconstrained_cells)
287  {
288  dof_overconstrained = local_dof_overconstrained;
289  dof_overconstrained.compress(VectorOperation::insert);
290 
291  for (const auto &cell : cell_iterator)
292  {
293  if (cell->is_locally_owned() || cell->is_ghost())
294  {
295  bool cell_is_cut;
296  bool cell_is_inside;
297  cell->get_dof_indices(local_dof_indices);
298  std::tie(cell_is_cut, std::ignore, std::ignore) =
299  cut_cells_map[cell];
300  std::tie(cell_is_inside, std::ignore) = cells_inside_map[cell];
301  if (!cell_is_cut && !cell_is_inside)
302  {
303  unsigned int number_of_vertices_in_cell =
304  local_dof_indices.size();
305  unsigned int number_of_vertices_cut = 0;
306  unsigned int particle_current_id =
307  std::numeric_limits<int>::max();
308 
309  double particle_current_distance(
310  std::numeric_limits<double>::max());
311  double particle_candidate_distance(
312  std::numeric_limits<double>::max());
313 
314  unsigned int global_dof_id;
315  // We count the number of vertices that are constrained.
316  for (unsigned int j = 0; j < number_of_vertices_in_cell; ++j)
317  {
318  global_dof_id = local_dof_indices[j];
319  if (dof_overconstrained(global_dof_id) > 0)
320  {
321  number_of_vertices_cut += 1;
322  }
323  }
324  // We check which particle is the closest to the cell, so
325  // that we know which one to use for applying the immersed
326  // boundary constraints.
327  if (number_of_vertices_in_cell == number_of_vertices_cut)
328  {
329  for (unsigned int p = 0; p < particles.size(); p++)
330  {
331  particle_candidate_distance =
332  particles[p].get_levelset(cell->barycenter(), cell);
333  if (abs(particle_candidate_distance) <
334  abs(particle_current_distance))
335  {
336  particle_current_id = p;
337  particle_current_distance =
338  particle_candidate_distance;
339  }
340  }
341  overconstrained_fluid_cell_map[cell] = {
342  true, particle_current_id, particle_current_distance};
343  }
344  }
345  }
346  }
347  }

350 template <int dim>
351 void
352 GLSSharpNavierStokesSolver<dim>::refinement_control(
353  const bool initial_refinement)
354 {
355  // We add a post-refinement check to update precalculations only if needed.
356  // Two booleans are used: the first one is used to update the precalculations,
357  // and the other one is used to remove superfluous nodes. We make this
358  // separation between the steps to avoid removing useful information until all
359  // triangulation refinement steps are done.
360  bool update_precalculations_flag = false;
361  bool remove_superfluous_data_flag = false;
362  this->triangulation->signals.post_refinement.connect(
363  [&update_precalculations_flag]() { update_precalculations_flag = true; });
364 
365  // This function applies the various refinement steps depending on the
366  // parameters and the state.
367  if (initial_refinement)
368  {
369  // Apply the initial box refinement
370  this->box_refine_mesh();
371  update_precalculations_for_ib();
372  }
373  if (this->simulation_parameters.particlesParameters
374  ->time_extrapolation_of_refinement_zone ||
375  initial_refinement)
376  {
377  // Stores variable for refinement around the particle.
378  double temp_refine =
379  this->simulation_parameters.mesh_adaptation.variables.begin()
380  ->second.refinement_fraction;
381  double temp_coarse =
382  this->simulation_parameters.mesh_adaptation.variables.begin()
383  ->second.coarsening_fraction;
384  this->simulation_parameters.mesh_adaptation.variables.begin()
385  ->second.refinement_fraction = 0;
386  this->simulation_parameters.mesh_adaptation.variables.begin()
387  ->second.coarsening_fraction = 0;
388 
389  for (unsigned int i = 0;
390  i <
391  this->simulation_parameters.particlesParameters->initial_refinement;
392  ++i)
393  {
394  update_precalculations_flag = false;
395  this->pcout << "Initial refinement around IB particles - Step : "
396  << i + 1 << " of "
397  << this->simulation_parameters.particlesParameters
398  ->initial_refinement
399  << std::endl;
400  refine_ib();
401  NavierStokesBase<dim, TrilinosWrappers::MPI::Vector, IndexSet>::
402  refine_mesh();
403  if (update_precalculations_flag)
404  {
405  remove_superfluous_data_flag = true;
406  update_precalculations_for_ib();
407  }
408  }
409  this->simulation_parameters.mesh_adaptation.variables.begin()
410  ->second.refinement_fraction = temp_refine;
411  this->simulation_parameters.mesh_adaptation.variables.begin()
412  ->second.coarsening_fraction = temp_coarse;
413  }
414  if (initial_refinement == false)
415  {
416  update_precalculations_flag = false;
417  refine_ib();
418  NavierStokesBase<dim, TrilinosWrappers::MPI::Vector, IndexSet>::
419  refine_mesh();
420  if (update_precalculations_flag)
421  {
422  remove_superfluous_data_flag = true;
423  update_precalculations_for_ib();
424  }
425  }
426  if (remove_superfluous_data_flag || initial_refinement)
427  {
428  for (unsigned int p_i = 0; p_i < particles.size(); ++p_i)
429  {
430  TimerOutput::Scope t(this->computing_timer, "removing_rbf_nodes");
431  particles[p_i].remove_superfluous_data(
432  this->dof_handler, particles[p_i].mesh_based_precalculations);
433  }
434  }
435 }

439 template <int dim>
440 bool
441 GLSSharpNavierStokesSolver<dim>::cell_cut_by_p_absolute_distance(
442  const typename DoFHandler<dim>::active_cell_iterator &cell,
443  std::map<types::global_dof_index, Point<dim>> &support_points,
444  unsigned int p)
445 {
446  // This function aims at defining if a cell is cut when the level set used to
447  // define the particle is not signed. This function works by analyzing the
448  // location of the projection of the support point and the cell centroid on
449  // the particle's surface. If the projected point at the surface of the
450  // particle is inside the cell then the cell is cut.
451 
452  // First step: define useful variables.
453  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
454  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
455  cell->get_dof_indices(local_dof_indices);
456  bool cell_is_cut = false;
457  Point<dim> centroid_of_cell = cell->barycenter();
458  Point<dim> projected_point;
459  particles[p].closest_surface_point(centroid_of_cell, projected_point, cell);
460 
461  // Check the centroid of the cell first.
462  if (cell->point_inside(projected_point))
463  {
464  cell_is_cut = true;
465  return cell_is_cut;
466  }
467  else
468  {
469  // If the projection of the centroid is not inside the cell we also check
470  // for each dof support point of the cell.
471  for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
472  {
473  // Only check for the support point of the velocity in x.
474  if (0 == this->fe->system_to_component_index(j).first)
475  {
476  particles[p].closest_surface_point(
477  support_points[local_dof_indices[j]], projected_point, cell);
478 
479  if (cell->point_inside(projected_point))
480  {
481  cell_is_cut = true;
482  break;
483  }
484  if ((projected_point - support_points[local_dof_indices[j]])
485  .norm() < cell->diameter() * 0.1)
486  {
487  cell_is_cut = true;
488  break;
489  }
490  }
491  }
492  }
493  return cell_is_cut;
494 }

496 template <int dim>
497 void
498 GLSSharpNavierStokesSolver<dim>::optimized_generate_cut_cells_map()
499 {
500  TimerOutput::Scope t(this->computing_timer, "optimized_cut_cells_mapping");
501  MappingQ1<dim> mapping;
502  unsigned int max_children = GeometryInfo<dim>::max_children_per_cell;
503 
504  std::set<typename DoFHandler<dim>::cell_iterator> all_candidate_cells;
505  std::set<typename DoFHandler<dim>::cell_iterator>
506  previous_all_candidate_cells;
507 
508  cut_cells_map.clear();
509  cells_inside_map.clear();
510  for (const auto &cell : this->dof_handler.active_cell_iterators())
511  {
512  if (cell->is_locally_owned() || cell->is_ghost())
513  {
514  cut_cells_map[cell] = {false, 0, 0};
515  cells_inside_map[cell] = {false, 0};
516  }
517  }
518 
519  // Loop over particles in reverse.
520  // This is done in reverse to guarantee that the lowest particle ID is
521  // associated to the cut cell or the inside particle cell. For instance, if
522  // you have two particles cutting the same cell, it guarantees that the lowest
523  // ID particle is associated with the cell in the map.
524  for (int p = particles.size() - 1; p >= 0; --p)
525  {
526  bool empty = true;
527  unsigned int lvl_iter = 0;
528 
529  // Fix max level search
530  unsigned int max_lvl_search =
531  this->dof_handler.get_triangulation().n_levels() - 2;
532  if (max_lvl_search < 4)
533  max_lvl_search = this->dof_handler.get_triangulation().n_levels();
534 
535  // Search for candidates until at least one is found or until the
536  // max_lvl_search is reached
537  while (empty || lvl_iter < max_lvl_search)
538  {
539  const auto &cell_iterator =
540  this->dof_handler.cell_iterators_on_level(lvl_iter);
541 
542  // Loop over the cells on level lvl_iter of the mesh
543  for (const auto &cell : cell_iterator)
544  {
545  auto is_candidate = generate_cut_cell_candidates(cell, p);
546 
547  // check whether the cell is candidate for inside of cut
548  if (is_candidate.first)
549  {
550  all_candidate_cells.insert(cell);
551  }
552  if (is_candidate.second)
553  {
554  all_candidate_cells.insert(cell);
555  }
556  }
557  if (all_candidate_cells.size() > 0)
558  empty = false;
559 
560  lvl_iter += 1;
561  }
562 
563  // Store list with all candidate cells
564  previous_all_candidate_cells = all_candidate_cells;
565 
566  // If there are candidate cells
567  if (all_candidate_cells.size() != 0)
568  {
569  bool all_cells_are_active = false;
570 
571  // Loop until candidate cells are active
572  while (!all_cells_are_active)
573  {
574  all_cells_are_active = true;
575  all_candidate_cells.clear();
576 
577  // loop over the last set of candidates to check if it is still a
578  // candidate
579  for (auto cell : previous_all_candidate_cells)
580  {
581  auto is_candidate = generate_cut_cell_candidates(cell, p);
582 
583  // Check if cell is inside particles
584  if (is_candidate.first)
585  {
586  // If it is an active cell, add it to cells_inside_map
587  if (cell->is_active())
588  {
589  cells_inside_map[cell] = {true, p};
590  }
591  else
592  {
593  // If we are here, the cell has children.
594  all_cells_are_active = false;
595  for (unsigned int j = 0; j < max_children; ++j)
596  {
597  // Store the children of the cell for
598  // further checks.
599  all_candidate_cells.insert(cell->child(j));
600  }
601  }
602  }
603 
604  if (is_candidate.second)
605  {
606  // If it is an active cell, add it to cells_cut_map
607  if (cell->is_active())
608  {
609  cut_cells_map[cell] = {true, p, 0};
610  }
611  else
612  {
613  // If we are here, the cell has children.
614  all_cells_are_active = false;
615  for (unsigned int j = 0; j < max_children; ++j)
616  {
617  // Store the children of the cell for
618  // further checks.
619  all_candidate_cells.insert(cell->child(j));
620  }
621  }
622  }
623  }
624  previous_all_candidate_cells.clear();
625  previous_all_candidate_cells = all_candidate_cells;
626  }
627  }
628  }
629 }

632 template <int dim>
633 std::pair<bool, bool>
634 GLSSharpNavierStokesSolver<dim>::generate_cut_cell_candidates(
635  const typename DoFHandler<dim>::cell_iterator &cell,
636  const unsigned int p_id)
637 {
638  bool cell_is_inside = false;
639  bool cell_is_cut = false;
640 
641  double search_radius = particles[p_id].radius * (1.0 - 1e-07);
642  auto position = particles[p_id].position;
643 
644  bool point_inside_cell = cell->point_inside(position);
645 
646  // Check how many vertices are inside the particle
647  unsigned int nb_vertices_inside = 0;
648  for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_cell; ++i)
649  {
650  if ((cell->vertex(i) - position).norm() < search_radius)
651  ++nb_vertices_inside;
652  }
653 
654  // If vertices are found inside the particle
655  if (nb_vertices_inside > 0)
656  {
657  // If the number of vertices inside the cell is equal to the number of
658  // vertices per cell, the cell is inside the particle
659  if (nb_vertices_inside == GeometryInfo<dim>::vertices_per_cell)
660  {
661  cell_is_inside = true;
662  cell_is_cut = false;
663  }
664  // Otherwise, the cell is cut
665  else
666  {
667  cell_is_inside = false;
668  cell_is_cut = true;
669  }
670  return {cell_is_inside, cell_is_cut};
671  }
672 
673  // If the particles is inside the cell and it is not known whether all
674  // vertices are inside the particles or not, set all true by default
675  if (point_inside_cell)
676  {
677  cell_is_inside = true;
678  cell_is_cut = true;
679  return {cell_is_inside, cell_is_cut};
680  }
681 
682  // The last check consists of projecting the particle's position on the cells'
683  // face and checking whether the projected points are within the particle. If
684  // one of the projected points is inside the particle, either the cell is
685  // inside or is cut by the particles. If this is true, the next
686  // level will be tested again by this function.
687 
688  // Initialize superpoint of manifold
689  std::vector<Point<dim>> manifold_points(
690  GeometryInfo<dim - 1>::vertices_per_cell);
691 
692  // Loop through faces
693  for (const auto face : cell->face_indices())
694  {
695  auto face_iter = cell->face(face);
696  auto &local_face_manifold = face_iter->get_manifold();
697 
698  // Loop through face vertices
699  for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_face; ++i)
700  {
701  // Assign vertex to manifold superpoint
702  manifold_points[i] = face_iter->vertex(i);
703  }
704 
705  // Create array of points on face
706  auto surrounding_face_points =
707  make_array_view(manifold_points.begin(), manifold_points.end());
708 
709  // Loop through face vertices
710  for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_face; ++i)
711  {
712  // Project points to face
713  Point<dim> projected_point =
714  local_face_manifold.project_to_manifold(surrounding_face_points,
715  position);
716 
717  bool projected_point_over_face = true;
718  Point<dim> unit_cell_projected_point;
719  Tensor<1, dim> projected_point_tensor(projected_point);
720 
721  // Check whether the projected point is within the cell
722  try
723  {
724  projected_point_over_face = cell->point_inside(projected_point);
725 
726  if ((position - projected_point).norm() < search_radius &&
727  projected_point_over_face)
728  {
729  cell_is_inside = true;
730  cell_is_cut = true;
731  return {cell_is_inside, cell_is_cut};
732  }
733  }
734  // If the method crashes, change default for false
735  catch (...)
736  {
737  projected_point_over_face = false;
738  }
739 
740  for (unsigned int d = 0; d != dim; d++)
741  {
742  if (projected_point_tensor[d] < 0 ||
743  projected_point_tensor[d] > 1)
744  {
745  projected_point_over_face = false;
746  }
747  }
748  }
749  }
750  return {cell_is_inside, cell_is_cut};
751 }

807 template <int dim>
808 void
809 GLSSharpNavierStokesSolver<dim>::refine_ib()
810 {
811  TimerOutput::Scope t(this->computing_timer, "refine_around_ib");
812  Point<dim> center_immersed;
813  std::map<types::global_dof_index, Point<dim>> support_points;
814  DoFTools::map_dofs_to_support_points(*this->mapping,
815  this->dof_handler,
816  support_points);
817  double dt = this->simulation_control->get_time_steps_vector()[0];
818 
819  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
820  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
821 
822  bool extrapolate_particle_position =
823  this->simulation_parameters.particlesParameters
824  ->time_extrapolation_of_refinement_zone;
825 
826  double smallest_cut_cell = std::numeric_limits<double>::max();
827  bool minimal_crown_refinement_enabled =
828  abs(this->simulation_parameters.particlesParameters->outside_radius - 1) <
829  1e-16 &&
830  abs(this->simulation_parameters.particlesParameters->inside_radius - 1) <
831  1e-16;
832  if (minimal_crown_refinement_enabled)
833  {
834  const auto &cell_iterator_smallest_cell =
835  this->dof_handler.active_cell_iterators();
836  for (const auto &cell : cell_iterator_smallest_cell)
837  {
838  if (cell->is_locally_owned())
839  smallest_cut_cell = std::min(smallest_cut_cell, cell->diameter());
840  }
841  smallest_cut_cell =
842  Utilities::MPI::min(smallest_cut_cell, this->mpi_communicator);
843  }
844 
845  const auto &cell_iterator = this->dof_handler.active_cell_iterators();
846  for (const auto &cell : cell_iterator)
847  {
848  if (cell->is_locally_owned())
849  {
850  cell->get_dof_indices(local_dof_indices);
851  for (unsigned int p = 0; p < particles.size(); ++p)
852  {
853  unsigned int count_small = 0;
854  center_immersed = particles[p].position;
855  Tensor<1, dim> r;
856  r[0] = particles[p].radius;
857 
858  Point<dim> particle_position;
859  Tensor<1, 3> particle_orientation;
860  if (extrapolate_particle_position)
861  {
862  particle_position = particles[p].position;
863  particle_orientation = particles[p].orientation;
864  particles[p].orientation =
865  particles[p].orientation + particles[p].omega * dt;
866  particles[p].position[0] =
867  particles[p].position[0] + particles[p].velocity[0] * dt;
868  particles[p].position[1] =
869  particles[p].position[1] + particles[p].velocity[1] * dt;
870 
871  if constexpr (dim == 3)
872  {
873  particles[p].position[2] = particles[p].position[2] +
874  particles[p].velocity[2] * dt;
875  }
876  particles[p].set_position(particles[p].position);
877  particles[p].set_orientation(particles[p].orientation);
878  }
879  // Check if a point on the random point on the IB is contained in
880  // that cell. If the particle is much smaller than the cell, all
881  // its vertices may be outside of the particle. In that case the
882  // cell won't be refined. To prevent that, we check if a random
883  // point on the boundary is contained in the cell.
884  bool cell_as_ib_inside =
885  cell->point_inside(particles[p].position + r);
886  for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
887  {
888  // Only check the dof of velocity in x.
889  if (this->fe->system_to_component_index(j).first == 0)
890  {
891  // Count the number of DOFs that fall in the refinement
892  // zone around the particle. To fall in the zone, the
893  // radius of the DOFs to the center of the particle must
894  // be bigger than the inside radius of the refinement zone
895  // and smaller than the outside radius of the refinement
896  // zone.
897  bool is_inside_crown;
898  if (minimal_crown_refinement_enabled)
899  {
900  // The factor should be equal to the length of the
901  // current smallest cell. Since cell->diameter()
902  // returns the longest diagonal, we divide the
903  // diameter by the diagonal of a unit hypercube to
904  // obtain the correct length.
905  double factor = 1 / sqrt(dim);
906  is_inside_crown = particles[p].is_inside_crown(
907  support_points[local_dof_indices[j]],
908  factor * smallest_cut_cell,
909  -factor * smallest_cut_cell,
910  true, // indicates that we use the absolute
911  // distance definition
912  cell);
913  }
914  else
915  {
916  is_inside_crown = particles[p].is_inside_crown(
917  support_points[local_dof_indices[j]],
918  this->simulation_parameters.particlesParameters
919  ->outside_radius,
920  this->simulation_parameters.particlesParameters
921  ->inside_radius,
922  false, // indicates that we use the
923  // radius relative distance definition
924  cell);
925  }
926  if (is_inside_crown)
927  ++count_small;
928  }
929  }
930 
931  if (extrapolate_particle_position)
932  {
933  particles[p].position = particle_position;
934  particles[p].orientation = particle_orientation;
935  particles[p].set_position(particles[p].position);
936  particles[p].set_orientation(particles[p].orientation);
937  }
938  if (count_small > 0 || cell_as_ib_inside)
939  {
940  cell->set_refine_flag();
941  break;
942  }
943  }
944  }
945  }
946 }

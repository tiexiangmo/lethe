40 /**
41  * This class is used for local-local and local-ghost fine particle-particle
42  * contact search. Fine search is used to find all the particle pairs which are
43  * physically in contact and obtain all the required information for calculation
44  * of the contact force
45  *
46  * @note
47  *
48  * @author Shahab Golshan, Polytechnique Montreal 2019-
49  */
50 
51 template <int dim>
52 class ParticleParticleFineSearch
53 {
54 public:
55  ParticleParticleFineSearch();
56 
57  /**
58  * Iterates over a vector of maps (pairs_in_contact) to see if the particles
59  * which were in contact in the last time step, are still in contact or not.
60  * If they are still in contact it will update the collision info, including
61  * tangential overlap, based on new properties of the particle pair, if they
62  * are not in contact anymore it will delete the pair from the
63  * pairs_in_contact and also its information from pairs_in_contact_info.
64  * Then it iterates over the contact candidates from broad search to see if
65  * they already exist in the pairs_in_contact or not, if they are not in the
66  * pairs_in_contact and they have overlap, the pair will be added to the
67  * pairs_in_contact and its contact information will be stored in the
68  * corresponding element of the pairs_in_contact_info
69  *
70  * @param particle_container A container that is used to obtain iterators to
71  * particles using their ids
72  * @param adjacent_particles A map of maps which stores all the required
73  * information for calculation of the contact force of particle pairs
74  * @param contact_pair_candidates The output of broad search which shows
75  * contact pair candidates
76  * @param neighborhood_threshold A value which defines the neighbor particles
77  * @param periodic_offset A tensor of the periodic offset to change the
78  * particle location of the particles on the periodic boundary 1 side,
79  * the tensor as 0.0 values by default
80  */
81 
82  void
83  particle_particle_fine_search(
84  typename DEM::dem_data_structures<dim>::particle_index_iterator_map
85  &particle_container,
86  typename DEM::dem_data_structures<dim>::adjacent_particle_pairs
87  &adjacent_particles,
88  const typename DEM::dem_data_structures<dim>::particle_particle_candidates
89  &contact_pair_candidates,
90  const double neighborhood_threshold,
91  const Tensor<1, dim> periodic_offset = Tensor<1, dim>());
92 };

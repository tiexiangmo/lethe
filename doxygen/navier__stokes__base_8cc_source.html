60 template <int dim, typename VectorType, typename DofsType>
61 NavierStokesBase<dim, VectorType, DofsType>::NavierStokesBase(
62  SimulationParameters<dim> &p_nsparam)
63  : PhysicsSolver<VectorType>(
64  p_nsparam.non_linear_solver.at(PhysicsID::fluid_dynamics))
65  , mpi_communicator(MPI_COMM_WORLD)
66  , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
67  , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
68  , dof_handler()
69  , computing_timer(this->mpi_communicator,
70  this->pcout,
71  TimerOutput::summary,
72  TimerOutput::wall_times)
73  , simulation_parameters(p_nsparam)
74  , flow_control(simulation_parameters.flow_control)
75  , velocity_fem_degree(p_nsparam.fem_parameters.velocity_order)
76  , pressure_fem_degree(p_nsparam.fem_parameters.pressure_order)
77  , number_quadrature_points(p_nsparam.fem_parameters.velocity_order + 1)
78  , mesh_controller(p_nsparam.mesh_adaptation.maximum_number_elements)
79 {
80  if (simulation_parameters.mesh.simplex)
81  {
82  // for simplex meshes
83  const FE_SimplexP<dim> velocity_fe(
84  p_nsparam.fem_parameters.velocity_order);
85  const FE_SimplexP<dim> pressure_fe(
86  p_nsparam.fem_parameters.pressure_order);
87  fe = std::make_shared<FESystem<dim>>(velocity_fe, dim, pressure_fe, 1);
88  mapping = std::make_shared<MappingFE<dim>>(velocity_fe);
89  cell_quadrature =
90  std::make_shared<QGaussSimplex<dim>>(number_quadrature_points);
91  face_quadrature =
92  std::make_shared<QGaussSimplex<dim - 1>>(number_quadrature_points);
93  triangulation =
94  std::make_shared<parallel::fullydistributed::Triangulation<dim>>(
95  this->mpi_communicator);
96  dof_handler.clear();
97  dof_handler.reinit(*this->triangulation);
98  }
99  else
100  {
101  // Usual case, for quad/hex meshes
102  fe = std::make_shared<FESystem<dim>>(
103  FE_Q<dim>(p_nsparam.fem_parameters.velocity_order),
104  dim,
105  FE_Q<dim>(p_nsparam.fem_parameters.pressure_order),
106  1);
107  mapping = std::make_shared<MappingQ<dim>>(
108  velocity_fem_degree, simulation_parameters.fem_parameters.qmapping_all);
109  cell_quadrature = std::make_shared<QGauss<dim>>(number_quadrature_points);
110  face_quadrature =
111  std::make_shared<QGauss<dim - 1>>(number_quadrature_points);
112  triangulation =
113  std::make_shared<parallel::distributed::Triangulation<dim>>(
114  this->mpi_communicator,
115  Triangulation<dim>::limit_level_difference_at_vertices,
116  (p_nsparam.linear_solver.at(fluid_dynamics).preconditioner ==
117  Parameters::LinearSolver::PreconditionerType::lsmg) ?
118  parallel::distributed::Triangulation<
119  dim>::construct_multigrid_hierarchy :
120  parallel::distributed::Triangulation<dim>::default_setting);
121  dof_handler.clear();
122  dof_handler.reinit(*this->triangulation);
123  }
124 
125  this->pcout.set_condition(
126  Utilities::MPI::this_mpi_process(this->mpi_communicator) == 0);
127 
128  // Check if the output directory exists
129  std::string output_dir_name =
130  simulation_parameters.simulation_control.output_folder;
131  struct stat buffer;
132 
133  // If output directory does not exist, create it
134  if (Utilities::MPI::this_mpi_process(this->mpi_communicator) == 0)
135  {
136  if (stat(output_dir_name.c_str(), &buffer) != 0)
137  {
138  create_output_folder(output_dir_name);
139  }
140  }
141 
142  if (simulation_parameters.simulation_control.method ==
143  Parameters::SimulationControl::TimeSteppingMethod::steady)
144  simulation_control = std::make_shared<SimulationControlSteady>(
145  simulation_parameters.simulation_control);
146  else if (simulation_parameters.simulation_control.method ==
147  Parameters::SimulationControl::TimeSteppingMethod::steady_bdf)
148  {
149  simulation_control = std::make_shared<SimulationControlAdjointSteady>(
150  simulation_parameters.simulation_control);
151  }
152  else
153  {
154  if (simulation_parameters.simulation_control.output_control ==
155  Parameters::SimulationControl::OutputControl::time)
156  simulation_control =
157  std::make_shared<SimulationControlTransientDynamicOutput>(
158  simulation_parameters.simulation_control);
159  else
160  simulation_control = std::make_shared<SimulationControlTransient>(
161  simulation_parameters.simulation_control);
162  }
163 
164  multiphysics = std::make_shared<MultiphysicsInterface<dim>>(
165  simulation_parameters, triangulation, simulation_control, this->pcout);
166 
167  // Pre-allocate memory for the previous solutions using the information
168  // of the BDF schemes
169  previous_solutions.resize(maximum_number_of_previous_solutions());
170 
171  // Change the behavior of the timer for situations when you don't want
172  // outputs
173  if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
174  this->computing_timer.disable_output();
175 
176  // Pre-allocate the force tables to match the number of boundary conditions
177  forces_on_boundaries.resize(simulation_parameters.boundary_conditions.size);
178  torques_on_boundaries.resize(simulation_parameters.boundary_conditions.size);
179  forces_tables.resize(simulation_parameters.boundary_conditions.size);
180  torques_tables.resize(simulation_parameters.boundary_conditions.size);
181 
182  // Get the exact solution from the parser
183  exact_solution = &simulation_parameters.analytical_solution->uvwp;
184 
185  // If there is a forcing function, get it from the parser
186  forcing_function = simulation_parameters.source_term.navier_stokes_source;
187 
188  if (this->simulation_parameters.post_processing.calculate_average_velocities)
189  average_velocities =
190  std::make_shared<AverageVelocities<dim, VectorType, DofsType>>(
191  dof_handler);
192 
193  this->pcout << "Running on "
194  << Utilities::MPI::n_mpi_processes(this->mpi_communicator)

546 // Do an iteration with the NavierStokes Solver
547 // Handles the fact that we may or may not be at a first
548 // iteration with the solver and sets the initial condition
549 template <int dim, typename VectorType, typename DofsType>
550 void
551 NavierStokesBase<dim, VectorType, DofsType>::iterate()
552 {
553  auto &present_solution = this->present_solution;
554 
555  if (simulation_parameters.multiphysics.fluid_dynamics)
556  {
557  // Solve and percolate the auxiliary physics that should be treated BEFORE
558  // the fluid dynamics
559  multiphysics->solve(false,
560  simulation_parameters.simulation_control.method);
561  multiphysics->percolate_time_vectors(false);
562 
563  if (simulation_parameters.non_linear_solver.at(PhysicsID::fluid_dynamics)
564  .verbosity != Parameters::Verbosity::quiet ||
565  simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
566  .verbosity != Parameters::Verbosity::quiet)
567  announce_string(this->pcout, "Fluid Dynamics");
568  PhysicsSolver<VectorType>::solve_non_linear_system(false);
569 
570  // Solve and percolate the auxiliary physics that should be treated AFTER
571  // the fluid dynamics
572  multiphysics->solve(true,
573  simulation_parameters.simulation_control.method);
574  // Dear future Bruno, percolating auxiliary physics before fluid dynamics
575  // is necessary because of the checkpointing mechanism. You spent an
576  // evening debugging this, trust me.
577  multiphysics->percolate_time_vectors(true);
578  }
579  else
580  {
581  // Fluid dynamics is not to be solved, but rather specified. Update
582  // condition and move on.
583 
584  // Solve and percolate the auxiliary physics that should be treated
585  // BEFORE the fluid dynamics
586  multiphysics->solve(false,
587  simulation_parameters.simulation_control.method);
588  multiphysics->percolate_time_vectors(false);
589 
590  if (simulation_parameters.multiphysics.use_time_average_velocity_field)
591  {
592  // We get the solution via the average solution
593  this->local_evaluation_point =
594  this->average_velocities->get_average_velocities();

874

876  this->simulation_parameters.mesh_adaptation.maximum_number_elements;
877  // Override the maximal number of elements if the controller is used. The
878  // controller will find a coarsening_factor that respects the user-defined

910  typename std::map<types::boundary_id,

912  present_solution,
913  estimated_error_per_cell,
914  this->fe->component_mask(velocity));

992  .minimum_refinement_level);
993  cell !=
994  tria.end_active(
995  this->simulation_parameters.mesh_adaptation.minimum_refinement_level);

1668 template <int dim, typename VectorType, typename DofsType>
1669 void
1670 NavierStokesBase<dim, VectorType, DofsType>::establish_solid_domain(
1671  const bool non_zero_constraints)
1672 {
1673  // If there are no solid regions, there is no work to be done and we can
1674  // return
1675  if (simulation_parameters.physical_properties_manager
1676  .get_number_of_solids() == 0)
1677  return;
1678 
1679  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
1680  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
1681 
1682  // We will need to identify which pressure degrees of freedom are connected to
1683  // fluid region For these we won't establish a zero pressure equation.
1684  std::unordered_set<types::global_dof_index> dof_is_connected_to_fluid;
1685  std::unordered_set<types::global_cell_index> cell_is_connected_to_fluid;
1686 
1687  // Loop through all cells to identify which cells are solid. This first step
1688  // is used to 1) constraint the velocity degree of freedom to be zero in the
1689  // solid region and 2) to identify which pressure degrees of freedom are
1690  // connected to fluid cells
1691  for (const auto &cell : dof_handler.active_cell_iterators())
1692  {
1693  if (cell->is_locally_owned() || cell->is_ghost())
1694  {
1695  cell->get_dof_indices(local_dof_indices);
1696  // If the material_id is 1, the region is a solid region. Constraint
1697  // the velocity DOF to be zero.
1698  if (cell->material_id() > 0)
1699  {
1700  for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
1701  {
1702  const unsigned int component =
1703  this->fe->system_to_component_index(i).first;
1704  if (component < dim)
1705  {
1706  // We apply a constraint to all DOFS in the solid region,
1707  // whether they are locally owned or not.
1708  if (non_zero_constraints)
1709  {
1710  this->nonzero_constraints.add_line(
1711  local_dof_indices[i]);
1712  this->nonzero_constraints.set_inhomogeneity(
1713  local_dof_indices[i], 0);
1714  }
1715  else
1716  this->zero_constraints.add_line(local_dof_indices[i]);
1717  }
1718  }
1719  }
1720  else
1721  {
1722  // Cell is a fluid cell and as such all the pressure DOFS of that
1723  // cell are connected to the fluid. This will be used later on to
1724  // identify which pressure cells
1725  for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
1726  {
1727  const unsigned int component =
1728  this->fe->system_to_component_index(i).first;
1729  if (component == dim)
1730  {
1731  dof_is_connected_to_fluid.insert(local_dof_indices[i]);
1732  }
1733  }
1734  }
1735  }
1736  }
1737 
1738  // All pressure DOFs which are not connected to a fluid cell are constrained
1739  // to ensure that the system matrix has adequate conditioning
1740  for (const auto &cell : dof_handler.active_cell_iterators())
1741  {
1742  if (cell->is_locally_owned() || cell->is_ghost())
1743  {
1744  cell->get_dof_indices(local_dof_indices);
1745  // If the material_id is >0, the region is a solid region
1746  if (cell->material_id() > 0)
1747  {
1748  bool connected_to_fluid = false;
1749  // First check if cell is connected to a fluid cell by checking if
1750  // one of the DOF of the cell is connected to a fluid cell
1751  for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
1752  {
1753  auto search =
1754  dof_is_connected_to_fluid.find(local_dof_indices[i]);
1755  if (search != dof_is_connected_to_fluid.end())
1756  connected_to_fluid = true;
1757  }
1758 
1759  // All of the pressure DOFS with the cell are not connected to the
1760  // fluid. Consequently we fix a constraint for the
1761  // pressure on these dofs.
1762  if (!connected_to_fluid)
1763  {
1764  for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
1765  {
1766  const unsigned int component =
1767  this->fe->system_to_component_index(i).first;
1768 
1769  // Only pressure DOFs have an additional Dirichlet
1770  // condition
1771  if (component == dim)
1772  {
1773  // We only apply the constraint on the locally owned
1774  // pressure DOFs since we have no way of verifying if
1775  // the locally relevant DOFs are connected to a fluid
1776  // cell
1777 
1778  bool dof_is_locally_owned = false;
1779  // For the GLS-family of solvers, we only have a
1780  // single index set
1781  if constexpr (std::is_same_v<DofsType, IndexSet>)
1782  {
1783  dof_is_locally_owned =
1784  locally_owned_dofs.is_element(
1785  local_dof_indices[i]);
1786  }
1787 
1788  // For the GD-family of solvers, we have two index
1789  // set. One fore velocities and one for pressure.
1790  if constexpr (std::is_same_v<DofsType,

302 double
303 Tracer<dim>::calculate_L2_error()
304 {
305  auto mpi_communicator = triangulation->get_communicator();
306 
307  FEValues<dim> fe_values(*mapping,
308  *fe,
309  *cell_quadrature,
310  update_values | update_gradients |
311  update_quadrature_points | update_JxW_values);
312 
313  const unsigned int n_q_points = cell_quadrature->size();
314 
315  std::vector<double> q_exact_solution(n_q_points);
316  std::vector<double> q_scalar_values(n_q_points);
317 
318  auto &exact_solution = simulation_parameters.analytical_solution->tracer;
319  exact_solution.set_time(simulation_control->get_current_time());
320 
321  double l2error = 0.;
322 
323  for (const auto &cell : dof_handler.active_cell_iterators())
324  {
325  if (cell->is_locally_owned())
326  {
327  fe_values.reinit(cell);
328  fe_values.get_function_values(present_solution, q_scalar_values);
329 
330  // Get the exact solution at all gauss points
331  exact_solution.value_list(fe_values.get_quadrature_points(),
332  q_exact_solution);
333 
334  for (unsigned int q = 0; q < n_q_points; q++)
335  {
336  double sim = q_scalar_values[q];
337  double exact = q_exact_solution[q];
338  l2error += (sim - exact) * (sim - exact) * fe_values.JxW(q);
339  }
340  }
341  }
342  l2error = Utilities::MPI::sum(l2error, mpi_communicator);
343  return l2error;

421 template <int dim>
422 void
423 Tracer<dim>::calculate_tracer_statistics()
424 {
425  auto mpi_communicator = triangulation->get_communicator();
426 
427  FEValues<dim> fe_values(*mapping,
428  *fe,
429  *cell_quadrature,
430  update_values | update_gradients |
431  update_quadrature_points | update_JxW_values);
432 
433  const unsigned int n_q_points = cell_quadrature->size();
434  std::vector<double> q_tracer_values(n_q_points);
435 
436  double volume_integral = 0;
437  double max_tracer_value = DBL_MIN;
438  double min_tracer_value = DBL_MAX;
439 
440  for (const auto &cell : dof_handler.active_cell_iterators())
441  {
442  if (cell->is_locally_owned())
443  {
444  fe_values.reinit(cell);
445  fe_values.get_function_values(present_solution, q_tracer_values);
446 
447  for (unsigned int q = 0; q < n_q_points; q++)
448  {
449  volume_integral += q_tracer_values[q] * fe_values.JxW(q);
450  max_tracer_value = std::max(q_tracer_values[q], max_tracer_value);
451  min_tracer_value = std::min(q_tracer_values[q], min_tracer_value);
452  }
453  }
454  }
455  volume_integral = Utilities::MPI::sum(volume_integral, mpi_communicator);
456  double global_volume = GridTools::volume(*triangulation, *mapping);
457  double tracer_average = volume_integral / global_volume;
458 
459  double variance_integral = 0;
460 
461  for (const auto &cell : dof_handler.active_cell_iterators())
462  {
463  if (cell->is_locally_owned())
464  {
465  fe_values.reinit(cell);
466  fe_values.get_function_values(present_solution, q_tracer_values);
467 
468  for (unsigned int q = 0; q < n_q_points; q++)
469  {
470  variance_integral += (q_tracer_values[q] - tracer_average) *
471  (q_tracer_values[q] - tracer_average) *
472  fe_values.JxW(q);
473  }
474  }
475  }
476 
477  variance_integral = Utilities::MPI::sum(variance_integral, mpi_communicator);
478  double tracer_variance = variance_integral / global_volume;
479  double tracer_std_deviation = std::sqrt(tracer_variance);
480 
481  // Console output
482  if (simulation_parameters.post_processing.verbosity ==
483  Parameters::Verbosity::verbose)
484  {
485  this->pcout << "Tracer statistics : " << std::endl;
486  this->pcout << "\t Min : " << min_tracer_value << std::endl;
487  this->pcout << "\t Max : " << max_tracer_value << std::endl;
488  this->pcout << "\t Average : " << tracer_average << std::endl;
489  this->pcout << "\t Std-Dev : " << tracer_std_deviation << std::endl;
490  }
491 
492  statistics_table.add_value("time", simulation_control->get_current_time());
493  statistics_table.add_value("min", min_tracer_value);
494  statistics_table.add_value("max", max_tracer_value);
495  statistics_table.add_value("average", tracer_average);
496  statistics_table.add_value("std-dev", tracer_std_deviation);
497 }

795 void
796 Tracer<dim>::solve_linear_system(const bool initial_step,
797  const bool /*renewed_matrix*/)
798 {
799  TimerOutput::Scope t(this->computing_timer, "Solve linear system");
800 
801  auto mpi_communicator = triangulation->get_communicator();
802 
803  const AffineConstraints<double> &constraints_used =
804  initial_step ? nonzero_constraints : this->zero_constraints;
805 
806  const double absolute_residual =
807  simulation_parameters.linear_solver.at(PhysicsID::tracer).minimum_residual;
808  const double relative_residual =
809  simulation_parameters.linear_solver.at(PhysicsID::tracer).relative_residual;
810 
811  const double linear_solver_tolerance =
812  std::max(relative_residual * system_rhs.l2_norm(), absolute_residual);
813 
814  if (this->simulation_parameters.linear_solver.at(PhysicsID::tracer)
815  .verbosity != Parameters::Verbosity::quiet)
816  {
817  this->pcout << " -Tolerance of iterative solver is : "
818  << linear_solver_tolerance << std::endl;
819  }
820 
821  const double ilu_fill =
822  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_fill;
823  const double ilu_atol =
824  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_atol;
825  const double ilu_rtol =
826  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_rtol;
827  TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
828  ilu_fill, ilu_atol, ilu_rtol, 0);
829 
830  TrilinosWrappers::PreconditionILU ilu_preconditioner;
831 
832  ilu_preconditioner.initialize(system_matrix, preconditionerOptions);
833 
834  TrilinosWrappers::MPI::Vector completely_distributed_solution(
835  locally_owned_dofs, mpi_communicator);
836 
837  SolverControl solver_control(
838  simulation_parameters.linear_solver.at(PhysicsID::tracer).max_iterations,
839  linear_solver_tolerance,
840  true,
841  true);
842 
843  TrilinosWrappers::SolverGMRES::AdditionalData solver_parameters(
844  false,
845  simulation_parameters.linear_solver.at(PhysicsID::tracer)
846  .max_krylov_vectors);
847 
848 
849  TrilinosWrappers::SolverGMRES solver(solver_control, solver_parameters);
850 
851 
852  solver.solve(system_matrix,
853  completely_distributed_solution,
854  system_rhs,
855  ilu_preconditioner);
856 
857  if (simulation_parameters.linear_solver.at(PhysicsID::tracer).verbosity !=
858  Parameters::Verbosity::quiet)
859  {
860  this->pcout << " -Iterative solver took : " << solver_control.last_step()
861  << " steps " << std::endl;
862  }
863 
864  constraints_used.distribute(completely_distributed_solution);
865  newton_update = completely_distributed_solution;
866 }

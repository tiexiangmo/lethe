19 void
20 PhysicalPropertiesManager::establish_fields_required_by_model(
21  PhysicalPropertyModel &model)
22 {
23  // Loop through the map. The use of or (||) is there to ensure
24  // that if a field is already required, it won't be erased.
25  for (auto &f : required_fields)
26  {
27  f.second = f.second || model.depends_on(f.first);
28  }
29 }

43 void
44 PhysicalPropertiesManager::initialize(
45  Parameters::PhysicalProperties physical_properties)
46 {
47  is_initialized = true;
48 
49  number_of_fluids = physical_properties.number_of_fluids;
50  number_of_solids = physical_properties.number_of_solids;
51  number_of_material_interactions =
52  physical_properties.number_of_material_interactions;
53  fluid_fluid_interactions_with_material_interaction_ids =
54  physical_properties.fluid_fluid_interactions_with_material_interaction_ids;
55  fluid_solid_interactions_with_material_interaction_ids =
56  physical_properties.fluid_solid_interactions_with_material_interaction_ids;
57 
58  non_newtonian_flow = false;
59  constant_density = true;
60  constant_surface_tension = true;
61 
62  required_fields[field::temperature] = false;
63  required_fields[field::previous_temperature] = false;
64  required_fields[field::shear_rate] = false;
65  required_fields[field::pressure] = false;
66  required_fields[field::phase_order_cahn_hilliard] = false;
67 
68  // For each fluid, declare the physical properties
69  for (unsigned int f = 0; f < number_of_fluids; ++f)
70  {
71  density.push_back(
72  DensityModel::model_cast(physical_properties.fluids[f]));
73  establish_fields_required_by_model(*density[f]);
74 
75  // Store an indicator for the density to indicate if it is not constant
76  // This indicator is used elsewhere in the code to throw assertions
77  // if non-constant density is not implemented in a post-processing utility
78  if (!density.back()->is_constant_density_model())
79  constant_density = false;
80 
81  specific_heat.push_back(
82  SpecificHeatModel::model_cast(physical_properties.fluids[f]));
83  establish_fields_required_by_model(*specific_heat[f]);
84 
85  thermal_conductivity.push_back(
86  ThermalConductivityModel::model_cast(physical_properties.fluids[f]));
87  establish_fields_required_by_model(*thermal_conductivity[f]);
88 
89  rheology.push_back(
90  RheologicalModel::model_cast(physical_properties.fluids[f]));
91  this->establish_fields_required_by_model(*rheology[f]);
92  rheology[f]->set_dynamic_viscosity(density[f]->get_density_ref());
93 
94  tracer_diffusivity.push_back(
95  TracerDiffusivityModel::model_cast(physical_properties.fluids[f]));
96  establish_fields_required_by_model(*tracer_diffusivity[f]);
97 
98  thermal_expansion.push_back(
99  ThermalExpansionModel::model_cast(physical_properties.fluids[f]));
100  establish_fields_required_by_model(*thermal_expansion[f]);
101 
102  if (rheology.back()->is_non_newtonian_rheological_model())
103  non_newtonian_flow = true;
104  }
105 
106  // For each solid, append the physical properties
107  for (unsigned int s = 0; s < number_of_solids; ++s)
108  {
109  density.push_back(
110  DensityModel::model_cast(physical_properties.solids[s]));
111  establish_fields_required_by_model(*density[s]);
112 
113  // Store an indicator for the density to indicate if it is not constant
114  // This indicator is used elsewhere in the code to throw assertions
115  // if non-constant density is not implemented in a post-processing utility
116  if (!density.back()->is_constant_density_model())
117  constant_density = false;
118 
119  specific_heat.push_back(
120  SpecificHeatModel::model_cast(physical_properties.solids[s]));
121  establish_fields_required_by_model(*specific_heat[s]);
122 
123  thermal_conductivity.push_back(
124  ThermalConductivityModel::model_cast(physical_properties.solids[s]));
125  establish_fields_required_by_model(*thermal_conductivity[s]);
126 
127  rheology.push_back(
128  RheologicalModel::model_cast(physical_properties.solids[s]));
129  this->establish_fields_required_by_model(*rheology[s]);
130  rheology.back()->set_dynamic_viscosity(density.back()->get_density_ref());
131 
132  tracer_diffusivity.push_back(
133  TracerDiffusivityModel::model_cast(physical_properties.solids[s]));
134  establish_fields_required_by_model(*tracer_diffusivity[s]);
135 
136  thermal_expansion.push_back(
137  ThermalExpansionModel::model_cast(physical_properties.solids[s]));
138  establish_fields_required_by_model(*thermal_expansion[s]);
139  }
140 
141  // For each pair of interacting materials (fluid-fluid or fluid-solid), append
142  // physical properties
143  for (unsigned int i = 0; i < number_of_material_interactions; ++i)
144  {
145  surface_tension.push_back(SurfaceTensionModel::model_cast(
146  physical_properties.material_interactions[i]));
147  establish_fields_required_by_model(*surface_tension[i]);
148  if (!surface_tension.back()->is_constant_surface_tension_model())
149  constant_surface_tension = false;
150  mobility_cahn_hilliard.push_back(MobilityCahnHilliardModel::model_cast(
151  physical_properties.material_interactions[i]));
152  establish_fields_required_by_model(*mobility_cahn_hilliard[i]);
153  }
154 }

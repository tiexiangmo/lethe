26  /**
27  * @brief Class used to rescale physical properties for cases where the main dimensions
28  * of the problems (Length, Mass, Time and Temperature) are not expressed in
29  * SI, but are expressed in another set of units. For example, this can be
30  * used to carry out a simulation where the mesh is expressed in millimeters
31  * instead of meters. This class is currently only used to rescale the
32  * physical properties in the FEM-based physics (e.g. all of those that are
33  * solvers). Support for DEM, CFD-DEM and rescaling of boundary conditions is
34  * planned in the mid-term.
35  */
36  class Dimensionality
37  {
38  public:
39  /**
40  * @brief
41  * Default constructor that assumes that the units used in the simulation
42  * are consistent.
43  */
44  Dimensionality()
45  : length(1)
46  , mass(1)
47  , time(1)
48  , temperature(1)
49  {
50  define_all_scales();
51  }
52 
53  /**
54  * @brief
55  * Defines all the sub scales that are relevant to simulations. This is used
56  * to define scaling laws for kinematic viscosity or other quantities which
57  * appear multiple times in the simulation. This prevents the recalculation
58  * of these scales (and thus, human errors).
59  */
60  void
61  define_all_scales();

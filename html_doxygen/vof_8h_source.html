92  {
93  // for simplex meshes

98  face_quadrature =
99  std::make_shared<QGaussSimplex<dim - 1>>(fe->degree + 1);

101  else
102  {
103  // Usual case, for quad/hex meshes
104  fe = std::make_shared<FE_Q<dim>>(
105  simulation_parameters.fem_parameters.VOF_order);
106  mapping = std::make_shared<MappingQ<dim>>(
107  fe->degree, simulation_parameters.fem_parameters.qmapping_all);
108  fe_projected_phase_fraction_gradient =
109  std::make_shared<FESystem<dim>>(FE_Q<dim>(fe->degree), dim);
110  fe_curvature = std::make_shared<FE_Q<dim>>(fe->degree);
111  projected_phase_fraction_gradient_mapping =
112  std::make_shared<MappingQ<dim>>(
113  fe_projected_phase_fraction_gradient->degree,
114  simulation_parameters.fem_parameters.qmapping_all);

116  fe_curvature->degree,

119  face_quadrature = std::make_shared<QGauss<dim - 1>>(fe->degree + 1);

117 class ThermalExpansionPhaseChange : public ThermalExpansionModel
118 {
119 public:
123  ThermalExpansionPhaseChange(const Parameters::PhaseChange phase_change_params)
124  : p_phase_change_params(phase_change_params)
125  {
126  this->model_depends_on[field::temperature] = true;
127  }
128 
134  double
135  value(const std::map<field, double> &fields_value) override
136  {
137  double thermal_expansion;
138  // Thermal expansion of solid phase
139  if (fields_value.at(field::temperature) < p_phase_change_params.T_solidus)
140  thermal_expansion = p_phase_change_params.thermal_expansion_s;
141  // Thermal expansion coefficient of liquid phase
142  else if (fields_value.at(field::temperature) >
143  p_phase_change_params.T_liquidus)
144  thermal_expansion = p_phase_change_params.thermal_expansion_l;
145  // Mean value of the thermal expansion coefficients of the solid and liquid
146  // phases
147  else
148  {
149  const double l_frac =
150  calculate_liquid_fraction(fields_value.at(field::temperature),
151  p_phase_change_params);
152 
153  thermal_expansion =
154  p_phase_change_params.thermal_expansion_l * l_frac +
155  p_phase_change_params.thermal_expansion_s * (1. - l_frac);
156  }
157 
158  return thermal_expansion;
159  };
160 
166  void
167  vector_value(const std::map<field, std::vector<double>> &field_vectors,
168  std::vector<double> &property_vector) override
169  {
170  const std::vector<double> &T = field_vectors.at(field::temperature);
171  for (unsigned int i = 0; i < property_vector.size(); ++i)
172  {
173  // Thermal expansion of solid phase
174  if (T[i] < p_phase_change_params.T_solidus)
175  property_vector[i] = p_phase_change_params.thermal_expansion_s;
176  // Thermal expansion of liquid phase
177  else if (T[i] > p_phase_change_params.T_liquidus)
178  property_vector[i] = p_phase_change_params.thermal_expansion_l;
179  else
180  {
181  const double l_frac =
182  calculate_liquid_fraction(T[i], p_phase_change_params);
183 
184  property_vector[i] =
185  p_phase_change_params.thermal_expansion_l * l_frac +
186  p_phase_change_params.thermal_expansion_s * (1. - l_frac);
187  }
188  }
189  };

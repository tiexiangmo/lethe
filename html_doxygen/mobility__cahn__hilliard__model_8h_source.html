165  MobilityCahnHilliardModelQuartic(

193  /**
194  * @brief value Calculates the mobility_cahn_hilliard.
195  * @param fields_value Value of the various field on which the property may depend.
196  * @return value of the physical property calculated with the fields_value.
197  */
198  double
199  value(const std::map<field, double> &fields_value) override
200  {
201  const double &phase_order_cahn_hilliard =
202  fields_value.at(field::phase_order_cahn_hilliard);
203  return mobility_cahn_hilliard_constant *
204  (1 - phase_order_cahn_hilliard * phase_order_cahn_hilliard) *
205  (1 - phase_order_cahn_hilliard * phase_order_cahn_hilliard);
206  }
207 
208  /**
209  * @brief vector_value Calculates the vector of mobility_cahn_hilliard_constant.
210  * @param field_vectors Vectors of the fields on which the mobility_cahn_hilliard_constant may depend.
211  * @param property_vector Vectors of the mobility_cahn_hilliard_constant values
212  */
213  void
214  vector_value(const std::map<field, std::vector<double>> &field_vectors,
215  std::vector<double> &property_vector) override
216  {
217  const std::vector<double> &phase_order_cahn_hilliard =
218  field_vectors.at(field::phase_order_cahn_hilliard);
219  for (unsigned int i = 0; i < property_vector.size(); ++i)
220  property_vector[i] =
221  mobility_cahn_hilliard_constant *
222  (1 - phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]) *
223  (1 - phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]);
224  }
225 
226  /**
227  * @brief jacobian Calculates the jacobian (the partial derivative) of the density with respect to a field
228  * @param field_values Value of the various fields on which the property may depend.
229  * @param id Indicator of the field with respect to which the jacobian
230  * should be calculated.
231  * @return value of the partial derivative of the density with respect to the field.
232  */
233 
234  double
235  jacobian(const std::map<field, double> &fields_value, field /*id*/) override
236  {
237  const double &phase_order_cahn_hilliard =
238  fields_value.at(field::phase_order_cahn_hilliard);
239  return -4 * phase_order_cahn_hilliard * mobility_cahn_hilliard_constant *
240  (1 - phase_order_cahn_hilliard * phase_order_cahn_hilliard);
241  }
242 
243  /**
244  * @brief vector_jacobian Calculates the derivative of the density with respect to a field.
245  * @param field_vectors Vector for the values of the fields used to evaluate the property.
246  * @param id Identifier of the field with respect to which a derivative should be calculated.
247  * @param jacobian vector of the value of the derivative of the density with respect to the field id.
248  */
249 
250  void
251  vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
252  const field /*id*/,
253  std::vector<double> &jacobian_vector) override
254  {
255  const std::vector<double> &phase_order_cahn_hilliard =
256  field_vectors.at(field::phase_order_cahn_hilliard);
257  for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
258  jacobian_vector[i] =
259  -mobility_cahn_hilliard_constant * 4 * phase_order_cahn_hilliard[i] *
260  (1 - phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]);
261  }
262 
263 private:
264  const double mobility_cahn_hilliard_constant;
265  const MobilityModel model = quartic;
266 };

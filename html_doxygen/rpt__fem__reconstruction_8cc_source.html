244 {
245  TimerOutput::Scope t(computing_timer, "assemble_system");
246 
247  system_rhs = 0;
248  system_matrix = 0;
249 
250  const QGaussSimplex<dim> quadrature_formula(fe.degree + 1);
251 
252  FEValues<dim> fe_values(mapping,
253  fe,
254  quadrature_formula,
255  update_values | update_quadrature_points |
256  update_JxW_values);
257 
258  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
259 
260  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
261  Vector<double> cell_rhs(dofs_per_cell);
262 
263  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
264 
265  for (const auto &cell : dof_handler.active_cell_iterators())
266  {

271  cell_rhs = 0;
272  fe_values.reinit(cell);
273 
274  for (const unsigned int q_index :
275  fe_values.quadrature_point_indices())
276  {
277  Point<dim> q_point_position = fe_values.quadrature_point(q_index);
278  RadioParticle<dim> particle(q_point_position, 0);
279 
280  ParticleDetectorInteractions<dim> p_q_interaction(
281  particle, detectors[no_detector], parameters);
282 
283  double count = p_q_interaction.calculate_count();
284 
285  if (count < 0 || count != count)
286  this->pcout
287  << "Invalid value encountered in count calculation, this should be investigated furthermore : "
288  << count <<
289 
290  std::endl;
291  else
292  {
293  for (const unsigned int i : fe_values.dof_indices())
294  {
295  for (const unsigned int j : fe_values.dof_indices())
296  {
297  cell_matrix(i, j) +=
298  (fe_values.shape_value(i, q_index) * // phi_i(x_q)
299  fe_values.shape_value(j, q_index) * // phi_j(x_q)
300  fe_values.JxW(q_index)); // dx
301  }
302 
303 
304  cell_rhs(i) +=
305  (count * // f(x)
306  fe_values.shape_value(i, q_index) * // phi_i(x_q)
307  fe_values.JxW(q_index)); // dx
308  }
309  }
310  }
311  cell->get_dof_indices(local_dof_indices);
312  constraints.distribute_local_to_global(cell_matrix,
313  cell_rhs,
314  local_dof_indices,
315  system_matrix,
316  system_rhs);
317  }
318  }
319  system_matrix.compress(VectorOperation::add);
320  system_rhs.compress(VectorOperation::add);

494 template <int dim>
495 Vector<double>
496 assemble_matrix_and_rhs(
497  std::vector<std::vector<double>> &vertex_count,
498  std::vector<double> &experimental_count,
499  Parameters::RPTFEMReconstructionParameters::FEMCostFunction
500  &cost_function_type)
501 {
502  Vector<double> reference_location;
503  unsigned int detector_size = vertex_count.size();
504  double sigma = 0;
505  LAPACKFullMatrix<double> sys_matrix;
506  Vector<double> sys_rhs;
507 
508  sys_matrix.reinit(dim);
509  sys_rhs.reinit(dim);
510 
511  if (cost_function_type ==
512  Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute)
513  {
514  // Assembling sys_matrix (Jacobian)
515  for (unsigned int i = 0; i < dim; ++i)
516  {
517  for (unsigned int j = 0; j < dim; ++j)
518  {
519  for (unsigned int d = 0; d < detector_size; ++d)
520  {
521  sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
522  (-vertex_count[d][0] + vertex_count[d][i + 1]);
523  }
524  sys_matrix.set(i, j, sigma);
525  sigma = 0;
526  }
527  }
528 
529  // Assembling sys_rhs
530  for (unsigned int i = 0; i < dim; ++i)
531  {
532  for (unsigned int d = 0; d < detector_size; ++d)
533  {
534  sigma += (vertex_count[d][0] - experimental_count[d]) *
535  (-vertex_count[d][0] + vertex_count[d][i + 1]);
536  }
537  sys_rhs[i] = -sigma;
538  sigma = 0;
539  }
540  }
541  else if (cost_function_type == Parameters::RPTFEMReconstructionParameters::
542  FEMCostFunction::relative)
543  {
544  std::vector<double> denom(detector_size);
545 
546  for (unsigned int d = 0; d < experimental_count.size(); ++d)
547  denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
548 
549  // Assembling sys_matrix
550  for (unsigned int i = 0; i < dim; ++i)
551  {
552  for (unsigned int j = 0; j < dim; ++j)
553  {
554  for (unsigned int d = 0; d < detector_size; ++d)
555  {
556  sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
557  (-vertex_count[d][0] + vertex_count[d][i + 1]) *
558  denom[d];
559  }
560  sys_matrix.set(i, j, sigma);
561  sigma = 0;
562  }
563  }
564 
565  // Assembling sys_rhs
566  for (unsigned int i = 0; i < dim; ++i)
567  {
568  for (unsigned int d = 0; d < detector_size; ++d)
569  {
570  sigma += (vertex_count[d][0] - experimental_count[d]) *
571  (-vertex_count[d][0] + vertex_count[d][i + 1]) *
572  denom[d];
573  }
574  sys_rhs[i] = -sigma;
575  sigma = 0;
576  }
577  }
578 
579  // Setup and solve linear system
580  sys_matrix.set_property(LAPACKSupport::general);
581  sys_matrix.compute_lu_factorization();
582  sys_matrix.solve(sys_rhs);
583 
584  reference_location = sys_rhs;
585  return reference_location;
586 }

589 template <int dim>
590 double
591 RPTFEMReconstruction<dim>::calculate_reference_location_error(
592  Vector<double> &reference_location,
593  const double &last_constraint)
594 {
595  std::vector<double> err_coordinates(dim + 1, 0);
596  double norm_error_coordinates = 0;
597 
598  // Check if the location is a valid one
599  for (unsigned int i = 0; i < dim; ++i)
600  {
601  // e_1 = \xi - 1; e_2 = \eta - 1; e_3 = \zeta - 1;
602  if (reference_location[i] > 1)
603  err_coordinates[i] = (reference_location[i] - 1);
604 
605  // e_1 = - \xi; e_2 = - \eta; e_3 = - \zeta;
606  if (reference_location[i] < 0)
607  err_coordinates[i] = std::abs(reference_location[i]);
608  }
609 
610  // Check the last constraint
611  if (last_constraint < 0)
612  {
613  // e_4 = |(1 - \xi - \eta - \zeta)|
614  err_coordinates[3] = std::abs(last_constraint);
615  }
616  else if (last_constraint > 1)
617  {
618  // e_4 = (1 - \xi - \eta - \zeta) - 1
619  err_coordinates[3] = last_constraint - 1;
620  }
621 
622  // Calculate the norm of the error coordinate vector
623  for (const double &error : err_coordinates)
624  {
625  norm_error_coordinates += error * error;
626  }
627 
628  return norm_error_coordinates;
629 }

632 template <int dim>
633 double
634 RPTFEMReconstruction<dim>::calculate_cost(
635  const TriaActiveIterator<DoFCellAccessor<dim, dim, false>> &cell,
636  Vector<double> &reference_location,
637  const double &last_constraint,
638  std::vector<double> &experimental_count)
639 {
640  double cost = 0;
641  double count = 0;
642 
643  if (fem_reconstruction_parameters.fem_cost_function ==
644  Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute)
645  {
646  // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
647  // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2
648  for (unsigned int d = 0; d < n_detector; ++d)
649  {
650  count +=
651  (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
652 
653  for (unsigned int i = 0; i < dim; ++i)
654  {
655  count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
656  reference_location[i]);
657  }
658  count -= experimental_count[d];
659  cost += count * count;
660  count = 0;
661  }
662  }
663  else if (fem_reconstruction_parameters.fem_cost_function ==
664  Parameters::RPTFEMReconstructionParameters::FEMCostFunction::
665  relative)
666  {
667  std::vector<double> denom(n_detector);
668 
669  // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
670  // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2 / (
671  // C_{exp,d}^2)
672  for (unsigned int d = 0; d < n_detector; ++d)
673  {
674  count +=
675  (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
676 
677  denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
678 
679  for (unsigned int i = 0; i < dim; ++i)
680  {
681  count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
682  reference_location[i]);
683  }
684 
685  count -= experimental_count[d];
686  cost += count * count * denom[d];
687  count = 0;
688  }
689  }
690 
691  return cost;
692 }

694 template <int dim>
695 void
696 RPTFEMReconstruction<dim>::search_position_in_reference_space(
697  std::vector<std::vector<double>> &count_from_all_detectors,
698  std::vector<double> &experimental_count,
699  const typename DoFHandler<dim>::active_cell_iterator &cell,
700  const double &tol_reference_location,
701  double &max_cost,
702  Vector<double> &reference_location,
703  bool &position_found,
704  Point<dim> &real_location)
705 {
706  double last_constraint_reference_location;
707  double norm_error_coordinates;
708  double calculated_cost;
709 
710 
711  // Solve linear system to find the location in reference coordinates
712  reference_location = assemble_matrix_and_rhs<dim>(
713  count_from_all_detectors,
714  experimental_count,
715  fem_reconstruction_parameters.fem_cost_function);
716 
717  // 4th constraint on the location of the particle in reference coordinates
718  last_constraint_reference_location =
719  1 - reference_location[0] - reference_location[1] - reference_location[2];
720 
721  // Evaluate the error of the reference position (Is it outside the
722  // reference tetrahedron ?)
723  norm_error_coordinates =
724  calculate_reference_location_error(reference_location,
725  last_constraint_reference_location);
726 
727  // Extrapolation limit
728  if (norm_error_coordinates < tol_reference_location)
729  {
730  // Calculate cost with the selected cost function
731  calculated_cost = calculate_cost(cell,
732  reference_location,
733  last_constraint_reference_location,
734  experimental_count);
735  // Check if the new cost is lower than the previously stored
736  // real_location's
737  if (calculated_cost < max_cost)
738  {
739  max_cost = calculated_cost;
740  position_found = true;
741  previous_position_cell = cell;
742 
743  // Evaluate the real location of the particle
744  for (unsigned int v = 0; v < cell->n_vertices(); ++v)
745  {
746  if (v == 0)
747  real_location = cell->vertex(v);
748  else
749  real_location += reference_location[v - 1] *
750  (cell->vertex(v) - cell->vertex(0));
751  }
752  }
753  }
754 }

795 bool
796 RPTFEMReconstruction<dim>::find_position_local_search(
797  std::vector<double> &experimental_counts,
798  const double tol_reference_location)
799 {
800  double max_cost = DBL_MAX;
801  bool position_found = false;
802  Point<dim> real_location;
803  Vector<double> reference_location;
804  std::vector<std::vector<double>> count_from_all_detectors(
805  n_detector, std::vector<double>(4));
806  const typename DoFHandler<dim>::active_cell_iterator cell =
807  previous_position_cell;
808 
809  // Find cells adjacent to the cell were found the previous position
810  std::vector<typename DoFHandler<dim>::active_cell_iterator> adjacent_cells;
811  std::set<typename DoFHandler<dim>::active_cell_iterator> all_adjacent_cells;
812 
813  // Add previous solution's cell
814  all_adjacent_cells.insert(cell);
815 
816  // Search for adjacent cells and stock them in the set container
817  for (unsigned int v = 0; v < cell->n_vertices(); ++v)
818  {
819  auto v_index = cell->vertex_index(v);
820  adjacent_cells =
821  GridTools::find_cells_adjacent_to_vertex(dof_handler, v_index);
822 
823  for (const auto &adjacent_cell : adjacent_cells)
824  all_adjacent_cells.insert(adjacent_cell);
825  }
826 
827  // Add adjacent cells according to the level of proximity
828  if (fem_reconstruction_parameters.search_proximity_level > 1)
829  {
830  std::set<typename DoFHandler<dim>::active_cell_iterator>
831  previous_adjacent_cells;
832  std::set<typename DoFHandler<dim>::active_cell_iterator>
833  previous_main_cells;
834 
835  previous_main_cells.insert(cell);
836 
837  for (unsigned int proximity_level = 2;
838  proximity_level <
839  fem_reconstruction_parameters.search_proximity_level + 1;
840  ++proximity_level)
841  {
842  previous_adjacent_cells = all_adjacent_cells;
843 
844  for (const auto &current_main_cell : previous_adjacent_cells)
845  {
846  if (*previous_main_cells.find(current_main_cell) !=
847  current_main_cell)
848  {
849  for (unsigned int v = 0; v < current_main_cell->n_vertices();
850  ++v)
851  {
852  auto v_index = current_main_cell->vertex_index(v);
853  adjacent_cells =
854  GridTools::find_cells_adjacent_to_vertex(dof_handler,
855  v_index);
856 
857  for (const auto &adjacent_cell : adjacent_cells)
858  all_adjacent_cells.insert(adjacent_cell);
859  }
860  }
861  previous_main_cells = previous_adjacent_cells;
862  }
863  }
864  }
865 
866  // Loop over adjacent cells, loop over detectors and get nodal count values
867  // for each vertex of the cell
868  for (const auto &adjacent_cell : all_adjacent_cells)
869  {
870  for (unsigned int d = 0; d < n_detector; ++d)
871  {
872  for (unsigned int v = 0; v < adjacent_cell->n_vertices(); ++v)
873  {
874  auto dof_index = adjacent_cell->vertex_dof_index(v, 0);
875  count_from_all_detectors[d][v] = nodal_counts[d][dof_index];
876  }
877  }
878 
879  search_position_in_reference_space(count_from_all_detectors,
880  experimental_counts,
881  adjacent_cell,
882  tol_reference_location,
883  max_cost,
884  reference_location,
885  position_found,
886  real_location);
887  }
888 
889  if (position_found)
890  {
891  found_positions.push_back(real_location);
892  return position_found;
893  }
894  else
895  {
896  position_found = find_position_global_search(experimental_counts,
897  tol_reference_location);
898  return position_found;
899  }

902 template <int dim>
903 void
904 RPTFEMReconstruction<dim>::trajectory()
905 {
906  // Tolerance or the extrapolation limit in the reference space for a found
907  // position
908  double tol_reference_location =
909  fem_reconstruction_parameters.extrapolation_tolerance;
910 
911  if (fem_reconstruction_parameters.mesh_type ==
912  Parameters::RPTFEMReconstructionParameters::FEMMeshType::dealii)
913  {
914  if (fem_reconstruction_parameters.extrapolation_tolerance == 0.005)
915  {
916  const unsigned int power =
917  pow(2, fem_reconstruction_parameters.mesh_refinement);
918  const unsigned int n_cell_z =
919  2 * fem_reconstruction_parameters.z_subdivisions * power;
920  tol_reference_location = parameters.reactor_height / n_cell_z * 1.15;
921  }
922  }
923 
924  // Read and store all experimental counts
925  std::vector<std::vector<double>> all_experimental_counts;
926  {
927  TimerOutput::Scope t(computing_timer, "input_experimental_counts_file");
928  all_experimental_counts = read_detectors_counts<dim>(
929  fem_reconstruction_parameters.experimental_counts_file, n_detector);
930  }
931 
932  {
933  TimerOutput::Scope t(computing_timer, "find_particle_positions");
934 
935  if (fem_reconstruction_parameters.search_type ==
936  Parameters::RPTFEMReconstructionParameters::FEMSearchType::local)
937  {
938  // It is set to "false" to force the global search on the first search
939  bool adjacent_cell_search = false;
940 
941  // Find the position of the particle with the experimental counts by
942  // prioritizing local search
943  for (std::vector<double> &experimental_counts : all_experimental_counts)
944  {
945  if (adjacent_cell_search)
946  adjacent_cell_search =
947  find_position_local_search(experimental_counts,
948  tol_reference_location);

950  adjacent_cell_search =
951  find_position_global_search(experimental_counts,
952  tol_reference_location);
953  }
954  }
955  else
956  {
957  // Find the position of the particle with the experimental counts by
958  // global search only
959  for (std::vector<double> &experimental_counts : all_experimental_counts)
960  {
961  find_position_global_search(experimental_counts,
962  tol_reference_location);
963  }
964  }
965  }

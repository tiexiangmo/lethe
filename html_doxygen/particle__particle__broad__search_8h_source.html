132  /**
133  * @brief Stores the candidate particle-particle collision pairs with a given
134  * particle iterator. particle_begin iterator is useful to skip storage of the
135  * first particle in main cell (particle_begin will be the iterator after the
136  * particles_to_evaluate.begin() in that case). When particle_begin is
137  * particles_to_evaluate.begin(), it stores all the particle id in
138  * contact_pair_candidates_container.
139  *
140  * @param particle_begin The particle iterator to start storing particle id
141  * @param particles_to_evaluate The particle range to evaluate with the
142  * particle iterator prior storing ids
143  * @param contact_pair_candidates_container A vector which will contain all
144  * the particle pairs which are collision candidates
145  */
146  inline void
147  store_candidates(
148  const typename Particles::ParticleHandler<
149  dim>::particle_iterator_range::iterator &particle_begin,
150  const typename Particles::ParticleHandler<dim>::particle_iterator_range
151  &particles_to_evaluate,
152  std::vector<types::particle_index> &contact_pair_candidates_container)
153  {
154  // Create a arbitrary temporary empty container
155  if (contact_pair_candidates_container.empty())
156  {
157  contact_pair_candidates_container.reserve(40);
158  }
159 
160  // Store particle ids from the selected particle iterator
161  for (auto particle_iterator = particle_begin;
162  particle_iterator != particles_to_evaluate.end();
163  ++particle_iterator)
164  {
165  contact_pair_candidates_container.emplace_back(
166  particle_iterator->get_id());
167  }
168  }

30  QGauss<dim - 1> face_quadrature_formula(1);
31  FEFaceValues<dim> fe_face_values(fe,
32  face_quadrature_formula,
33  update_values | update_quadrature_points |
34  update_normal_vectors);
35  Tensor<1, dim> normal_vector;
36  Point<dim> quad_point;
37 
38  // Store information of the cell
39  boundaries_information.cell = cell;
40  boundaries_information.boundary_id = cell->face(face_id)->boundary_id();
41 
42  // Store information of the periodic cell
43  boundaries_information.periodic_cell = cell->periodic_neighbor(face_id);
44 
45  // Find the normal vector of the boundary face and point
46  fe_face_values.reinit(cell, face_id);
47  boundaries_information.normal_vector = fe_face_values.normal_vector(0);
48  boundaries_information.point_on_face = fe_face_values.quadrature_point(0);
49 
50  // Find the normal vector of the periodic boundary face and point with the
51  // reinit FE face values
52  fe_face_values.reinit(boundaries_information.periodic_cell,
53  cell->periodic_neighbor_face_no(face_id));
54  normal_vector = fe_face_values.normal_vector(0);
55  quad_point = fe_face_values.quadrature_point(0);
56  boundaries_information.periodic_normal_vector = normal_vector;
57  boundaries_information.point_on_periodic_face = quad_point;

121 template <int dim>
122 void
123 PeriodicBoundariesManipulator<dim>::check_and_move_particles(
124  const periodic_boundaries_cells_info_struct<dim> &boundaries_cells_content,
125  typename Particles::ParticleHandler<dim>::particle_iterator_range
126  &particles_in_cell,
127  const bool particles_in_pb0_cell)
128 {
129  for (auto particle = particles_in_cell.begin();
130  particle != particles_in_cell.end();
131  ++particle)
132  {
133  // Get the current particle location
134  Point<dim> particle_position = particle->get_location();
135 
136  // Initialize points and normal vector related of the cell that contains
137  // the particle
138  Point<dim> point_on_face, point_on_periodic_face;
139  Tensor<1, dim> normal_vector, distance_between_faces;
140  if (particles_in_pb0_cell)
141  {
142  point_on_face = boundaries_cells_content.point_on_face;
143  normal_vector = boundaries_cells_content.normal_vector;
144  distance_between_faces = constant_periodic_offset;
145  }
146  else
147  {
148  point_on_face = boundaries_cells_content.point_on_periodic_face;
149  normal_vector = boundaries_cells_content.periodic_normal_vector;
150  distance_between_faces = -constant_periodic_offset;
151  }
152 
153  // Calculate distance between particle position and the face of
154  // boundary cell d = nâ€¢(pt_particle - pt_face)
155  double distance_with_face =
156  scalar_product(particle_position - point_on_face, normal_vector);
157 
158  // If distance >= 0, particle is outside of cell (or on face).
159  // If so, particle location is modified to get moved into the periodic
160  // cell.
161  if (distance_with_face >= 0.0)
162  {
163  // Move particle outside the current cell to the periodic cell.
164  particle_position += distance_between_faces;
165  particle->set_location(particle_position);
166  }
167  }

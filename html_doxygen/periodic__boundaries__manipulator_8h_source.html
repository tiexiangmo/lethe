35 /**
36  * This class corresponds to a manipulator of the particles crossing periodic
37  * cells in DEM. It maps cell information of the pairs of periodic cells
38  * and manipulate the particles location when they cross periodic boundaries,
39  * i.e. when a particle cross a periodic boundary (0 or 1) its location is
40  * modified with the offset between faces of periodic cells on periodic
41  * boundaries.
42  *
43  * @note Currently the code only supports one pair of periodic boundaries and
44  * those boundaries must be parallel and aligned with an axis of the domain
45  * (e.g., x axis).
46  */
47 
48 template <int dim>
49 class PeriodicBoundariesManipulator
50 {
51 public:
52  PeriodicBoundariesManipulator();
53 
54  /**
55  * @brief Sets the periodic boundaries parameters
56  *
57  * @param periodic_boundary_id_0 Id of the first periodic boundary
58  * @param periodic_boundary_id_1 Id of the second periodic boundary
59  * @param periodic_direction Perpendicular axis of PB
60  */
61  void
62  set_periodic_boundaries_information(
63  const types::boundary_id periodic_boundary_id_0,
64  const types::boundary_id periodic_boundary_id_1,
65  const unsigned int periodic_direction)
66  {
67  periodic_boundary_0 = periodic_boundary_id_0;
68  periodic_boundary_1 = periodic_boundary_id_1;
69  direction = periodic_direction;
70  }

80  void

96  void

102  /**

107  inline Tensor<1, dim>

109  {
110  return constant_periodic_offset;

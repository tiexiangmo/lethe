282  /**
283  * @brief Parse the information for a boundary condition
284  *
285  * @param prm A parameter handler which is currently used to parse the simulation information
286  *
287  * @param i_bc The boundary condition number (and not necessarily it's id).
288  */
289  template <int dim>
290  void
291  NSBoundaryConditions<dim>::parse_boundary(ParameterHandler &prm,
292  const unsigned int i_bc)
293  {
294  const std::string op = prm.get("type");
295  if (op == "none")
296  this->type[i_bc] = BoundaryType::none;
297  if (op == "noslip")
298  this->type[i_bc] = BoundaryType::noslip;
299  if (op == "slip")
300  this->type[i_bc] = BoundaryType::slip;
301  if (op == "function" || op == "function weak" || op == "partial slip")
302  {
303  if (op == "function")
304  this->type[i_bc] = BoundaryType::function;
305  else if (op == "partial slip")
306  this->type[i_bc] = BoundaryType::partial_slip;
307  else
308  this->type[i_bc] = BoundaryType::function_weak;
309  prm.enter_subsection("u");
310  bcFunctions[i_bc].u.parse_parameters(prm);
311  prm.leave_subsection();
312 
313  prm.enter_subsection("v");
314  bcFunctions[i_bc].v.parse_parameters(prm);
315  prm.leave_subsection();
316 
317  prm.enter_subsection("w");
318  bcFunctions[i_bc].w.parse_parameters(prm);
319  prm.leave_subsection();
320 
321  prm.enter_subsection("center of rotation");
322  bcFunctions[i_bc].center_of_rotation[0] = prm.get_double("x");
323  bcFunctions[i_bc].center_of_rotation[1] = prm.get_double("y");
324 
325  if (dim == 3)
326  bcFunctions[i_bc].center_of_rotation[2] = prm.get_double("z");
327  prm.leave_subsection();
328  }
329  if (op == "pressure")
330  {
331  this->type[i_bc] = BoundaryType::pressure;
332  prm.enter_subsection("p");
333  bcPressureFunction[i_bc].p.parse_parameters(prm);
334  prm.leave_subsection();
335 
336  prm.enter_subsection("center of rotation");
337  bcPressureFunction[i_bc].center_of_rotation[0] = prm.get_double("x");
338  bcPressureFunction[i_bc].center_of_rotation[1] = prm.get_double("y");
339  if (dim == 3)
340  bcPressureFunction[i_bc].center_of_rotation[2] = prm.get_double("z");
341  prm.leave_subsection();
342  }
343  if (op == "periodic")
344  {
345  this->type[i_bc] = BoundaryType::periodic;
346  this->periodic_id[i_bc] = prm.get_integer("periodic_id");
347  this->periodic_direction[i_bc] = prm.get_integer("periodic_direction");
348  }
349  if (op == "outlet")
350  {
351  this->type[i_bc] = BoundaryType::outlet;
352  }
353 
354  this->id[i_bc] = prm.get_integer("id");
355  this->beta[i_bc] = prm.get_double("beta");
356  this->boundary_layer_thickness[i_bc] =
357  prm.get_double("boundary layer thickness");
358  }

16  *
17  * Implementation of tracer as an auxiliary physics.
18  * Equation solved:
19  * dT/dt + u * gradT = D * div(grad T) + f
20  * with T the tracer function, D the diffusivity and f the forcing
21  *
22  * Author: Bruno Blais, Polytechnique Montreal, 2020-
23  */

55 public:
56  Tracer(MultiphysicsInterface<dim> *multiphysics_interface,
57  const SimulationParameters<dim> &p_simulation_parameters,
58  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
59  p_triangulation,
60  std::shared_ptr<SimulationControl> p_simulation_control)
61  : AuxiliaryPhysics<dim, TrilinosWrappers::MPI::Vector>(
62  p_simulation_parameters.non_linear_solver.at(PhysicsID::tracer))
63  , multiphysics(multiphysics_interface)
64  , computing_timer(p_triangulation->get_communicator(),
65  this->pcout,
66  TimerOutput::summary,
67  TimerOutput::wall_times)
68  , simulation_parameters(p_simulation_parameters)
69  , triangulation(p_triangulation)
70  , simulation_control(p_simulation_control)
71  , dof_handler(*triangulation)
72  {
73  if (simulation_parameters.mesh.simplex)
74  {
75  // for simplex meshes
76  fe = std::make_shared<FE_SimplexP<dim>>(
77  simulation_parameters.fem_parameters.tracer_order);
78  mapping = std::make_shared<MappingFE<dim>>(*fe);
79  cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
80  }
81  else
82  {
83  // Usual case, for quad/hex meshes
84  fe = std::make_shared<FE_Q<dim>>(
85  simulation_parameters.fem_parameters.tracer_order);
86  mapping = std::make_shared<MappingQ<dim>>(
87  fe->degree, simulation_parameters.fem_parameters.qmapping_all);
88  cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
89  }
90 
91  // Allocate solution transfer
92  solution_transfer =
93  std::make_shared<parallel::distributed::
94  SolutionTransfer<dim, TrilinosWrappers::MPI::Vector>>(
95  dof_handler);
96 
97  // Set size of previous solutions using BDF schemes information
98  previous_solutions.resize(maximum_number_of_previous_solutions());
99 
100  // Prepare previous solutions transfer
101  previous_solutions_transfer.reserve(previous_solutions.size());
102  for (unsigned int i = 0; i < previous_solutions.size(); ++i)
103  {
104  previous_solutions_transfer.emplace_back(
105  parallel::distributed::
106  SolutionTransfer<dim, TrilinosWrappers::MPI::Vector>(
107  this->dof_handler));
108  }
109 
110  // Change the behavior of the timer for situations when you don't want
111  // outputs
112  if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
113  this->computing_timer.disable_output();
114  }

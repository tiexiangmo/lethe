59 /** @brief Class that manages the physical properties
60  * model which are required to calculate the various physical properties
61  * This centralizes the place where the models are created.
62  * The class can be constructed empty and initialized from parameters
63  * or it can be constructed directly with a Parameters section.
64  * The architecture of the PhysicalPropertiesManager is made to ensure that it
65  * can store as many physical properties models as required. It first stores
66  * the physical properties of the fluids, which are always assumed to lie in
67  * material_id=0, and then stores the physical properties of the solid, which
68  * may be in cells for which material_id>0.
69  *
70  */

286  }
287  else
288  throw(std::runtime_error(

299  /** @brief Calculates the global id of the physical property. By default. Lethe stores all
300  * the properties of the fluids, then the properties of the solid. Fluids
301  * need to have a material id of 0, then solids are consequential.
302  *
303  * @param fluid_id the id of the fluid (0 for single phase simulations, 0 or 1 for VOF simulations)
304  *
305  * @param material_id the material id of the cell. Cells with material_id=0 are fluid cells (0 or 1) and cells with material_id>0 are solid cells
306  */
307  unsigned int
308  calculate_global_id(const unsigned int fluid_id,
309  const unsigned int material_id) const
310  {
311  if (material_id < 1)
312  return fluid_id;
313  else
314  {
315  return number_of_fluids + material_id - 1;
316  }
317  }

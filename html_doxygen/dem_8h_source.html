78  /**
79  * The cell_weight() function indicates to the triangulation how much
80  * computational work is expected to happen on this cell, and consequently
81  * how the domain needs to be partitioned so that every MPI rank receives a
82  * roughly equal amount of work (potentially not an equal number of cells).
83  * While the function is called from the outside, it is connected to the
84  * corresponding signal from inside this class, therefore it can be private.
85  * This function is the key component that allow us to dynamically balance the
86  * computational load. The function attributes a weight to
87  * every cell that represents the computational work on this cell. Here the
88  * majority of work is expected to happen on the particles, therefore the
89  * return value of this function (representing "work for this cell") is
90  * calculated based on the number of particles in the current cell.
91  * The function is connected to the cell_weight() signal inside the
92  * triangulation, and will be called once per cell, whenever the triangulation
93  * repartitions the domain between ranks (the connection is created inside the
94  * particles_generation() function of this class).
95  */
96 # if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
97  unsigned int
98  cell_weight(
99  const typename parallel::distributed::Triangulation<dim>::cell_iterator
100  &cell,
101  const typename parallel::distributed::Triangulation<dim>::CellStatus status)
102  const;
103 # else
104  unsigned int
105  cell_weight(
106  const typename parallel::distributed::Triangulation<dim>::cell_iterator
107  &cell,
108  const CellStatus status) const;
109 # endif
110 
111  /**
112  * Similar to the cell_weight() function, this function is used when the cell
113  * weight is adapted to the mobility status. For instance, if the
114  * cell is inactive, its computational load will be significantly lower than
115  * if it is a mobile cell since there is no force calculation and no velocity
116  * integration for the particles that lie within it. The weight of the cells
117  * must thus be adapted to the status of the cell.
118  *
119  * cell load = cell weight + load balancing factor * n particles * particle
120  * weight
121  *
122  * @param cell The cell for which the load is calculated
123  *
124  * @param status The status of the cell used to inform functions in derived
125  * classes how the cell with the given cell iterator is going to change
126  *
127  * @param mobility_status The mobility status of the cell
128  */
129 
130 # if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
131  unsigned int
132  cell_weight_with_mobility_status(
133  const typename parallel::distributed::Triangulation<dim>::cell_iterator
134  &cell,

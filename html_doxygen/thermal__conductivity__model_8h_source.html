192 /**
193  * @brief Implements a phase-dependant thermal conductivity
194  */
195 class ThermalConductivityPhaseChange : public ThermalConductivityModel
196 {
197 public:
198  /**
199  * @brief Default constructor
200  */
201  ThermalConductivityPhaseChange(
202  const Parameters::PhaseChange phase_change_params)
203  : p_phase_change_params(phase_change_params)
204  , thermal_conductivity_s(phase_change_params.thermal_conductivity_s)
205  , thermal_conductivity_l(phase_change_params.thermal_conductivity_l)
206  , T_solidus(phase_change_params.T_solidus)
207  , T_liquidus(phase_change_params.T_liquidus)
208  {
209  this->model_depends_on[field::temperature] = true;
210  }
211 
212  /**
213  * @brief value Calculates the value of the thermal conductivity
214  * @param fields_value Value of the various field on which the thermal conductivity depends.
215  * @return value of the thermal conductivity calculated with the fields_value.
216  */
217  double
218  value(const std::map<field, double> &fields_value) override
219  {
220  double thermal_conductivity;
221  // Thermal conductivity of solid phase
222  if (fields_value.at(field::temperature) < T_solidus)
223  thermal_conductivity = thermal_conductivity_s;
224  // Thermal conductivity of liquid phase
225  else if (fields_value.at(field::temperature) > T_liquidus)
226  thermal_conductivity = thermal_conductivity_l;
227  // Mean value of the conductivities of the solid and liquid phases
228  else
229  {
230  const double l_frac =
231  calculate_liquid_fraction(fields_value.at(field::temperature),
232  p_phase_change_params);
233 
234  thermal_conductivity = thermal_conductivity_l * l_frac +
235  thermal_conductivity_s * (1. - l_frac);
236  }
237 
238  return thermal_conductivity;
239  };
240 
241  /**
242  * @brief vector_value Calculates the vector value of thermal conductivities
243  * @param field_vectors Vector of properties on which the thermal conductivities depend
244  * @param property_vector Values of the thermal conductivities
245  */
246  void
247  vector_value(const std::map<field, std::vector<double>> &field_vectors,
248  std::vector<double> &property_vector) override
249  {
250  const std::vector<double> &T = field_vectors.at(field::temperature);
251  for (unsigned int i = 0; i < property_vector.size(); ++i)
252  {
253  // Thermal conductivity of solid phase
254  if (T[i] < T_solidus)
255  property_vector[i] = thermal_conductivity_s;
256  // Thermal conductivity of liquid phase
257  else if (T[i] > T_liquidus)
258  property_vector[i] = thermal_conductivity_l;
259  else
260  {
261  const double l_frac =
262  calculate_liquid_fraction(T[i], p_phase_change_params);
263 
264  property_vector[i] = thermal_conductivity_l * l_frac +
265  thermal_conductivity_s * (1. - l_frac);
266  }
267  }
268  };

272  * @param field_values Value of the various fields on which the property may depend.

300  };

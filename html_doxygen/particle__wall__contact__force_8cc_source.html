22 // Updates the contact information (contact_info) based on the new information
23 // of particles pair in the current time step
24 template <int dim>
25 void
26 ParticleWallContactForce<dim>::update_contact_information(
27  particle_wall_contact_info<dim> &contact_info,
28  const Point<3> &particle_position,
29  const ArrayView<const double> &particle_properties,
30  const double dt)
31 {
32  // i is the particle, j is the wall.
33  // we need to put a minus sign infront of the normal_vector to respect the
34  // convention (i -> j)
35  auto normal_vector = -contact_info.normal_vector;
36  const unsigned int boundary_id = contact_info.boundary_id;
37 
38  // Using velocity and angular velocity of particle as
39  // local vectors
40  Tensor<1, 3> particle_velocity;
41  particle_velocity[0] = particle_properties[DEM::PropertiesIndex::v_x];
42  particle_velocity[1] = particle_properties[DEM::PropertiesIndex::v_y];
43  particle_velocity[2] = particle_properties[DEM::PropertiesIndex::v_z];
44 
45  Tensor<1, 3> particle_angular_velocity;
46  particle_angular_velocity[0] =
47  particle_properties[DEM::PropertiesIndex::omega_x];
48  particle_angular_velocity[1] =
49  particle_properties[DEM::PropertiesIndex::omega_y];
50  particle_angular_velocity[2] =
51  particle_properties[DEM::PropertiesIndex::omega_z];
52 
53  // Calculate approximation of the contact point using the normal vector
54  Point<3> contact_point =
55  particle_position +
56  0.5 * particle_properties[DEM::PropertiesIndex::dp] * normal_vector;
57 
58  // Get vector pointing from the contact point to the origin of the rotation
59  // axis
60  Tensor<1, 3> vector_to_rotating_axis =
61  contact_point - this->point_on_rotation_vector[boundary_id];
62 
63  // Remove the rotating axis component of that vector
64  vector_to_rotating_axis =
65  vector_to_rotating_axis -
66  (vector_to_rotating_axis * this->boundary_rotational_vector[boundary_id]) *
67  this->boundary_rotational_vector[boundary_id];
68 
69  // Tensor<1,3> vector_to_rotation_axis = this->boundary_rotational_speed_map
70 
71  // Defining relative contact velocity using the convention
72  // v_ij = v_j - v_i
73  Tensor<1, 3> contact_relative_velocity =
74  this->boundary_translational_velocity_map[boundary_id] - particle_velocity +
75  cross_product_3d((-0.5 * particle_properties[DEM::PropertiesIndex::dp] *
76  particle_angular_velocity),
77  normal_vector) +
78  cross_product_3d(this->boundary_rotational_speed_map[boundary_id] *
79  this->boundary_rotational_vector[boundary_id],
80  vector_to_rotating_axis);
81 
82  // Calculation of normal relative velocity
83  double normal_relative_velocity_value =
84  contact_relative_velocity * normal_vector;
85  Tensor<1, 3> normal_relative_velocity =
86  normal_relative_velocity_value * normal_vector;
87 
88  // Calculation of tangential relative velocity
89  Tensor<1, 3> tangential_relative_velocity =
90  contact_relative_velocity - normal_relative_velocity;
91 
92  // Calculation of new tangential_overlap, since this value is
93  // history-dependent it needs the value at previous time-step
94  // This variable is the main reason that we have iteration over
95  // two different vectors (pairs_in_contact and
96  // contact_pair_candidates): tangential_overlap of the particles
97  // which were already in contact (pairs_in_contact) needs to be
98  // modified using its history, while the tangential_overlaps of
99  // new particles are equal to zero
100  Tensor<1, 3> modified_tangential_overlap =
101  contact_info.tangential_overlap + tangential_relative_velocity * dt;
102 
103  // Updating the contact_info container based on the new calculated values
104  contact_info.normal_relative_velocity = normal_relative_velocity_value;
105  contact_info.tangential_overlap = modified_tangential_overlap;
106  contact_info.tangential_relative_velocity = tangential_relative_velocity;
107 }

109 template <int dim>
110 void
111 ParticleWallContactForce<dim>::
112  update_particle_floating_wall_contact_information(
113  particle_wall_contact_info<dim> &contact_info,
114  const ArrayView<const double> &particle_properties,
115  const double dt,
116  const Tensor<1, 3> &cut_cell_translational_velocity,
117  const Tensor<1, 3> &cut_cell_rotational_velocity,
118  const double center_of_rotation_particle_distance)
119 {
120  // i is the particle, j is the wall.
121  // we need to put a minus sign infront of the normal_vector to respect the
122  // convention (i -> j)
123  const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
124 
125  // Using velocity and angular velocity of particle as
126  // local vectors
127  Tensor<1, 3> particle_velocity;
128  particle_velocity[0] = particle_properties[DEM::PropertiesIndex::v_x];
129  particle_velocity[1] = particle_properties[DEM::PropertiesIndex::v_y];
130  particle_velocity[2] = particle_properties[DEM::PropertiesIndex::v_z];
131 
132 
133  Tensor<1, 3> particle_angular_velocity;
134  particle_angular_velocity[0] =
135  particle_properties[DEM::PropertiesIndex::omega_x];
136  particle_angular_velocity[1] =
137  particle_properties[DEM::PropertiesIndex::omega_y];
138  particle_angular_velocity[2] =
139  particle_properties[DEM::PropertiesIndex::omega_z];
140 
141  // Defining relative contact velocity
142  // v_ij = v_j - v_i
143  Tensor<1, 3> contact_relative_velocity =
144  cut_cell_translational_velocity - particle_velocity +
145  cross_product_3d((center_of_rotation_particle_distance *
146  cut_cell_rotational_velocity -
147  0.5 * particle_properties[DEM::PropertiesIndex::dp] *
148  particle_angular_velocity),
149  normal_vector);
150 
151  // Calculation of normal relative velocity
152  double normal_relative_velocity_value =
153  contact_relative_velocity * normal_vector;
154  Tensor<1, 3> normal_relative_velocity =
155  normal_relative_velocity_value * normal_vector;
156 
157  // Calculation of tangential relative velocity
158  Tensor<1, 3> tangential_relative_velocity =
159  contact_relative_velocity - normal_relative_velocity;
160 
161  // Calculate the new tangential_overlap, since this value is
162  // history-dependent, it needs the value at previous time-step
163  Tensor<1, 3> modified_tangential_overlap =
164  contact_info.tangential_overlap + tangential_relative_velocity * dt;
165 
166  // Updating the contact_info container based on the new calculated values
167  contact_info.normal_relative_velocity = normal_relative_velocity_value;
168  contact_info.tangential_overlap = modified_tangential_overlap;
169  contact_info.tangential_relative_velocity = tangential_relative_velocity;
170 }

172 template <int dim>
173 void
174 ParticleWallContactForce<dim>::calculate_force_and_torque_on_boundary(
175  const unsigned int boundary_id,
176  Tensor<1, 3> add_force,
177  const Point<3> point_contact)
178 {
179  if (calculate_force_torque_on_boundary == true)
180  {
181  mpi_correction_over_calculation_of_forces_and_torques();
182 
183  force_on_walls[boundary_id] = force_on_walls[boundary_id] - add_force;
184 
185  torque_on_walls[boundary_id] =
186  torque_on_walls[boundary_id] -
187  cross_product_3d(point_contact - center_mass_container, add_force);
188  }
189 }

203 template <int dim>
204 void
205 ParticleWallContactForce<
206  dim>::mpi_correction_over_calculation_of_forces_and_torques()
207 {
208  for (const auto &it : boundary_index)
209  {
210  force_on_walls[it] =
211  Utilities::MPI::sum(force_on_walls[it], MPI_COMM_WORLD);
212  torque_on_walls[it] =
213  Utilities::MPI::sum(torque_on_walls[it], MPI_COMM_WORLD);
214  }

385 std::string
386 get_last_value_of_parameter(const std::string &file_name,
387  const std::string &parameter_name)
388 {
389  std::string return_value;
390 
391  std::ifstream x_file(file_name);
392  AssertThrow(x_file.fail() == false, ExcIO());
393 
394  while (x_file)
395  {
396  // Get one line and then match a regex to it that matches the parameter
397  // we are looking for. Before we do that, strip spaces from the front
398  // and back of the line:
399  std::string line;
400  std::getline(x_file, line);
401 
402  while ((line.size() > 0) && (line[0] == ' ' || line[0] == '\t'))
403  line.erase(0, 1);
404  while ((line.size() > 0) &&
405  (line[line.size() - 1] == ' ' || line[line.size() - 1] == '\t'))
406  line.erase(line.size() - 1, std::string::npos);
407 
408  std::match_results<std::string::const_iterator> matches;
409  const std::string regex =
410  "set[ \t]+" + parameter_name + "[ \t]*=[ \t]*(.*)";
411  if (std::regex_match(line, matches, std::regex(regex)))
412  {
413  // Since the line as a whole matched, the 'matches' variable needs to
414  // contain two entries: [0] denotes the whole string, and [1] the
415  // one that was matched by the '(.*)' expression.
416  Assert(matches.size() == 2, dealii::ExcInternalError());
417  return_value = std::string(matches[1].first, matches[1].second);
418  }
419  }
420 
421  return return_value;
422 }

469 unsigned int
470 get_dimension(const std::string &file_name)
471 {
472  const std::string dimension =
473  get_last_value_of_parameter(file_name, "dimension");
474 
475  if (dimension.size() > 0)
476  {
477  // Extracted from ASPECT
478  // A common problem is that people have .prm files that were generated
479  // on Windows, but then run this on Linux where the line endings are
480  // different. This is pernicious because it means that the conversion
481  // of a string such as "2\r" to an integer fails, but if we print
482  // this string, it comes out completely garbled because it contains
483  // a carriage-return without a newline -- so the error message looks
484  // like this:
485  //
486  // >. While reading the dimension from the input file, ASPECT found a
487  // string that can not be converted to an integer: <2
488  //
489  // Note how the end of the error message overwrites the beginning
490  // of the line.
491  //
492  // To avoid this kind of error, specifically test up front that the
493  // text in question does not contain '\r' characters. If we are on
494  // linux, then this kind of character would means that the line endings
495  // are wrong. On the other hand, if we are on windows, then the
496  // getline command we have used in finding 'dimension' would have
497  // filtered it out. So its presence points to a problem.
498 
499  AssertThrow(
500  dimension.find('\r') == std::string::npos,
501  dealii::ExcMessage(
502  "It appears that your input file uses Windows-style "
503  "line endings ('\\r\\n') but you are running on a system where "
504  "the C++ run time environment expects input files to have "
505  "Unix-style line endings ('\\n'). You need to convert your "
506  "input file to use the correct line endings before running "
507  "ASPECT with it."));
508  try
509  {
510  return dealii::Utilities::string_to_int(dimension);
511  }
512  catch (...)
513  {
514  AssertThrow(false,
515  dealii::ExcMessage(
516  "While reading the dimension from the input file, "
517  "Lethe found a string that can not be converted to "
518  "an integer: <" +
519  dimension + ">."));
520  return 0; // we should never get here.
521  }
522  }
523  else
524  {
525  AssertThrow(
526  false,
527  dealii::ExcMessage(
528  "While reading the dimension from the input file, "
529  "Lethe found a value that is neither 2 or 3. Since August 2023, "
530  "Lethe requires that the user explicitly specify the dimension of the problem within the parameter file. This can be achieved by adding set dimension = 2 or set dimension = 3 within the parameter file"));
531  }
532 }

72 /**
73  * @brief Class used to calculate a filtered phase for VOF simulations.
74  * The filtered phase is defined as
75  * \f$\phi_f = 0.5 \tanh(\beta(\phi-0.5)) + 0.5\f$.
76  */
77 class VolumeOfFluidTanhFilter : public VolumeOfFluidFilterBase
78 {
79 public:
80  VolumeOfFluidTanhFilter(const double beta)
81  : beta(beta)
82  {}
83 
84  /**
85  * @brief filter_phase calculates the value of the filtered phase fraction
86  * @param unfiltered_phase Value of the phase fraction before applying the filter
87  * @return Value of the phase fraction after applying the tanh filter
88  */
89  virtual double
90  filter_phase(const double &unfiltered_phase) override
91  {
92  return 0.5 * tanh(beta * (unfiltered_phase - 0.5)) + 0.5;
93  }
94 
95 private:
96  // User-defined parameter that influences the definition of the interface
97  const double beta;
98 };

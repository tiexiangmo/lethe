96  /**
97  * @brief Call for the solution of the physics that should be solved
98  *
99  * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
100  * already solved or not. See the map `solve_pre_fluid` to know which
101  * subphysics are solved before the fluid dynamics.
102  *
103  * @param time_stepping_method Time-Stepping method with which the assembly is called
104  */
105  void
106  solve(const bool fluid_dynamics_has_been_solved,
107  const Parameters::SimulationControl::TimeSteppingMethod
108  time_stepping_method)
109  {
110  // Loop through all the elements in the physics map. Consequently, iphys is
111  // an std::pair where iphys.first is the PhysicsID and iphys.second is the
112  // AuxiliaryPhysics pointer. This is how the map can be traversed
113  // sequentially.
114  for (auto &iphys : physics)
115  {
116  // If iphys.first should be solved BEFORE fluid dynamics
117  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
118  solve_physics(iphys.first, time_stepping_method);
119 
120  // If iphys.first should be solved AFTER fluid dynamics OR if is not
121  // present in solve_pre_fluid map
122  else if (fluid_dynamics_has_been_solved &&
123  (!solve_pre_fluid[iphys.first] ||
124  solve_pre_fluid.count(iphys.first) == 0))
125  solve_physics(iphys.first, time_stepping_method);
126  }
127 
128  for (auto &iphys : block_physics)
129  {
130  // If iphys.first should be solved BEFORE fluid dynamics
131  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
132  solve_block_physics(iphys.first, time_stepping_method);
133 
134  // If iphys.first should be solved AFTER fluid dynamics OR if is not
135  // present in solve_pre_fluid map
136  else if (fluid_dynamics_has_been_solved &&
137  (!solve_pre_fluid[iphys.first] ||
138  solve_pre_fluid.count(iphys.first) == 0))
139  solve_block_physics(iphys.first, time_stepping_method);
140  }
141  }

143  /**
144  * @brief Call for the solution of a single physic
145  *
146  * @param time_stepping_method Time-Stepping method with which the assembly is called
147  */
148  void
149  solve_physics(const PhysicsID physics_id,
150  const Parameters::SimulationControl::TimeSteppingMethod
151  time_stepping_method)
152  {
153  // Announce physic solved (verbosity = non_linear_solver.verbosity)
154  if (verbosity.at(physics_id) != Parameters::Verbosity::quiet)
155  announce_physics(physics_id);
156 
157  AssertThrow(std::find(active_physics.begin(),
158  active_physics.end(),
159  physics_id) != active_physics.end(),
160  ExcInternalError());
161 
162  physics[physics_id]->time_stepping_method = time_stepping_method;
163  physics[physics_id]->solve_non_linear_system(false);
164  physics[physics_id]->modify_solution();
165  }

227  /**
228  * @brief Rearrange vector solution correctly for transient simulations for
229  * all auxiliary physics.
230  *
231  * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
232  * already solved or not. See the map `solve_pre_fluid` to know which
233  * subphysics are solved before the fluid dynamics.
234  */
235  void
236  percolate_time_vectors(const bool fluid_dynamics_has_been_solved)
237  {
238  // Loop through all the elements in the physics map. Consequently, iphys is
239  // an std::pair where iphys.first is the PhysicsID and iphys.second is the
240  // AuxiliaryPhysics pointer. This is how the map can be traversed
241  // sequentially.
242  for (auto &iphys : physics)
243  {
244  // If iphys.first should be percolated BEFORE fluid dynamics is solved
245  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
246  iphys.second->percolate_time_vectors();
247 
248  // If iphys.first should be percolated AFTER fluid dynamics is solved OR
249  // if is not present in solve_pre_fluid map
250  else if (fluid_dynamics_has_been_solved &&
251  (!solve_pre_fluid[iphys.first] ||
252  solve_pre_fluid.count(iphys.first) == 0))
253  iphys.second->percolate_time_vectors();
254  }
255  for (auto &iphys : block_physics)
256  {
257  // If iphys.first should be percolated BEFORE fluid dynamics is solved
258  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
259  iphys.second->percolate_time_vectors();
260 
261  // If iphys.first should be percolated AFTER fluid dynamics is solved OR
262  // if is not present in solve_pre_fluid map
263  else if (fluid_dynamics_has_been_solved &&
264  (!solve_pre_fluid[iphys.first] ||
265  solve_pre_fluid.count(iphys.first) == 0))
266  iphys.second->percolate_time_vectors();
267  }

127 template <typename VectorType>
128 PhysicsSolver<VectorType>::PhysicsSolver(
129  const Parameters::NonLinearSolver non_linear_solver_parameters)
130  : pcout(std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
131 {
132  switch (non_linear_solver_parameters.solver)
133  {
134  case Parameters::NonLinearSolver::SolverType::newton:
135  non_linear_solver =
136  new NewtonNonLinearSolver<VectorType>(this,
137  non_linear_solver_parameters);
138  break;
139  case Parameters::NonLinearSolver::SolverType::kinsol_newton:
140  non_linear_solver = new KinsolNewtonNonLinearSolver<VectorType>(
141  this, non_linear_solver_parameters);
142  break;
143  case Parameters::NonLinearSolver::SolverType::inexact_newton:
144  non_linear_solver = new InexactNewtonNonLinearSolver<VectorType>(
145  this, non_linear_solver_parameters);
146  break;
147  default:
148  break;
149  }
150 }

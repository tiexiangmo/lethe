118 /**
119  * @brief This model takes into account the phase change of a material
120  * by considering the latent heat into the specific heat over a phase change
121  * interval determined by [T_solidus,T_liquidus].
122  */
123 class PhaseChangeSpecificHeat : public SpecificHeatModel
124 {
125 public:
126  /**
127  * @brief Default constructor
128  */
129  PhaseChangeSpecificHeat(
130  const Parameters::PhaseChange p_phase_change_parameters)
131  : param(p_phase_change_parameters)
132  {
133  this->model_depends_on[field::temperature] = true;
134  this->model_depends_on[field::previous_temperature] = true;
135  }
136 
137  /**
138  * @brief value Calculates the value of the phase change specific heat.
139  * @param fields_value Value of the various field on which the specific heat depends.
140  * @return value of the specific heat. */
141  double
142  value(const std::map<field, double> &fields_value) override
143  {
144  const double temperature = fields_value.at(field::temperature);
145  const double previous_temperature =
146  fields_value.at(field::previous_temperature);
147  if (temperature > previous_temperature)
148  {
149  const double dT = std::max(temperature - previous_temperature, 1e-6);
150  return (enthalpy(temperature) - enthalpy(temperature - dT)) / dT;
151  }
152  else
153  {
154  const double dT = std::max(previous_temperature - temperature, 1e-6);
155  return (enthalpy(temperature + dT) - enthalpy(temperature)) / dT;
156  }
157  }

160  /**
161  * @brief vector_value Calculates the vector of specific heat.
162  * @param field_vectors Vector of fields on which the specific heat may depend.
163  * @param property_vector Vector of specific_heat values.
164  */
165  void
166  vector_value(const std::map<field, std::vector<double>> &field_vectors,
167  std::vector<double> &property_vector) override
168  {
169  const std::vector<double> &temperature_vec =
170  field_vectors.at(field::temperature);
171  const std::vector<double> &previous_temperature_vec =
172  field_vectors.at(field::previous_temperature);
173 
174  const unsigned int n_values = temperature_vec.size();
175 
176  Assert(n_values == previous_temperature_vec.size(),
177  SizeOfFields(n_values, previous_temperature_vec.size()));
178  Assert(n_values == property_vector.size(),
179  SizeOfFields(n_values, property_vector.size()));
180 
181  for (unsigned int i = 0; i < n_values; ++i)
182  {
183  const double temperature = temperature_vec[i];
184  const double previous_temperature = previous_temperature_vec[i];
185  if (temperature > previous_temperature)
186  {
187  const double dT =
188  std::max(temperature - previous_temperature, 1e-6);
189  property_vector[i] =
190  (enthalpy(temperature) - enthalpy(temperature - dT)) / dT;
191  }
192  else
193  {
194  const double dT =
195  std::max(previous_temperature - temperature, 1e-6);
196  property_vector[i] =
197  (enthalpy(temperature + dT) - enthalpy(temperature)) / dT;
198  }
199  }
200  }

232  /**
233  * @brief enthalpy Calculates the enthalpy of a phase change material for a temperature T
234  * The enthalpy is defined as :
235  *
236  * Pure liquid
237  * ------------
238  * if (T>T_liquidus) : H = cp_solid * T_solidus + 0.5*(cp_solid+cp_liquid) *
239  * (T_liquidus-T_solidus) + latent_enthalpy + cp_liquid * (T-T_liquidus)
240  *
241  * Liquid-solid mixture
242  * -----------------
243  * else if (T>T_solidus) : cp_solid * T_solidus + 0.5*(cp_solid+cp_liquid) *
244  * (T-T_solidus) + liquid_fraction * latent_enthalpy
245  *
246  * Pure solid
247  * ------------
248  * else : cp_solid * T
249  *
250  * @param T temperature at which to calculate the enthalpy
251  * @return Value of the enthalpy
252  */
253 
254  inline double
255  enthalpy(const double T)
256  {
257  if (T > param.T_liquidus)
258  return (param.cp_s * param.T_solidus +
259  0.5 * (param.cp_l + param.cp_s) *
260  (param.T_liquidus - param.T_solidus) +
261  param.latent_enthalpy + (T - param.T_liquidus) * param.cp_l);
262  else if (T > param.T_solidus)
263  {
264  liquid_fraction = calculate_liquid_fraction(T, param);
265  return (param.cp_s * param.T_solidus +
266  0.5 * (param.cp_l + param.cp_s) * (T - param.T_solidus) +
267  param.latent_enthalpy * liquid_fraction);
268  }
269  else
270  return param.cp_s * T;
271  }
272 
273 private:
274  const Parameters::PhaseChange param;
275  double liquid_fraction;
276 };

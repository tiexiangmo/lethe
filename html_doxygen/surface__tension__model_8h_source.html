140 /**
141  * @brief Linear surface tension. The surface tension is given by:
142  * sigma_0 + dsimga/dT * (T-T_0).
143  */
144 class SurfaceTensionLinear : public SurfaceTensionModel
145 {
146 public:
147  /**
148  * @brief Default constructor
149  * @param p_surface_tension_parameters The parameters for the surface tension
150  * model, including the surface tension coefficient sigma_0 and surface
151  * tension gradient dsimga/dT at the reference temperature T_0
152  */
153  SurfaceTensionLinear(
154  const Parameters::SurfaceTensionParameters &p_surface_tension_parameters)
155  : surface_tension_coefficient(
156  p_surface_tension_parameters.surface_tension_coefficient)
157  , T_0(p_surface_tension_parameters.T_0)
158  , surface_tension_gradient(
159  p_surface_tension_parameters.surface_tension_gradient)
160  {
161  this->model_depends_on[field::temperature] = true;
162  }
163 
164  /**
165  * @brief value Calculates the surface tension coefficient
166  * @param field_vectors Vectors of the fields on which the surface tension
167  * depends. In this case, it depends on the temperature.
168  * @return value of the physical property calculated with the fields_value
169  */
170  double
171  value(const std::map<field, double> &fields_value) override
172  {
173  const double temperature = fields_value.at(field::temperature);
174  return surface_tension_coefficient +
175  surface_tension_gradient * (temperature - T_0);
176  }
177 
178  /**
179  * @brief vector_value Calculates the vector of surface tension coefficient.
180  * @param field_vectors Vectors of the fields on which the surface tension
181  * depends. In this case, it depends on the temperature.
182  * @param property_vector Vectors of the surface tension coefficient values
183  */
184  void
185  vector_value(const std::map<field, std::vector<double>> &field_vectors,
186  std::vector<double> &property_vector) override
187  {
188  const std::vector<double> &temperature =
189  field_vectors.at(field::temperature);
190  for (unsigned int i = 0; i < property_vector.size(); ++i)
191  property_vector[i] = surface_tension_coefficient +
192  surface_tension_gradient * (temperature[i] - T_0);
193  }
194 
195  /**
196  * @brief jacobian Calculates the jacobian (the partial derivative) of the
197  * surface tension coefficient with respect to a field
198  * @param field_values Value of the various fields on which the property may
199  * depend.
200  * @param id Indicator of the field with respect to which the jacobian
201  * should be calculated.
202  * @return value of the partial derivative of the surface tension coefficient
203  * with respect to the field.
204  */
205  double
206  jacobian(const std::map<field, double> & /*field_values*/, field id) override
207  {
208  if (id == field::temperature)
209  return surface_tension_gradient;
210  else
211  return 0;
212  }
213 
214  /**
215  * @brief vector_jacobian Calculates the derivative of the surface tension
216  * coefficient with respect to a field.
217  * @param field_vectors Vector for the values of the fields used to evaluate
218  * the property.
219  * @param id Identifier of the field with respect to which a derivative should
220  * be calculated.
221  * @param jacobian vector of the value of the derivative of the surface
222  * tension coefficient with respect to the field id.
223  */
224  void
225  vector_jacobian(
226  const std::map<field, std::vector<double>> & /*field_vectors*/,
227  const field id,
228  std::vector<double> &jacobian_vector) override
229  {
230  if (id == field::temperature)
231  std::fill(jacobian_vector.begin(),
232  jacobian_vector.end(),
233  surface_tension_gradient);

235  std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0);
236  }
237 
238 protected:
239  const double surface_tension_coefficient;
240  const double T_0;
241  const double surface_tension_gradient;

245  * @brief Linear surface tension with phase change. The surface tension is given by:
246  * alpha_l*(sigma_0 + dsigma/dT * (T-T_0)).

248 class SurfaceTensionPhaseChange : public SurfaceTensionLinear
249 {
250 public:
251  /**
252  * @brief Default constructor
253  * @param p_surface_tension_parameters The parameters for the surface tension
254  * model, including the surface tension coefficient sigma_0 and surface
255  * tension gradient dsimga/dT at the reference temperature T_0, and the
256  * solidus and liquidus temperatures T_solidus and T_liquidus respectively.
257  */
258  SurfaceTensionPhaseChange(
259  const Parameters::SurfaceTensionParameters &p_surface_tension_parameters)
260  : SurfaceTensionLinear(p_surface_tension_parameters)
261  , T_solidus(p_surface_tension_parameters.T_solidus)
262  , T_liquidus(p_surface_tension_parameters.T_liquidus)
263  {
264  this->model_depends_on[field::temperature] = true;
265  }
266 
267  /**
268  * @brief value Calculates the surface tension coefficient.
269  * @param field_vectors Vectors of the fields on which the surface tension
270  * depends. In this case, it depends on the temperature.
271  * @return value of the physical property calculated with the fields_value.
272  */
273  double
274  value(const std::map<field, double> &fields_value) override
275  {
276  const double temperature = fields_value.at(field::temperature);
277 
278  double surface_tension;
279 
280  if (temperature < T_solidus)
281  surface_tension = 0.0;
282  else if (temperature > T_liquidus)
283  surface_tension = surface_tension_coefficient +
284  surface_tension_gradient * (temperature - T_0);
285  else
286  {
287  const double l_frac =
288  calculate_liquid_fraction(temperature, T_solidus, T_liquidus);
289 
290  surface_tension = (surface_tension_coefficient +
291  surface_tension_gradient * (temperature - T_0)) *
292  l_frac;
293  }
294  return surface_tension;
295  }

302  */

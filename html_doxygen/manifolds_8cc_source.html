259 attach_manifolds_to_triangulation(
260  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
261  Parameters::Manifolds manifolds)
262 {
263  for (unsigned int i = 0; i < manifolds.types.size(); ++i)
264  {
265  if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::spherical)
266  {
267  Point<spacedim> circleCenter;
268  if constexpr (spacedim == 2)
269  circleCenter =
270  Point<spacedim>(manifolds.arg1[i], manifolds.arg2[i]);
271  else
272  circleCenter = Point<spacedim>(manifolds.arg1[i],
273  manifolds.arg2[i],
274  manifolds.arg3[i]);
275  static const SphericalManifold<dim, spacedim> manifold_description(
276  circleCenter);
277  triangulation.set_manifold(manifolds.id[i], manifold_description);
278  }
279  else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::iges)
280  {
281  attach_cad_to_manifold(triangulation,
282  manifolds.cad_files[i],
283  manifolds.id[i]);
284  }
285  else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::none)
286  {
287  }
288  else
289  throw std::runtime_error("Unsupported manifolds type");
290  }
291 }

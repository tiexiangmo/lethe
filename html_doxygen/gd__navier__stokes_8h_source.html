284 template <class BSPreconditioner>
285 void
286 BlockSchurPreconditioner<BSPreconditioner>::vmult(
287  TrilinosWrappers::MPI::BlockVector &dst,
288  const TrilinosWrappers::MPI::BlockVector &src) const
289 {
290  // TimerOutput computing_timer(std::cout,
291  // TimerOutput::summary,
292  // TimerOutput::wall_times);
293 
294  TrilinosWrappers::MPI::Vector utmp(src.block(0));
295  {
296  // computing_timer.enter_section("Pressure");
297  SolverControl solver_control(
298  linear_solver_parameters.max_iterations,
299  std::max(
300  1e-3 * src.block(1).l2_norm(),
301  // linear_solver_parameters.relative_residual*src.block(0).l2_norm(),
302  linear_solver_parameters.minimum_residual));
303  TrilinosWrappers::SolverCG cg(solver_control);
304 
305  dst.block(1) = 0.0;
306  cg.solve(pressure_mass_matrix,
307  dst.block(1),
308  src.block(1),
309  *pmass_preconditioner);
310  dst.block(1) *= -(kinematic_viscosity + gamma);
311  // computing_timer.exit_section("Pressure");
312  }
313 
314  {
315  // computing_timer.enter_section("Operations");
316  stokes_matrix.block(0, 1).vmult(utmp, dst.block(1));
317  utmp *= -1.0;
318  utmp += src.block(0);
319  // computing_timer.exit_section("Operations");
320  }
321  {
322  // computing_timer.enter_section("A Matrix");
323  SolverControl solver_control(
324  linear_solver_parameters.max_iterations,
325  std::max(1e-1 * src.block(0).l2_norm(),
326  linear_solver_parameters.minimum_residual));
327 
328 
329 
330  // TrilinosWrappers::SolverBicgstab solver(solver_control);
331  TrilinosWrappers::SolverGMRES solver(solver_control);
332 
333  // A_inverse.solve(stokes_matrix.block(0, 0),dst.block(0), utmp,
334  // mp_preconditioner);
335  solver.solve(stokes_matrix.block(0, 0),
336  dst.block(0),
337  utmp,
338  *amat_preconditioner);
339  // computing_timer.exit_section("A Matrix");
340  }
341 }

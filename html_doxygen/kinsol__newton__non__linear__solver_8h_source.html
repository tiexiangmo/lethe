63 template <typename VectorType>
64 void
65 KinsolNewtonNonLinearSolver<VectorType>::solve(const bool is_initial_step)
66 {
67 #ifdef DEAL_II_WITH_SUNDIALS
68  bool first_step = is_initial_step;
69 
70  PhysicsSolver<VectorType> *solver = this->physics_solver;
71 
72  auto &local_evaluation_point = solver->get_local_evaluation_point();
73 
74  VectorType &evaluation_point = solver->get_evaluation_point();
75  VectorType &present_solution = solver->get_present_solution();
76 
77  typename SUNDIALS::KINSOL<VectorType>::AdditionalData additional_data;
78  additional_data.function_tolerance = this->params.tolerance;
79  additional_data.maximum_non_linear_iterations = this->params.max_iterations;
80  additional_data.step_tolerance = this->params.tolerance;
81 
82  // Kinsol solution strategy, the default one is line search
83  if (this->params.kinsol_strategy ==
84  Parameters::NonLinearSolver::KinsolStrategy::normal_newton)
85  {
86  additional_data.strategy =
87  SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::newton;
88  }
89  else if (this->params.kinsol_strategy ==
90  Parameters::NonLinearSolver::KinsolStrategy::line_search)
91  {
92  additional_data.strategy = SUNDIALS::KINSOL<
93  VectorType>::AdditionalData::SolutionStrategy::linesearch;
94  }
95  else if (this->params.kinsol_strategy ==
96  Parameters::NonLinearSolver::KinsolStrategy::picard)
97  {
98  additional_data.strategy =
99  SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::picard;
100  }
101 
102  SUNDIALS::KINSOL<VectorType> nonlinear_solver(additional_data);
103 
104  nonlinear_solver.reinit_vector = [&](VectorType &x) {
105  x.reinit(local_evaluation_point);
106  };
107 
108  nonlinear_solver.residual = [&](const VectorType &evaluation_point_for_kinsol,
109  VectorType &residual) {
110  solver->pcout << "Computing residual vector..." << std::endl;
111  evaluation_point = evaluation_point_for_kinsol;
112  solver->apply_constraints();
113  solver->assemble_system_rhs();
114  auto &current_residual = solver->get_system_rhs();
115  residual = current_residual;
116  solver->pcout << " -Residual: " << residual.l2_norm() << std::endl;
117  return 0;
118  };
119 
120  nonlinear_solver.setup_jacobian =
121  [&](const VectorType &present_solution_for_kinsol,
122  const VectorType & /*current_f*/) {
123  solver->pcout << "Computing jacobian matrix..." << std::endl;
124  evaluation_point = present_solution_for_kinsol;
125  solver->assemble_system_matrix();
126  return 0;
127  };
128 
129  nonlinear_solver.solve_with_jacobian = [&](const VectorType & /* residual */,
130  VectorType &dst,
131  const double /* tolerance */) {
132  solver->pcout << "Solving linear system..." << std::endl;
133  solver->solve_linear_system(first_step);
134  dst = solver->get_newton_update();
135  return 0;
136  };
137 
138  local_evaluation_point = present_solution;
139 
140  nonlinear_solver.solve(local_evaluation_point);
141  present_solution = local_evaluation_point;
142 #else
143  (void)is_initial_step;
144  throw std::runtime_error(
145  "Kinsol newton nonlinear solver requires DEAL_II to be compiled with SUNDIALS");
146 #endif // DEAL_II_WITH_SUNDIALS
147 }
148
